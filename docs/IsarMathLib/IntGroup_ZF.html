section ‚ÄπInteger powers of group elements‚Ä∫

theory IntGroup_ZF imports Group_ZF_2 Int_ZF_1 Cardinal_ZF

begin

text‚ÄπIn the ‚ÄπMonoid_ZF_1‚Ä∫ theory we consider multiplicities of $n\cdot x$ of monoid elements, i.e.
  special cases of expressions of the form $x_1\oplus x_2\oplus\dots\oplus x_n$ where $x_i=x$ 
  for $i=1..n$.
  In the group context where we usually use multiplicative notation this translates to the "power"
  $x^n$ where $n\in \mathbb{N}$, see also section "Product of a list of group elements" in 
  the ‚ÄπGroup_ZF‚Ä∫ theory. In the group setting the notion of raising an element to natural power can
  be naturally generalized to the notion of raising an element to an integer power. ‚Ä∫

subsection‚ÄπProperties of natural powers of an element and its inverse‚Ä∫

text‚ÄπThe integer power is defined in terms of natural powers of an element and its inverse. 
  In this section we study properties of expressions $(x^n)\cdot(x^{-1})^k$, where $x$ is a group
  element and $n,k$ are natural numbers. ‚Ä∫

text‚ÄπThe natural power of an element's inverse is the inverse of the power of the element.‚Ä∫

lemma (in group0) nat_pow_inverse: assumes "n‚àànat" "x‚ààG"
  shows "pow(n,x¬Ø) = (pow(n,x))¬Ø"
proof -
  from assms(1) have "n‚àànat" and "pow(0,x¬Ø) = (pow(0,x))¬Ø"
    using monoid.nat_mult_zero group_inv_of_one by auto
  moreover from assms(2) have 
    "‚àÄk‚àànat. pow(k,x¬Ø) = (pow(k,x))¬Ø ‚ü∂ pow(k #+ 1,x¬Ø) = (pow(k #+ 1,x))¬Ø"
    using nat_pow_add_one inverse_in_group group_inv_of_two monoid.nat_mult_type
      by simp
  ultimately show ?thesis by (rule ind_on_nat1)
qed

text‚ÄπThe natural power of $x$ multiplied by the same power of $x^{-1}$ cancel out to give
  the neutral element of the group.‚Ä∫

corollary (in group0) nat_pow_inv_cancel: assumes "n‚àànat" "x‚ààG"
  shows "pow(n,x)‚ãÖpow(n,x¬Ø) = ùü≠" "pow(n,x¬Ø)‚ãÖpow(n,x) = ùü≠"
  using nat_pow_inverse assms group0_2_L6 monoid.nat_mult_type assms by auto

text‚ÄπIf $k\leq n$ are natural numbers and $x$ an element of the group, then
  $x^n\cdot (x^{-1})^k = x^{n-k}$. ‚Ä∫

lemma (in group0) nat_pow_cancel_less: assumes "n‚àànat" "k‚â§n" "x‚ààG"
  shows "pow(n,x)‚ãÖpow(k,x¬Ø) = pow(n #- k,x)"
proof -
  from assms have "k‚àànat" "n #- k ‚àà nat" and 
    "pow((n #- k),x)‚ààG" "pow(k,x)‚ààG" "pow(k,x¬Ø)‚ààG"
    using leq_nat_is_nat diff_type monoid.nat_mult_type 
      inverse_in_group by simp_all
  from assms(3) ‚Äπk‚àànat‚Ä∫ ‚Äπn #- k ‚àà nat‚Ä∫ have 
    "pow((n #- k) #+ k,x) = pow((n #- k),x)‚ãÖpow(k,x)"
    using monoid.nat_mult_add by simp
  with assms(1,2) ‚Äπpow((n #- k),x)‚ààG‚Ä∫ ‚Äπpow(k,x)‚ààG‚Ä∫ ‚Äπpow(k,x¬Ø)‚ààG‚Ä∫ 
    have "pow(n,x)‚ãÖpow(k,x¬Ø) = pow(n #- k,x)‚ãÖ(pow(k,x)‚ãÖpow(k,x¬Ø))"
      using add_diff_inverse2 group_oper_assoc by simp
  with assms(3) ‚Äπk‚àànat‚Ä∫ ‚Äπpow((n #- k),x) ‚àà G‚Ä∫ show ?thesis
    using nat_pow_inv_cancel(1) group0_2_L2 by simp 
qed

text‚ÄπIf $k\leq n$ are natural numbers and $x$ an element of the group, then
  $(x^{-1})^n\cdot x^k = (x^{-1})^(k-n)$. ‚Ä∫

lemma (in group0) nat_pow_cancel_less1: assumes "n‚àànat" "k‚â§n" "x‚ààG"
  shows "pow(n,x¬Ø)‚ãÖpow(k,x) = pow(n #- k,x¬Ø)"
proof -
  from assms have "pow(n,x¬Ø)‚ãÖpow(k,(x¬Ø)¬Ø) = pow(n #- k,x¬Ø)"
    using inverse_in_group nat_pow_cancel_less by simp
  with assms(3) show ?thesis using group_inv_of_inv by simp
qed

text‚ÄπIf $k\leq n$ are natural numbers and $x$ an element of the group, then
  $x^k\cdot (x^{-1})^n = (x^{-1})^(k-n)$. ‚Ä∫

lemma (in group0) nat_pow_cancel_more: assumes "n‚àànat" "k‚â§n" "x‚ààG"
  shows "pow(k,x¬Ø)‚ãÖpow(n,x) = pow(n #- k,x)"
proof -
  from assms have "k‚àànat" "n #- k ‚àà nat" and 
    "pow((n #- k),x) ‚àà G" "pow(k,x) ‚àà G" "pow(k,x¬Ø) ‚àà G"
    using leq_nat_is_nat diff_type monoid.nat_mult_type inverse_in_group 
      by simp_all
  from assms(3) ‚Äπk‚àànat‚Ä∫ ‚Äπn #- k ‚àà nat‚Ä∫ have 
    "pow(k #+ (n #- k),x) = pow(k,x)‚ãÖpow((n #- k),x)"
    using monoid.nat_mult_add by simp
  with assms(1,2) ‚Äπpow((n #- k),x)‚ààG‚Ä∫ ‚Äπpow(k,x)‚ààG‚Ä∫ ‚Äπpow(k,x¬Ø)‚ààG‚Ä∫ 
  have "pow(k,x¬Ø)‚ãÖpow(n,x) = (pow(k,x¬Ø)‚ãÖpow(k,x))‚ãÖpow(n #- k,x)"
    using add_diff_inverse group_oper_assoc by simp
  with assms(3) ‚Äπk‚àànat‚Ä∫ ‚Äπpow((n #- k),x) ‚àà G‚Ä∫ show ?thesis
    using nat_pow_inv_cancel(2) group0_2_L2 by simp  
qed

text‚ÄπIf $k\leq n$ are natural numbers and $x$ an element of the group, then
  $x^k\cdot (x^{-1})^n = x^{n-k)}$. ‚Ä∫

lemma (in group0) nat_pow_cancel_more1: assumes "n‚àànat" "k‚â§n" "x‚ààG"
  shows "pow(k,x)‚ãÖpow(n,x¬Ø) = pow(n #- k,x¬Ø)"
proof -
  from assms have "pow(k,(x¬Ø)¬Ø)‚ãÖpow(n,x¬Ø) = pow(n #- k,x¬Ø)"
    using inverse_in_group nat_pow_cancel_more by simp
  with assms(3) show ?thesis using group_inv_of_inv by simp
qed

text‚ÄπThe power to a product is the power of the power.‚Ä∫

lemma (in group0) nat_pow_mult:
  assumes "z‚á©1‚àànat" "z‚á©2‚àànat" "g‚ààG"
  shows "pow(z‚á©1 #* z‚á©2,g) = pow(z‚á©2,pow(z‚á©1,g))"
proof -
  have B: "pow(z‚á©1 #* 0,g) = pow(0,pow(z‚á©1,g))" by simp
  { fix k assume "k‚àànat" and I: "pow(z‚á©1 #* k,g) = pow(k,pow(z‚á©1,g))" 
    have "z‚á©1 #* (k #+ 1) = (z‚á©1 #* k) #+ (z‚á©1 #* 1)"
      by (rule add_mult_distrib_left)
    then have "pow(z‚á©1 #* (k #+ 1),g) = pow((z‚á©1 #* k) #+ (z‚á©1 #* 1),g)"
      by (rule same_constr)
    with assms(1,3) ‚Äπk‚àànat‚Ä∫ I have "pow(z‚á©1 #* (k #+ 1),g)=pow(k #+ 1,pow(z‚á©1,g))"
      using mult_1_right mult_type nat_pow_sum_exps 
        nat_pow_type nat_pow_add_one by simp 
  } hence "‚àÄk‚àànat. pow(z‚á©1 #* k,g) = pow(k,pow(z‚á©1,g)) ‚ü∂ 
            pow(z‚á©1 #* (k #+ 1),g) = pow(k #+ 1,pow(z‚á©1,g))"
    by simp
  with assms(2) B show ?thesis by (rule ind_on_nat1)
qed

text‚ÄπAn application of pigeonhole principle to finite groups: if a group is finite
  then every element has a non-zero power that is equal to the neutral element.‚Ä∫

lemma (in group0) finite_fin_order: assumes "Finite(G)" "x‚ààG"
  shows "‚àÉn‚àànat‚àñ{0}. pow(n,x) = ùü≠"
proof -
  from assms(1) obtain k where "k‚àànat" and "G‚âàk"
    unfolding Finite_def by auto
  let ?m = "k #+ 1"
  let ?b = "{‚ü®i,pow(i,x)‚ü©. i‚àà?m}"
  from ‚Äπk‚àànat‚Ä∫ ‚ÄπG‚âàk‚Ä∫ have "?m‚àànat" and "G‚â∫?m"
    using nat_less_add_one nat_into_Card lt_Card_imp_lesspoll 
      eq_lesspoll_trans by simp_all
  from assms(2) ‚Äπ?m‚àànat‚Ä∫ have "‚àÄi‚àà?m. pow(i,x)‚ààG"
    using elem_nat_is_nat(2) nat_pow_type by blast
  then have "?b:?m‚ÜíG" by (rule ZF_fun_from_total)
  with ‚Äπ?m‚àànat‚Ä∫ ‚ÄπG‚â∫?m‚Ä∫ obtain i j where 
    "i‚àà?m" "j‚àà?m" "i < j" "?b`(i) = ?b`(j)"
    using pigeonhole_list by blast
  from ‚Äπ?m‚àànat‚Ä∫ ‚Äπi‚àà?m‚Ä∫ ‚Äπj‚àà?m‚Ä∫ ‚Äπi < j‚Ä∫ have "i‚àànat" "j‚àànat" "i‚â§j"
    using elem_nat_is_nat(2) nat_into_Ord le_iff by auto
  from ‚Äπi‚àà?m‚Ä∫ have "?b`(i) = pow(i,x)" by (rule ZF_fun_from_tot_val1)
  from ‚Äπj‚àà?m‚Ä∫ have "?b`(j) = pow(j,x)" by (rule ZF_fun_from_tot_val1)
  with ‚Äπ?b`(i) = ?b`(j)‚Ä∫ ‚Äπ?b`(i) = pow(i,x)‚Ä∫ have "pow(j,x)‚ãÖpow(i,x¬Ø) = pow(i,x)‚ãÖpow(i,x¬Ø)" 
    by simp
  with assms(2) ‚Äπi‚àànat‚Ä∫ ‚Äπj‚àànat‚Ä∫ ‚Äπi‚â§j‚Ä∫ have "pow(j #- i,x) = ùü≠"
    using nat_pow_cancel_less nat_pow_inv_cancel by simp
  from ‚Äπi‚àànat‚Ä∫ ‚Äπj‚àànat‚Ä∫ ‚Äπi < j‚Ä∫ have "j #- i ‚àà nat" and "0 < j #- i"
    using zero_less_diff by simp_all
  with ‚Äπpow(j #- i,x) = ùü≠‚Ä∫ show ?thesis using lt_not_refl by auto
qed

subsection‚ÄπInteger powers‚Ä∫

text‚ÄπIn this section we introduce notation basic properties of integer power in group context.
  The goal is to show that the power homomorphism property: if $x$ is an element of the group
  and $n,m$ are integers then $x^{n+m}=x^n\cdot x^m$. ‚Ä∫

text‚ÄπWe extend the ‚Äπgroup0‚Ä∫ context with some notation from ‚Äπint0‚Ä∫ context.
  Since we inherit the multiplicative notation from the ‚Äπgroup0‚Ä∫ context
  the integer "one" is denoted ‚Äπùü≠‚á©Z‚Ä∫ rather than just ‚Äπùü≠‚Ä∫ (which is the group's neutral element). ‚Ä∫

locale group_int0 = group0 +

  fixes ints ("‚Ñ§")
  defines ints_def [simp]: "‚Ñ§ ‚â° int"

  fixes ia (infixl "\<ra>" 69)
  defines ia_def [simp]: "a\<ra>b ‚â° IntegerAddition`‚ü® a,b‚ü©"

  fixes iminus ("\<rm> _" 72)
  defines rminus_def [simp]: "\<rm>a ‚â° GroupInv(‚Ñ§,IntegerAddition)`(a)"

  fixes isub (infixl "\<rs>" 69)
  defines isub_def [simp]: "a\<rs>b ‚â° a\<ra> (\<rm> b)"
  
  fixes imul (infixl "‚ãÖ‚á©Z" 69)
  defines imul_def [simp]: "a‚ãÖ‚á©Zb ‚â° IntegerMultiplication`‚ü®a,b‚ü©"

  fixes setneg ("\<sm> _" 72)
  defines setneg_def [simp]: "\<sm>A ‚â° GroupInv(‚Ñ§,IntegerAddition)``(A)"

  fixes izero ("ùü¨")
  defines izero_def [simp]: "ùü¨ ‚â° TheNeutralElement(‚Ñ§,IntegerAddition)"

  fixes ione ("ùü≠‚á©Z")
  defines ione_def [simp]: "ùü≠‚á©Z ‚â° TheNeutralElement(‚Ñ§,IntegerMultiplication)"
  
  fixes itwo ("ùüÆ")
  defines itwo_def [simp]: "ùüÆ ‚â° ùü≠‚á©Z\<ra>ùü≠‚á©Z"
 
  fixes ithree ("ùüØ")
  defines ithree_def [simp]: "ùüØ ‚â° ùüÆ\<ra>ùü≠‚á©Z"
  
  fixes nonnegative ("‚Ñ§‚áß+")
  defines nonnegative_def [simp]: 
  "‚Ñ§‚áß+ ‚â° Nonnegative(‚Ñ§,IntegerAddition,IntegerOrder)"

  fixes positive ("‚Ñ§‚á©+")
  defines positive_def [simp]:

  fixes abs

  defines sless_def [simp]: "a \<ls> b ‚â° a\<lsq>b ‚àß a‚â†b"

  fixes interval (infix ".." 70)
  defines interval_def [simp]: "m..n ‚â° Interval(IntegerOrder,m,n)"

  fixes maxf
  defines maxf_def [simp]: "maxf(f,A) ‚â° Maximum(IntegerOrder,f``(A))"

  defines minf_def [simp]: "minf(f,A) ‚â° Minimum(IntegerOrder,f``(A))"

  defines oddext_def [simp]: "f¬∞ ‚â° OddExtension(‚Ñ§,IntegerAddition,IntegerOrder,f)"

text‚ÄπNext define notation for the integer power ‚Äπpowz(z,x)‚Ä∫. 
  The difficulty here is that in ZF set theory nonnegative integers and natural numbers are 
  different things. So, we use the notion of ‚Äπzmagnitude‚Ä∫ defined in the standard Isabelle/ZF 
  ‚ÄπInt‚Ä∫ theory. For an integer number $z$, ‚Äπzmagnitude(z)‚Ä∫ is like absolute value of $z$ 
  but interpreted as a natural number. 
  Hence, we define the integer power ‚Äπpowz(z,x)‚Ä∫ as $x$ raised to the magnitude of $z$ if $z$ is 
  nonnegative or $x^{-1}$ raised to the same natural power otherwise.‚Ä∫

definition (in group_int0) powz where
  "powz(z,x) ‚â° pow(zmagnitude(z),if ùü¨\<lsq>z then x else x¬Ø)"

text‚ÄπWe bring in all the results about integers in ‚Äπint0‚Ä∫ with the notation included in ‚Äπgroup_int0‚Ä∫.‚Ä∫

sublocale group_int0 < ints:int0 "‚Ñ§" ia iminus isub imul setneg izero ione itwo 
  ithree nonnegative positive abs lesseq sless interval maxf minf oddext
by auto

text‚ÄπAn integer power of a group element is in the group.‚Ä∫

lemma (in group_int0) powz_type: assumes "z‚àà‚Ñ§" "x‚ààG" shows "powz(z,x) ‚àà G"
  using assms zmagnitude_type monoid.nat_mult_type inverse_in_group 
  unfolding powz_def by simp

text‚ÄπA group element raised to (integer) zero'th power is equal to the group's neutral element. 
  An element raised to (integer) power one is the same element. ‚Ä∫

lemma (in group_int0) int_power_zero_one: assumes "x‚ààG"
  shows "powz(ùü¨,x) = ùü≠" and "powz(ùü≠‚á©Z,x) = x"
  using assms ints.Int_ZF_1_L8A(1) ints.int_ord_is_refl1 ints.zmag_zero_one 
    monoid.nat_mult_zero ints.Int_ZF_2_L16A(1) monoid.nat_mult_one 
    unfolding powz_def by auto

text‚ÄπThe neutral raised to any (integer) power is equal to the group's neutral element.‚Ä∫

lemma (in group_int0) int_power_neutral: assumes "z‚àà‚Ñ§"
  shows "powz(z,ùü≠) = ùü≠"
  using monoid.nat_mult_neutral group_inv_of_one zmagnitude_type
    unfolding powz_def by auto

text‚ÄπIf $x$ is an element of the group and $z_1,z_2$ are nonnegative integers then
  $x^{z_1+z_2}=x^{z_1}\cdot x^{z_2}$, i.e. the power homomorphism property holds.‚Ä∫

lemma (in group_int0) powz_hom_nneg_nneg: assumes "ùü¨\<lsq>z‚á©1" "ùü¨\<lsq>z‚á©2" "x‚ààG"
  shows "powz(z‚á©1\<ra>z‚á©2,x) = powz(z‚á©1,x)‚ãÖpowz(z‚á©2,x)"
  using assms ints.add_nonneg_ints(1,2) monoid.nat_mult_add 
  unfolding powz_def by simp

text‚ÄπIf $x$ is an element of the group and $z_1,z_2$ are negative integers then
  the power homomorphism property holds.‚Ä∫

lemma (in group_int0) powz_hom_neg_neg: 
  assumes "z‚á©1\<ls>ùü¨" "z‚á©2\<ls>ùü¨" "x‚ààG"
  shows "powz(z‚á©1\<ra>z‚á©2,x) = powz(z‚á©1,x)‚ãÖpowz(z‚á©2,x)"
  using assms ints.neg_not_nonneg ints.add_neg_ints(1,2) 
    inverse_in_group monoid.nat_mult_add 
  unfolding powz_def by simp

text‚ÄπWhen the integers are of different signs we further split into cases depending on
  which magnitude is greater. If $x$ is an element of the group and $z_1$ is nonnegative, 
  $z_2$ is negative and $|z_2|\leq |z_1|$ then the power homomorphism property holds.
  The proof of this lemma is presented with more detail than necessary, 
  to show the schema of the proofs of the remaining lemmas that we let Isabelle prove 
  automatically.‚Ä∫

lemma (in group_int0) powz_hom_nneg_neg1: 
  assumes "ùü¨\<lsq>z‚á©1"  "z‚á©2\<ls>ùü¨" "zmagnitude(z‚á©2) ‚â§ zmagnitude(z‚á©1)" "x‚ààG"
  shows "powz(z‚á©1\<ra>z‚á©2,x) = powz(z‚á©1,x)‚ãÖpowz(z‚á©2,x)"
proof -
  let ?m‚á©1 = "zmagnitude(z‚á©1)" 
  let ?m‚á©2 = "zmagnitude(z‚á©2)"
  from assms(1,2,3) have "powz(z‚á©1\<ra>z‚á©2,x) = pow(zmagnitude(z‚á©1\<ra>z‚á©2),x)"
    using ints.add_nonneg_neg1(1) unfolding powz_def by simp
  also from assms(1,2,3) have "... =  pow(?m‚á©1 #- ?m‚á©2,x)" 
    using ints.add_nonneg_neg1(2) by simp
  also from assms(3,4) have "... = pow(?m‚á©1,x)‚ãÖpow(?m‚á©2,x¬Ø)"
    using nat_pow_cancel_less by simp
  also from assms(1,2) have "... = powz(z‚á©1,x)‚ãÖpowz(z‚á©2,x)" 
    using ints.neg_not_nonneg unfolding powz_def by simp
  finally show ?thesis by simp
qed

text‚ÄπIf $x$ is an element of the group and $z_1$ is nonnegative, 
  $z_2$ is negative and $|z_1|<|z_2|$ then the power homomorphism property holds.‚Ä∫

lemma (in group_int0) powz_hom_nneg_neg2:
  assumes "ùü¨\<lsq>z‚á©1"  "z‚á©2\<ls>ùü¨"  "zmagnitude(z‚á©1) < zmagnitude(z‚á©2)" "x‚ààG"
  shows "powz(z‚á©1\<ra>z‚á©2,x) = powz(z‚á©1,x)‚ãÖpowz(z‚á©2,x)"
  using assms ints.add_nonneg_neg2 ints.neg_not_nonneg leI nat_pow_cancel_more1
  unfolding powz_def by simp

text‚ÄπIf $x$ is an element of the group and $z_1$ is negative, 
  $z_2$ is nonnegative and $|z_1|\leq |z_2|$ then the power homomorphism property holds.‚Ä∫

lemma (in group_int0) powz_hom_neg_nneg1:
  assumes "z‚á©1\<ls>ùü¨" "ùü¨\<lsq>z‚á©2" "zmagnitude(z‚á©1) ‚â§ zmagnitude(z‚á©2)" "x‚ààG"
  shows "powz(z‚á©1\<ra>z‚á©2,x) = powz(z‚á©1,x)‚ãÖpowz(z‚á©2,x)"
  using assms ints.add_neg_nonneg1 nat_pow_cancel_more ints.neg_not_nonneg
  unfolding powz_def by simp

text‚ÄπIf $x$ is an element of the group and $z_1$ is negative,
