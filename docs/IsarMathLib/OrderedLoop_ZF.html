section ‚Äπ Ordered loops ‚Ä∫

theory OrderedLoop_ZF imports Loop_ZF Order_ZF

begin

text‚Äπ This theory file is about properties of loops (the algebraic structures introduced 
  in IsarMathLib in the ‚ÄπLoop_ZF‚Ä∫ theory) with an additional order relation that is in
  a way compatible with the loop's binary operation. The oldest reference I have found on the subject is
  \cite{Zelinski1948}. ‚Ä∫

subsection‚ÄπDefinition and notation‚Ä∫

text‚Äπ An ordered loop $(G,A)$ is a loop with a partial order relation r that is 
  "translation invariant" with respect to the loop operation $A$.‚Ä∫
  
text‚Äπ A triple $(G,A,r)$ is an ordered loop if $(G,A)$ is a loop and $r$ is a relation
  on $G$ (i.e. a subset of $G\times G$ with is a partial order and for all elements $x,y,z \in G$
  the condition $\langle x,y\rangle \in r$ is equivalent to both
  $\langle A\langle x,z\rangle, A\langle x,z\rangle\rangle \in r$ and 
  $\langle A\langle z,x\rangle, A\langle z,x\rangle\rangle \in r$. 
  This looks a bit awkward in the basic set theory notation, but using the additive notation 
  for the group operation and $x\leq y$ to instead of $\langle x,y \rangle \in r$ this just means that
  $x\leq y$ if and only if $x+z\leq y+z$ and $x\leq y$ if and only if $z+x\leq z+y$. ‚Ä∫

definition
  "IsAnOrdLoop(L,A,r) ‚â° 
  IsAloop(L,A) ‚àß r‚äÜL√óL ‚àß IsPartOrder(L,r) ‚àß (‚àÄx‚ààL. ‚àÄy‚ààL. ‚àÄz‚ààL. 
  ((‚ü®x,y‚ü© ‚àà r ‚ü∑ ‚ü®A`‚ü® x,z‚ü©,A`‚ü®y,z‚ü©‚ü© ‚àà r) ‚àß (‚ü®x,y‚ü© ‚àà r ‚ü∑ ‚ü®A`‚ü®z,x‚ü©,A`‚ü®z,y‚ü©‚ü© ‚àà r )))"

text‚ÄπWe define the set of nonnegative elements  in the obvious way as $L^+ =\{x\in L: 0 \leq x\}$.‚Ä∫

definition
  "Nonnegative(L,A,r) ‚â° {x‚ààL. ‚ü® TheNeutralElement(L,A),x‚ü© ‚àà r}"

text‚ÄπThe ‚ÄπPositiveSet(L,A,r)‚Ä∫ is a set similar to  ‚ÄπNonnegative(L,A,r)‚Ä∫, but without the neutral element.‚Ä∫

definition
  "PositiveSet(L,A,r) ‚â° 
  {x‚ààL. ‚ü® TheNeutralElement(L,A),x‚ü© ‚àà r ‚àß TheNeutralElement(L,A)‚â† x}"

text‚Äπ We will use the additive notation for ordered loops.‚Ä∫

locale loop1 =
  fixes L and A and r

  assumes ordLoopAssum: "IsAnOrdLoop(L,A,r)"

  fixes neut ("ùü¨")
  defines neut_def[simp]: "ùü¨ ‚â° TheNeutralElement(L,A)"
  
  fixes looper (infixl "\<ra>" 69)
  defines looper_def[simp]: "x \<ra> y ‚â° A`‚ü® x,y‚ü©"
  
  fixes lesseq (infix "\<lsq>" 68)
  defines lesseq_def [simp]: "x \<lsq> y ‚â° ‚ü®x,y‚ü© ‚àà r"
  
  fixes sless (infix "\<ls>" 68)
  defines sless_def[simp]: "x \<ls> y ‚â° x\<lsq>y ‚àß x‚â†y"
  
  fixes nonnegative ("L‚áß+")
  defines nonnegative_def [simp]: "L‚áß+ ‚â° Nonnegative(L,A,r)"
  
  fixes positive ("L‚á©+")
  defines positive_def[simp]: "L‚á©+ ‚â° PositiveSet(L,A,r)"
  
  fixes leftdiv ("\<rm> _ \<ad> _")
  defines leftdiv_def[simp]: "\<rm>x\<ad>y ‚â° LeftDiv(L,A)`‚ü®x,y‚ü©"
  
  fixes rightdiv (infixl "\<rs>" 69)
  defines rightdiv_def[simp]:"x\<rs>y ‚â° RightDiv(L,A)`‚ü®y,x‚ü©"

text‚Äπ In an ordered loop the order is translation invariant. ‚Ä∫

lemma (in loop1) ord_trans_inv: assumes "x\<lsq>y" "z‚ààL"
  shows "x\<ra>z \<lsq> y\<ra>z" and "z\<ra>x \<lsq> z\<ra>y"
proof -
  from ordLoopAssum assms have 
    "(‚ü®x,y‚ü© ‚àà r ‚ü∑ ‚ü®A`‚ü® x,z‚ü©,A`‚ü®y,z‚ü©‚ü© ‚àà r) ‚àß (‚ü®x,y‚ü© ‚àà r ‚ü∑ ‚ü®A`‚ü®z,x‚ü©,A`‚ü®z,y‚ü©‚ü© ‚àà r )"
    using lsq_members unfolding IsAnOrdLoop_def by blast
  with assms(1) show "x\<ra>z \<lsq> y\<ra>z" and "z\<ra>x \<lsq> z\<ra>y" by auto
qed

text‚Äπ In an ordered loop the strict order is translation invariant. ‚Ä∫

lemma (in loop1) strict_ord_trans_inv: assumes "x\<ls>y" "z‚ààL"
  shows "x\<ra>z \<ls> y\<ra>z" and "z\<ra>x \<ls> z\<ra>y"
proof -
  from assms have "x\<ra>z \<lsq> y\<ra>z" and "z\<ra>x \<lsq> z\<ra>y"
    using ord_trans_inv by auto
  moreover have "x\<ra>z ‚â† y\<ra>z" and "z\<ra>x ‚â† z\<ra>y"
  proof -
    { assume "x\<ra>z = y\<ra>z"
      with assms have "x=y" using less_members qg_cancel_right by blast
      with assms(1) have False by simp
    } thus "x\<ra>z ‚â† y\<ra>z" by auto
    { assume "z\<ra>x = z\<ra>y"
      with assms have "x=y" using less_members qg_cancel_left by blast
      with assms(1) have False by simp
    } thus "z\<ra>x ‚â† z\<ra>y" by auto
  qed
  ultimately show "x\<ra>z \<ls> y\<ra>z" and "z\<ra>x \<ls> z\<ra>y"
    by auto
qed

text‚Äπ The order in an ordered loop is antisymmeric. ‚Ä∫

lemma (in loop1) loop_ord_antisym: assumes "x\<lsq>y" and "y\<lsq>x"
  shows "x=y"
proof -
  from ordLoopAssum assms have "antisym(r)" "‚ü®x,y‚ü© ‚àà r" "‚ü®y,x‚ü© ‚àà r" 
    unfolding IsAnOrdLoop_def IsPartOrder_def by auto
  then show "x=y" by (rule Fol1_L4)
qed

text‚Äπ The loop order is transitive. ‚Ä∫

lemma (in loop1) loop_ord_trans: assumes "x\<lsq>y" and "y\<lsq>z" shows "x\<lsq>z"
proof - 
  from ordLoopAssum assms have "trans(r)" and "‚ü®x,y‚ü© ‚àà r ‚àß ‚ü®y,z‚ü© ‚àà r"
    unfolding IsAnOrdLoop_def IsPartOrder_def by auto
  then have "‚ü®x,z‚ü© ‚àà r" by (rule Fol1_L3)
  thus ?thesis by simp
qed

text‚Äπ A form of mixed transitivity for the strict order: ‚Ä∫

lemma (in loop1) loop_strict_ord_trans: assumes "x\<lsq>y" and "y\<ls>z"
  shows "x\<ls>z"
proof -
  from assms have "x\<lsq>y" and "y\<lsq>z" by auto
  then have "x\<lsq>z" by (rule loop_ord_trans)
  with assms show "x\<ls>z" using loop_ord_antisym by auto
qed

text‚Äπ Another form of mixed transitivity for the strict order: ‚Ä∫

lemma (in loop1) loop_strict_ord_trans1: assumes "x\<ls>y" and "y\<lsq>z"
  shows "x\<ls>z"
proof -
  from assms have "x\<lsq>y" and "y\<lsq>z" by auto
  then have "x\<lsq>z" by (rule loop_ord_trans)
  with assms show "x\<ls>z" using loop_ord_antisym by auto
qed


text‚Äπ We can move an element to the other side of an inequality. Well, not exactly, but
  our notation creates an illusion to that effect. ‚Ä∫

lemma (in loop1) lsq_other_side: assumes "x\<lsq>y" 
  shows "ùü¨ \<lsq> \<rm>x\<ad>y"  "(\<rm>x\<ad>y) ‚àà L‚áß+" "ùü¨ \<lsq> y\<rs>x" "(y\<rs>x) ‚àà L‚áß+"
proof -
  from assms have "x‚ààL" "y‚ààL" "ùü¨‚ààL" "(\<rm>x\<ad>y) ‚àà L" "(y\<rs>x) ‚àà L" 
    using lsq_members neut_props_loop(1) lrdiv_props(2,5) by auto
  then have "x = x\<ra>ùü¨" and "y = x\<ra>(\<rm>x\<ad>y)" using neut_props_loop(2) lrdiv_props(6)
    by auto
  with assms have "x\<ra>ùü¨ \<lsq> x\<ra>(\<rm>x\<ad>y)" by simp
  with ‚Äπx‚ààL‚Ä∫ ‚Äπùü¨‚ààL‚Ä∫ ‚Äπ(\<rm>x\<ad>y) ‚àà L‚Ä∫ show "ùü¨ \<lsq> \<rm>x\<ad>y" using ineq_cancel_left
    by simp
  then show "(\<rm>x\<ad>y) ‚àà L‚áß+" using nonneg_definition by simp
  from ‚Äπx‚ààL‚Ä∫ ‚Äπy‚ààL‚Ä∫ have "x = ùü¨\<ra>x" and "y = (y\<rs>x)\<ra>x"
    using neut_props_loop(2) lrdiv_props(3) by auto
  with assms have "ùü¨\<ra>x \<lsq> (y\<rs>x)\<ra>x" by simp
  with ‚Äπx‚ààL‚Ä∫ ‚Äπùü¨‚ààL‚Ä∫ ‚Äπ(y\<rs>x) ‚àà L‚Ä∫ show "ùü¨ \<lsq> y\<rs>x" using ineq_cancel_right
    by simp
  then show "(y\<rs>x) ‚àà L‚áß+" using  nonneg_definition by simp
qed

text‚Äπ We can move an element to the other side of a strict inequality. ‚Ä∫

lemma (in loop1) ls_other_side: assumes "x\<ls>y" 
  shows "ùü¨ \<ls> \<rm>x\<ad>y"  "(\<rm>x\<ad>y) ‚àà L‚á©+" "ùü¨ \<ls> y\<rs>x" "(y\<rs>x) ‚àà L‚á©+"
proof -
  from assms have "x‚ààL" "y‚ààL" "ùü¨‚ààL" "(\<rm>x\<ad>y) ‚àà L" "(y\<rs>x) ‚àà L" 
    using lsq_members neut_props_loop(1) lrdiv_props(2,5) by auto
  then have "x = x\<ra>ùü¨" and "y = x\<ra>(\<rm>x\<ad>y)" using neut_props_loop(2) lrdiv_props(6)
    by auto
  with assms have "x\<ra>ùü¨ \<ls> x\<ra>(\<rm>x\<ad>y)" by simp
  with ‚Äπx‚ààL‚Ä∫ ‚Äπùü¨‚ààL‚Ä∫ ‚Äπ(\<rm>x\<ad>y) ‚àà L‚Ä∫ show "ùü¨ \<ls> \<rm>x\<ad>y" using strict_ineq_cancel_left
    by simp
  then show "(\<rm>x\<ad>y) ‚àà L‚á©+" using posset_definition1 by simp
  from ‚Äπx‚ààL‚Ä∫ ‚Äπy‚ààL‚Ä∫ have "x = ùü¨\<ra>x" and "y = (y\<rs>x)\<ra>x"
    using neut_props_loop(2) lrdiv_props(3) by auto
  with assms have "ùü¨\<ra>x \<ls> (y\<rs>x)\<ra>x" by simp
  with ‚Äπx‚ààL‚Ä∫ ‚Äπùü¨‚ààL‚Ä∫ ‚Äπ(y\<rs>x) ‚àà L‚Ä∫ show "ùü¨ \<ls> y\<rs>x" using strict_ineq_cancel_right
    by simp
  then show "(y\<rs>x) ‚àà L‚á©+" using posset_definition1 by simp
qed
