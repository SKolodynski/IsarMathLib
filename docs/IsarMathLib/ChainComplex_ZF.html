section ‹Chain complexes of abelian groups›

theory ChainComplex_ZF imports Group_ZF_6 IntDiv_ZF_IML

begin

text‹A \emph{chain complex} of abelian groups consists of a family of abelian groups
  $(C_n)_{n \in \mathbb{Z}}$ together with group homomorphisms
  $d_n \colon C_n \to C_{n-1}$, satisfying $d_{n-1} \circ d_n = 0$ for every
  $n \in \mathbb{Z}$ (i.e.\ the image of $d_n$ is contained in the kernel of
  $d_{n-1}$).  Here $C$ and $P$ are Isabelle-level functions $\mathbb{Z} \to
  \mathbf{Set}$ giving the carriers and operations, and $d$ is an
  Isabelle-level function giving the boundary maps.›

definition (in int0) IsAchainComplex :: "[i⇒i, i⇒i, i⇒i] ⇒ o" where
  "IsAchainComplex(C, P, d) ≡
    (∀n ∈ int. IsAgroup(C(n), P(n)) ∧ (P(n)) {is commutative on} (C(n))) ∧
    (∀n ∈ int. Homomor(d(n), C(n), P(n), C(n \<rs> 𝟭), P(n  \<rs> 𝟭))) ∧
    (∀n ∈ int. (d(n))``(C(n)) ⊆
      (d(n \<rs> 𝟭))-``{TheNeutralElement(C(n  \<rs> 𝟭 \<rs> 𝟭), P(n  \<rs> 𝟭 \<rs> 𝟭))})"

text‹The locale ‹chain_complex› packages a fixed chain complex.›

locale chain_complex = int0 +
  fixes C :: "i⇒i" and P :: "i⇒i" and d :: "i⇒i"
  assumes cplx: "IsAchainComplex(C, P, d)"

abbreviation (in chain_complex) zero_C
  where "zero_C ≡ λn. TheNeutralElement(C(n),P(n))"

text‹Additive notation for the group operation at degree $n$:
  $x +_n y$ stands for $P_n(x,y)$.›

abbreviation (in chain_complex) cgrp_add :: "[i, i, i] ⇒ i"
    ("_ \<ra>⇘_⇙ _" [65,65,66] 65) where
  "x \<ra>⇘n⇙ y ≡ P(n)`⟨x, y⟩"

text‹Each $C_n$ is a group.›

lemma (in chain_complex) cplx_group:
  assumes "n ∈ ints"
  shows "IsAgroup(C(n), P(n))"
  using assms cplx unfolding IsAchainComplex_def by auto

text‹Each $C_n$ is abelian.›

lemma (in chain_complex) cplx_abelian:
  assumes "n ∈ ints"
  shows "(P(n)) {is commutative on} (C(n))"
  using assms cplx unfolding IsAchainComplex_def by auto

text‹The boundary map $d_n$ is a group homomorphism from $C_n$ to $C_{n-1}$.›

lemma (in chain_complex) cplx_hom:
  assumes "n ∈ ints"
  shows "Homomor(d(n), C(n), P(n), C(n \<rs>𝟭), P(n \<rs>𝟭))"
  using assms cplx unfolding IsAchainComplex_def by auto

text‹The image of $d_n$ is contained in the kernel of $d_{n-1}$:
  the boundary-squared-zero condition.›

lemma (in chain_complex) bnd_sq_zero:
  assumes "n ∈ ints"
  shows "(d(n))``(C(n)) ⊆
    (d(n\<rs>𝟭))-``{zero_C(n \<rs>𝟭\<rs>𝟭)}"
  using assms cplx unfolding IsAchainComplex_def by auto

text‹For any $x \in C_n$, applying $d_{n-1}$ to $d_n(x)$ gives the neutral element
  of $C_{n-2}$.›

lemma (in chain_complex) bnd_sq_zero_elem:
  assumes "n ∈ ints" and "x ∈ C(n)"
  shows "d(n \<rs> 𝟭)`(d(n)`x) = zero_C(n \<rs> 𝟭\<rs> 𝟭)"
proof -
  from assms(1) have pn: "n\<rs> 𝟭 ∈ int" using Int_ZF_1_L10(1) Int_ZF_1_L8A(2) by simp
  from assms(1) have fn: "d(n) : C(n) → C(n \<rs> 𝟭)"
    using cplx_hom isub_def Int_ZF_1_L10 unfolding Homomor_def by simp
  from fn assms(2) have bx: "d(n)`x ∈ C(n \<rs>𝟭)" using apply_type by simp
  from fn assms(2) have "d(n)`x ∈ (d(n))``(C(n))"
    using func_imagedef by auto
  with bnd_sq_zero[OF assms(1)] have
    mem: "d(n)`x ∈ (d(n \<rs>𝟭))-``{zero_C(n \<rs>𝟭\<rs>𝟭)}"
    by auto
  from pn have fpn: "d(n \<rs>𝟭) : C(n \<rs>𝟭) → C(n \<rs>𝟭\<rs>𝟭)"
    using cplx_hom unfolding Homomor_def by simp
  from fpn bx mem show ?thesis using func1_1_L15 by auto
qed

text‹The composition $d_{n-1} \circ d_n$ is the constant zero map on $C_n$.›

lemma (in chain_complex) bnd_sq_zero_2:
  assumes "n ∈ int"
  shows "d(n \<rs> 𝟭) O d(n) =
         ConstantFunction(C(n), zero_C(n \<rs> 𝟭\<rs> 𝟭))"
proof -
  let ?e = "zero_C(n \<rs> 𝟭\<rs> 𝟭)"
  from assms have pn: "n \<rs> 𝟭 ∈ ints"
    using Int_ZF_1_L10(1) Int_ZF_1_L8A(2) by simp
  from pn have ppn: "n \<rs> 𝟭 \<rs> 𝟭 ∈ ints"
    using Int_ZF_1_L10(1) Int_ZF_1_L8A(2) by simp
  from assms have fn: "d(n) : C(n) → C(n \<rs> 𝟭)"
    using cplx_hom unfolding Homomor_def by simp
  from pn have fpn: "d(n \<rs> 𝟭) : C(n \<rs> 𝟭) → C(n\<rs> 𝟭\<rs> 𝟭)"
    using cplx_hom unfolding Homomor_def by simp
  from fn fpn have lhs_fun: "d(n\<rs> 𝟭) O d(n) : C(n) → C(n \<rs> 𝟭\<rs> 𝟭)"
    using comp_fun by simp
  from ppn have grp: "IsAgroup(C(n\<rs> 𝟭\<rs> 𝟭), P(n \<rs> 𝟭\<rs> 𝟭))"
    using cplx_group by simp
  from grp have eC: "?e ∈ C(n \<rs> 𝟭\<rs> 𝟭)"
    using group0.group0_2_L2 unfolding group0_def by auto
  from eC have rhs_fun: "ConstantFunction(C(n), ?e) : C(n) → C(n \<rs> 𝟭\<rs> 𝟭)"
    using func1_3_L1 by simp
  show ?thesis
  proof (rule fun_extension[OF lhs_fun rhs_fun])
    fix x assume xn: "x ∈ C(n)"
    have "(d(n \<rs> 𝟭) O d(n))`x = d(n \<rs> 𝟭)`(d(n)`x)"
      using comp_fun_apply fn fpn xn by simp
    also have "… = ?e" using bnd_sq_zero_elem assms xn by simp
    also have "… = ConstantFunction(C(n), ?e)`x"
      using func1_3_L2 xn by simp
    finally show "(d(n \<rs> 𝟭) O d(n))`x = ConstantFunction(C(n), ?e)`x" .
  qed
qed

subsection ‹Chain maps›

text‹A \emph{chain map} between two chain complexes is a family of group
  homomorphisms $f_n \colon C_n \to C'_n$ that commute with the boundary
  maps: $d'_n \circ f_n = f_{n-1} \circ d_n$ for every $n \in \mathbb{Z}$.›

definition (in int0) IsAchainMap ::
  "[i⇒i, i⇒i, i⇒i, i⇒i, i⇒i, i⇒i, i⇒i] ⇒ o" where
  "IsAchainMap(f, C, P, d, C', P', d') ≡
     (∀n ∈ int. Homomor(f(n), C(n), P(n), C'(n), P'(n))) ∧
     (∀n ∈ int. d'(n) O f(n) = f(n \<rs> 𝟭) O d(n))"

text‹The locale ‹chain_map› packages two fixed chain complexes and a chain
  map between them.›

locale chain_map = int0 +
  fixes C  :: "i⇒i" and P  :: "i⇒i" and d  :: "i⇒i"
    and C' :: "i⇒i" and P' :: "i⇒i" and d' :: "i⇒i"
    and f  :: "i⇒i"
  assumes src_cplx: "IsAchainComplex(C, P, d)"
    and   tgt_cplx: "IsAchainComplex(C', P', d')"
    and   cmap:     "IsAchainMap(f, C, P, d, C', P', d')"

sublocale chain_map ⊆ src: chain_complex
  using src_cplx by unfold_locales

sublocale chain_map ⊆ tgt: chain_complex where C=C' and P=P' and d=d'
  using tgt_cplx by unfold_locales

text‹Each component $f_n$ is a group homomorphism.›

lemma (in chain_map) cmap_hom:
  assumes "n ∈ ints"
  shows "Homomor(f(n), C(n), P(n), C'(n), P'(n))"
  using assms cmap unfolding IsAchainMap_def by auto

text‹The chain map commutes with the boundary maps.›

lemma (in chain_map) cmap_comm:
  assumes "n ∈ ints"
  shows "d'(n) O f(n) = f(n \<rs> 𝟭) O d(n)"
  using assms cmap unfolding IsAchainMap_def by auto


end
