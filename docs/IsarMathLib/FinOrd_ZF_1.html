section ‹Finite choice and order relations›

theory FinOrd_ZF_1 imports FinOrd_ZF Cardinal_ZF

begin

text‹In this theory we continue the subject of finite sets and order relation 
  from ‹FinOrd_ZF› with some consequences of finite choice for down-directed sets. ›

subsection‹Finte choice and preorders›

text‹In the ‹Order_ZF› theory we define what it means that a relation $r$ down-directs a set $X$:
  each two elements of $X$ have a common lower bound in $X$. If the relation is a preorder 
  (i.e. is reflexive and transitive) and it down-directs $X$ we say that $X$ is a down-directed
  set (by the relation $r$). ›

text‹The next lemma states that each finite subset of a down-directed set 
  has a lower bound in $X$.›

lemma fin_dir_set_bounded: 
  assumes "IsDownDirectedSet(X,r)" and "B∈FinPow(X)"
  shows "∃x∈X.∀t∈B. ⟨x,t⟩∈r"
proof -  
  from assms(1) have "∃x∈X.∀t∈∅. ⟨x,t⟩∈r"
    unfolding IsDownDirectedSet_def DownDirects_def by auto
  moreover have 
    "∀A∈FinPow(X). (∃x∈X.∀t∈A. ⟨x,t⟩∈r)⟶(∀a∈X. ∃m∈X.∀t∈A∪{a}. ⟨m,t⟩∈r)"
  proof -
    { fix A assume "A∈FinPow(X)" and I: "∃x∈X.∀t∈A. ⟨x,t⟩∈r"
      { fix a assume "a∈X"
        from I obtain x where "x∈X" and II: "∀t∈A. ⟨x,t⟩∈r" by auto
        from assms(1) ‹a∈X› ‹x∈X› obtain m where 
          "m∈X" "⟨m,a⟩∈r" "⟨m,x⟩∈r"
          unfolding IsDownDirectedSet_def DownDirects_def by auto
        with assms(1) II have "∃m∈X.∀t∈A∪{a}. ⟨m,t⟩∈r"
          unfolding IsDownDirectedSet_def IsPreorder_def trans_def 
            by blast
      } hence "∀a∈X. ∃x∈X.∀t∈A∪{a}. ⟨x,t⟩∈r" by blast
    } thus ?thesis by simp
  qed
  moreover note assms(2)
  ultimately show ?thesis by (rule FinPow_induct)
qed

text‹Suppose $Y$ is a set down-directed by a (preorder) relation $r$
  and $f,g$ are funtions defined on two finite subsets $A,B$, resp., of $X$, valued in Y 
  (i.e.$f:A\rightarrow Y$, $f:B\rightarrow Y$ and $A,B$ are finite subsets of $X$). 
  Then there exist a function $h:A\cup B\rightarrow Y$ that is a lower bound for 
  $f$ on $A$ and for $g$ on $B$.›

lemma two_fun_low_bound: 
  assumes "IsDownDirectedSet(Y,r)" "A∈FinPow(X)" "B∈FinPow(X)" "f:A→Y" "g:B→Y"
  shows "∃h∈A∪B→Y. (∀x∈A. ⟨h`(x),f`(x)⟩∈r) ∧ (∀x∈B. ⟨h`(x),g`(x)⟩∈r)"
proof -
  from assms(2,3) have "Finite(A∪B)" using union_finpow 
    unfolding FinPow_def by simp
  { fix x assume "x∈A∪B"
    from assms(4,5) have "f``{x}∪g``{x} ∈ FinPow(Y)"
      using image_singleton_fin union_finpow by simp
    with assms(1) have "∃y∈Y.∀t∈(f``{x}∪g``{x}). ⟨y,t⟩∈r"
      using fin_dir_set_bounded by simp
  } hence "∀x∈A∪B. ∃y∈Y.∀t∈(f``{x}∪g``{x}). ⟨y,t⟩∈r" by auto
  with ‹Finite(A∪B)› have 
    "∃h∈A∪B→Y. ∀x∈A∪B. ∀t∈(f``{x}∪g``{x}). ⟨h`(x),t⟩∈r"
    by (rule finite_choice_fun)
  then obtain h where "h∈A∪B→Y" and 
    "∀x∈A∪B. ∀t∈(f``{x}∪g``{x}). ⟨h`(x),t⟩∈r"
    by auto
  with assms(4,5) have 
    "∀x∈A. ⟨h`(x),f`(x)⟩∈r" and "∀x∈B. ⟨h`(x),g`(x)⟩∈r"
    using func_imagedef by simp_all
  with ‹h∈A∪B→Y› show ?thesis by auto
qed

end
