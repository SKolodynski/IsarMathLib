section ‚Äπ$\mathbb{Z}$ modules‚Ä∫

theory IntModule_ZF imports Module_ZF IntGroup_ZF

text‚ÄπIn this section we show that the integers, as a ring, have only one module structure on 
  each abelian group. We will show that the module structure is unique, but we will also show 
  which action is the one that defines that module structure.‚Ä∫

subsection‚ÄπInteger powers as the only integer actions on abelian groups‚Ä∫

text‚ÄπAs we show in theorem ‚Äπpowz_maps_int_End‚Ä∫ in the ‚ÄπIntGroup‚Ä∫ theory
  the mapping $\mathbb{Z}\ni n\mapsto (G\ni x\mapsto x^n)$ maps integers to 
  endomorphisms of the abelian group $G$. We will show here that this mapping is the unique 
  action of integers on that group.‚Ä∫

text‚ÄπWhen $\mathbb{Z}$ acts on a group, that action is unique.‚Ä∫

lemma (in int0) action_unique:
  assumes "IsLeftModule(int,IntegerAddition,IntegerMultiplication,G,f,S‚á©1)" and  
    "IsLeftModule(int,IntegerAddition,IntegerMultiplication,G,f,S‚á©2)"
  shows "S‚á©1 = S‚á©2"
proof -
  let ?A‚á©Z = "IntegerAddition"
  let ?M‚á©Z = "IntegerMultiplication"
  from assms have mod0: "module0(‚Ñ§,?A‚á©Z,?M‚á©Z,G,f,S‚á©1)" "module0(‚Ñ§,?A‚á©Z,?M‚á©Z,G,f,S‚á©2)"
      unfolding module0_def IsLeftModule_def ring0_def module0_axioms_def by auto
  { fix r assume rg: "r‚àà‚Ñ§"    
    from mod0(1) have eq: "‚àÄg‚ààG. (S‚á©1`(ùü¨))`(g) = TheNeutralElement(G, f)"
      using module0.mult_zero by simp
    with mod0(1) have "S‚á©1`(ùü¨) = {‚ü®g,TheNeutralElement(G,f)‚ü©. g‚ààG}"
      using Int_ZF_1_L8A(1) module0.H_val_type(2) fun_is_set_of_pairs
      by force
    then have s1: "S‚á©1`(ùü¨) = ConstantFunction(G,TheNeutralElement(G, f))"
      unfolding ConstantFunction_def by auto
    from mod0(2) have eq: "‚àÄg‚ààG. (S‚á©2`(ùü¨))`(g) = TheNeutralElement(G, f)"
      using module0.mult_zero by auto
    with mod0(2) have "(S‚á©2`(ùü¨)):G‚ÜíG" using Int_ZF_1_L8A(1) module0.H_val_type
      by auto
    with eq s1 have "S‚á©1`(ùü¨) = S‚á©2`(ùü¨)"
      unfolding ConstantFunction_def using fun_is_set_of_pairs by force
    { assume "r\<lsq>ùü¨"
      then have "r‚àà‚Ñ§" using OrderedGroup_ZF_1_L4(1) by simp
      have "S‚á©1`(r) = S‚á©2`(r)"
      proof -
        { fix n assume "n\<lsq>ùü¨" "S‚á©1`(n) = S‚á©2`(n)"
          from ‚Äπn\<lsq>ùü¨‚Ä∫ have "n‚àà‚Ñ§" using OrderedGroup_ZF_1_L4(1) by simp
          have "(\<rm>ùü≠) ‚àà ‚Ñ§" using group0.inverse_in_group Int_ZF_1_T2(3)
              ring0.Ring_ZF_1_L2(2) Int_ZF_1_1_L2(3) by simp
          { fix t assume t: "t‚ààG"
            with mod0(1) ‚ÄπS‚á©1`(n) = S‚á©2`(n)‚Ä∫ ‚Äπn‚àà‚Ñ§‚Ä∫ ‚Äπ(\<rm>ùü≠)‚àà‚Ñ§‚Ä∫ have 
              "S‚á©1`(n\<ra>(\<rm> ùü≠))`(t) = (f`‚ü®(S‚á©2`(n))`(t),(S‚á©1`(\<rm>ùü≠))`(t)‚ü©)"
              using  module0.module_ax2 t by simp
            moreover from mod0(1) have ex: "‚àÄg‚ààG. S‚á©1`(\<rm>ùü≠)`(g) = GroupInv(G, f)`(g)" 
              using module0.inv_module apply_type module0.H_val_type(2) ring0.Ring_ZF_1_L2(2) 
                Int_ZF_1_1_L2(3) module0.module_ax4 by simp
            moreover from mod0(1) have "S‚á©1`(\<rm> ùü≠):G‚ÜíG"
              using module0.H_val_type(2) ring0.Ring_ZF_1_L2(2) Int_ZF_1_1_L2(3)
                ring0.Ring_ZF_1_L3(1) Int_ZF_1_1_L2(3) by auto
            moreover 
            from assms(1) have "GroupInv(G, f): G‚ÜíG"
              using group0_2_T2 unfolding IsLeftModule_def by blast
            with ex ‚ÄπS‚á©1`(\<rm> ùü≠):G‚ÜíG‚Ä∫ have "S‚á©1`(\<rm> ùü≠) =  GroupInv(G, f)" 
              using func_eq by simp
            ultimately have A: "S‚á©1`(n\<ra>(\<rm>ùü≠))`(t) = f`‚ü®(S‚á©2`(n))`(t),GroupInv(G, f) `t‚ü©" 
              by simp
            from assms(2) ‚Äπ(\<rm>ùü≠) ‚àà ‚Ñ§‚Ä∫ ‚Äπn‚àà‚Ñ§‚Ä∫ mod0(2) t ‚ÄπS‚á©1`(n) = S‚á©2`(n)‚Ä∫ have
              "(S‚á©2`(n\<ra>\<rm>ùü≠))`(t) = f`‚ü®(S‚á©2`(n))`(t),(S‚á©2`(\<rm>ùü≠))`(t)‚ü©"
              using module0.module_ax2 by simp
            moreover from mod0(2) have ex: "‚àÄg‚ààG. (S‚á©2`(\<rm>ùü≠))`(g) = GroupInv(G, f)`(g)"
              using module0.inv_module apply_type module0.H_val_type(2) 
                ring0.Ring_ZF_1_L2(2) Int_ZF_1_1_L2(3) module0.module_ax4 by simp
            moreover 
            from mod0(2) have "S‚á©2`(\<rm> ùü≠): G‚ÜíG"
              using module0.H_val_type(2) ring0.Ring_ZF_1_L2(2) Int_ZF_1_1_L2(3)
                ring0.Ring_ZF_1_L3(1) Int_ZF_1_1_L2(3) by auto
            with ex ‚ÄπGroupInv(G, f): G‚ÜíG‚Ä∫ have "S‚á©2`(\<rm>ùü≠) = GroupInv(G, f)" 
              using func_eq by blast
            ultimately have "S‚á©2`(n\<ra>(\<rm>ùü≠))`(t) = f`‚ü®(S‚á©2`(n))`(t),GroupInv(G, f)`(t)‚ü©"
              by simp
            with A have "(S‚á©1`(n\<ra>(\<rm> ùü≠)))`(t) = S‚á©2`(n\<ra>(\<rm>ùü≠))`(t)" by simp
          } hence "‚àÄt‚ààG. S‚á©1`(n \<ra> \<rm> ùü≠)`(t) = S‚á©2`(n\<ra>(\<rm>ùü≠))`(t)" by simp
          moreover from mod0 ‚Äπ(\<rm>ùü≠) ‚àà ‚Ñ§‚Ä∫ ‚Äπn‚àà‚Ñ§‚Ä∫ have 
            "S‚á©1`(n\<ra>(\<rm>ùü≠)):G‚ÜíG" and "S‚á©2`(n\<ra>(\<rm>ùü≠)):G‚ÜíG"

            by simp_all
          ultimately have "S‚á©1`(n\<ra>(\<rm>ùü≠)) = S‚á©2`(n\<ra>(\<rm>ùü≠))"
            using func_eq by simp
        } hence "‚àÄn. (n\<lsq>ùü¨) ‚àß (S‚á©1`(n) = S‚á©2`(n))‚ü∂S‚á©1`(n\<rs>ùü≠) = S‚á©2`(n\<rs>ùü≠)"
          by auto
        with ‚Äπr\<lsq>ùü¨‚Ä∫ ‚ÄπS‚á©1`(ùü¨) = S‚á©2`(ùü¨)‚Ä∫ show "S‚á©1`(r) = S‚á©2`(r)"
          using Back_induct_on_int by simp
      qed
    }
    moreover
    { assume "ùü¨\<lsq>r"
      then have "r‚àà‚Ñ§" using OrderedGroup_ZF_1_L4(2) by simp
      have "S‚á©1`(r) = S‚á©2`(r)"
      proof -
        { fix m assume "ùü¨\<lsq>m" "S‚á©1`(m) = S‚á©2`(m)"
          from ‚Äπùü¨\<lsq>m‚Ä∫ have "m‚àà‚Ñ§" using OrderedGroup_ZF_1_L4(2) by simp
          have "ùü≠‚àà‚Ñ§" using Int_ZF_1_L8A(2) by simp
          { fix t assume  "t‚ààG"
            with assms(1) ‚Äπm‚àà‚Ñ§‚Ä∫ ‚Äπùü≠‚àà‚Ñ§‚Ä∫ mod0(1) ‚ÄπS‚á©1`(m) = S‚á©2`(m)‚Ä∫ have 
              "S‚á©1`(m\<ra>ùü≠)`t = f`‚ü®(S‚á©2`(m))`(t),t‚ü©"
              using module0.module_ax2 module0.module_ax4 by simp
            moreover from mod0(2) ‚Äπm‚àà‚Ñ§‚Ä∫ ‚Äπùü≠‚àà‚Ñ§‚Ä∫ ‚Äπt‚ààG‚Ä∫ have

          moreover from mod0 ‚Äπùü≠‚àà‚Ñ§‚Ä∫ ‚Äπm‚àà‚Ñ§‚Ä∫ have 
            "S‚á©1`(m\<ra>ùü≠):G‚ÜíG" and "S‚á©2`(m\<ra>ùü≠):G‚ÜíG"

    unfolding EndAdd_def IsAgroup_def IsAmonoid_def IsAssociative_def

lemma (in abgroup_int1) group_action_int_mult_morphism:
  assumes "r‚àà‚Ñ§" "s‚àà‚Ñ§"
  shows "ùíÆ`(r‚ãÖ‚á©Zs) = EndMult(G,P)`‚ü®ùíÆ`(r),ùíÆ`(s)‚ü©"
proof -  
  from assms have 
    "ùíÆ`(r) ‚àà End(G,P)" "ùíÆ`(s) ‚àà End(G,P)" "ùíÆ`(r‚ãÖ‚á©Zs) ‚àà End(G,P)"
    using powz_maps_int_End ints.int_prod_type apply_funtype 
    by simp_all
  from assms have 
    I: "ùíÆ`(r‚ãÖ‚á©Zs) = {‚ü®x,powz(s‚ãÖ‚á©Zr,x)‚ü©. x‚ààG}" 
    "ùíÆ`(r) = {‚ü®x,powz(r,x)‚ü©. x‚ààG}" "ùíÆ`(s) = {‚ü®x,powz(s,x)‚ü©. x‚ààG}"
    using ints.int_prod_type ZF_fun_from_tot_val1 ints.Int_ZF_1_L4(2) 
    by simp_all
  from assms(1,2) have "{‚ü®x,powz(s‚ãÖ‚á©Zr,x)‚ü©. x‚ààG} = {‚ü®x,powz(r,powz(s,x))‚ü©. x‚ààG}" 
    using powz_mult by simp
  with assms(1,2) I have "ùíÆ`(r‚ãÖ‚á©Zs) = ùíÆ`(r) O ùíÆ`(s)"
    using powz_type comp_fun_expr by force
  with ‚ÄπùíÆ`(r) ‚àà End(G,P)‚Ä∫ ‚ÄπùíÆ`(s) ‚àà End(G,P)‚Ä∫ show ?thesis
    unfolding EndMult_def using inend_composition_val by simp
qed

text‚ÄπThe action maps the integer $1$ to the identity, i.e. the neutral element of the
  composition of endomorphisms (which is the multiplication operation of the ring of 
  endomorphisms of the group).‚Ä∫

lemma (in abgroup_int1) group_action_int_unit:
  shows "ùíÆ`(ùü≠‚á©Z) = TheNeutralElement(End(G,P),EndMult(G,P))"
proof -
  have "TheNeutralElement(End(G,P),EndMult(G,P)) = id(G)"
    using end_comp_monoid(2) unfolding EndMult_def by blast
  then show ?thesis
    using ints.Int_ZF_1_L8A(2) ZF_fun_from_tot_val1 int_power_zero_one(2) id_diagonal
      by simp
  qed

text‚ÄπThe function from integers to endomorphisms of $G$ defined by $z\mapsto (x\mapsto x^z)$ 
  is a module.‚Ä∫

theorem (in abgroup_int1) group_action_int:
  shows "IsLeftModule(‚Ñ§,IntegerAddition,IntegerMultiplication,G,P,ùíÆ)"
  using groupAssum isAbelian ints.Int_ZF_1_1_L2 powz_maps_int_End
    group_action_int_add_morphism_fun group_action_int_mult_morphism
    group_action_int_unit
  unfolding IsMorphism_def IsRingHomomor_def IsAction_def IsLeftModule_def
    by simp
