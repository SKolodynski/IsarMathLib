lemma(in abelian_group) group_action_int_add_morphism:
  defines "S ‚â° {‚ü®z,{‚ü®x,powz(z,x)‚ü©. x‚ààG}‚ü©. z‚ààint}"
  shows "‚àÄr‚ààint. ‚àÄs‚ààint. ‚àÄg‚ààG. S ` (IntegerAddition` ‚ü®r, s‚ü©) ` g = P ` ‚ü®(S ` r) ` g, (S ` s) ` g‚ü©"
proof(safe)
  have S_fun:"S:int‚ÜíEnd(G,P)" using int0_abgroup.powz_maps_int_End
    unfolding S_def powz_def by blast
  fix r s g assume as:"r:int" "s:int" "g:G"
  from as(1) int_cases obtain n where n:"n‚àànat" "r=$#n ‚à® r=$- $# succ(n)" by blast
  from as(2) int_cases obtain m where m:"m‚àànat" "s=$#m ‚à® s=$- $# succ(m)" by blast
  from as(1,2) have "‚ü®r,s‚ü©‚ààint√óint" unfolding Sigma_def by blast
  then have int:"IntegerAddition`‚ü®r, s‚ü©‚ààint" 
    using apply_type[of IntegerAddition "int√óint" "Œª_. int" "‚ü®r,s‚ü©"]
    by (simp only: Int_ZF_1_L1(1))
  then have "‚ü®IntegerAddition`‚ü®r,s‚ü©,{‚ü®x,powz(IntegerAddition`‚ü®r,s‚ü©,x)‚ü©. x‚ààG}‚ü©‚ààS"
    unfolding S_def by blast
  then have Q:"S`(IntegerAddition`‚ü®r,s‚ü©) = {‚ü®x,powz(IntegerAddition`‚ü®r,s‚ü©,x)‚ü©. x‚ààG}"
    using apply_equality[OF _ S_fun] by blast
  have SS_fun:"S`(IntegerAddition`‚ü®r,s‚ü©):G‚ÜíG"
    using apply_type[OF S_fun int] unfolding End_def by (simp only:)
  have SS_r_fun:"S`r:G‚ÜíG"
    using apply_type[OF S_fun as(1)] unfolding End_def by (simp only:)
  have SS_s_fun:"S`s:G‚ÜíG"
    using apply_type[OF S_fun as(2)] unfolding End_def by (simp only:)
  {
    fix x assume "x‚ààG"
    then have "powz(IntegerAddition`‚ü®r,s‚ü©,x) = powz(r,x)‚ãÖpowz(s,x)"
      using int0_abgroup.powz_hom_prop[of r s x] as(1,2) unfolding powz_def by blast
  }
  then have R:"‚ãÄy. y‚ààG ‚üπ powz(IntegerAddition`‚ü®r,s‚ü©,y) = powz(r,y)‚ãÖpowz(s,y)" by blast
  {
    fix t assume as:"t‚àà{‚ü®x,powz(IntegerAddition`‚ü®r,s‚ü©,x)‚ü©. x‚ààG}"
    then obtain tx where t:"t=‚ü®tx,powz(IntegerAddition`‚ü®r,s‚ü©,tx)‚ü©" "tx‚ààG"
      by blast
    from t(1) have "t=‚ü®tx,powz(r,tx)‚ãÖpowz(s,tx)‚ü©" by (simp only:t(2) R[of tx])
    then have "t‚àà{‚ü®x,powz(r,x)‚ãÖpowz(s,x)‚ü©. x‚ààG}" using t(2) by blast
  }
  then have A1:"{‚ü®x,powz(IntegerAddition`‚ü®r,s‚ü©,x)‚ü©. x‚ààG} ‚äÜ {‚ü®x,powz(r,x)‚ãÖpowz(s,x)‚ü©. x‚ààG}" by blast
  {
    fix t assume as:"t‚àà{‚ü®x,powz(r,x)‚ãÖpowz(s,x)‚ü©. x‚ààG}"
    then obtain tx where t:"t=‚ü®tx,powz(r,tx)‚ãÖpowz(s,tx)‚ü©" "tx‚ààG"
      by blast
    from t(1) have "t=‚ü®tx,powz(IntegerAddition`‚ü®r,s‚ü©,tx)‚ü©" by (simp only:t(2) R[of tx])
    then have "t‚àà{‚ü®x,powz(IntegerAddition`‚ü®r,s‚ü©,x)‚ü©. x‚ààG}" using t(2) by blast
  }
  then have A2:"{‚ü®x,powz(r,x)‚ãÖpowz(s,x)‚ü©. x‚ààG} ‚äÜ {‚ü®x,powz(IntegerAddition`‚ü®r,s‚ü©,x)‚ü©. x‚ààG}"
    by blast
  from A1 A2 have A:"{‚ü®x,powz(IntegerAddition`‚ü®r,s‚ü©,x)‚ü©. x‚ààG} = {‚ü®x,powz(r,x)‚ãÖpowz(s,x)‚ü©. x‚ààG}"
    by blast
  have E:"S`(IntegerAddition`‚ü®r,s‚ü©) = {‚ü®x,powz(r,x)‚ãÖpowz(s,x)‚ü©. x‚ààG}"
    using Q by (simp only: A)
  have "‚ü®g,powz(r,g)‚ãÖpowz(s,g)‚ü©‚àà{‚ü®x,powz(r,x)‚ãÖpowz(s,x)‚ü©. x‚ààG}"
    using as(3) by blast
  then have "‚ü®g,powz(r,g)‚ãÖpowz(s,g)‚ü©‚àà S`(IntegerAddition`‚ü®r,s‚ü©)"
    by (simp only: E)
  from apply_equality[OF this SS_fun] have B:"S`(IntegerAddition`‚ü®r,s‚ü©)`g = powz(r,g)‚ãÖpowz(s,g)"
    by blast
  have "‚ü®r,{‚ü®x,powz(r,x)‚ü©. x‚ààG}‚ü©‚ààS"
    unfolding S_def using as(1) by blast
  then have Qr:"S`r = {‚ü®x,powz(r,x)‚ü©. x‚ààG}"
    using apply_equality[OF _ S_fun] by blast
  have "‚ü®g,powz(r,g)‚ü©‚àà{‚ü®x,powz(r,x)‚ü©. x‚ààG}" using as(3) by blast
  then have "‚ü®g,powz(r,g)‚ü©‚àà S`r"
    by (simp only: Qr)
  from apply_equality[OF this SS_r_fun]
    have C:"S`r`g = powz(r,g)".
  have "‚ü®s,{‚ü®x,powz(s,x)‚ü©. x‚ààG}‚ü©‚ààS"
    unfolding S_def using as(2) by blast
  then have Qr:"S`s = {‚ü®x,powz(s,x)‚ü©. x‚ààG}"
    using apply_equality[OF _ S_fun] by blast
  have "‚ü®g,powz(s,g)‚ü©‚àà{‚ü®x,powz(s,x)‚ü©. x‚ààG}" using as(3) by blast
  then have "‚ü®g,powz(s,g)‚ü©‚àà S`s"
    by (simp only: Qr)
  from apply_equality[OF this SS_s_fun]
    have D:"S`s`g = powz(s,g)".
  from B show "S ` (IntegerAddition` ‚ü®r, s‚ü©) ` g = P ` ‚ü®(S ` r) ` g, (S ` s) ` g‚ü©"
    unfolding groper_def by (simp only:C D)
qed
