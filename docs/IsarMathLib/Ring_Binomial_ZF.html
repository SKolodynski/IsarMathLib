section ‚ÄπBinomial theorem‚Ä∫

theory Ring_Binomial_ZF imports Monoid_ZF_1 Ring_ZF

begin

text‚ÄπThis theory aims at formalizing sufficient background to be able to state and prove the
  binomial theorem. ‚Ä∫

subsection‚ÄπSums of multiplicities of powers of ring elements and binomial theorem‚Ä∫

text‚ÄπThe binomial theorem asserts that for any two elements of a commutative ring
  the n-th power of the sum $x+y$ can be written as a sum of certain multiplicities of terms 
  $x^{n-k}y^k$, where $k \in 0..n$. In this section we setup the notation and prove basic properties
  of such multiplicities and powers of ring elements. We show the binomial theorem as 
  an application. ‚Ä∫

text‚ÄπThe next locale (context) extends the ‚Äπring0‚Ä∫ locale with notation for powers, 
  multiplicities and sums and products of finite lists of ring elements.‚Ä∫

locale ring3 = ring0 +
  fixes listsum ("‚àë _" 70)
  defines listsum_def [simp]: "‚àës ‚â° Fold(A,ùü¨,s)"

  fixes listprod ("‚àè _" 70)
  defines listprod_def [simp]: "‚àès ‚â° Fold(M,ùü≠,s)"

  fixes nat_mult (infix "\<nm>" 95)
  defines nat_mult_def [simp]: "n\<nm>x ‚â° ‚àë{‚ü®k,x‚ü©. k‚ààn}"

  fixes pow
  defines pow_def [simp]: "pow(n,x) ‚â° ‚àè{‚ü®k,x‚ü©. k‚ààn}"

text‚ÄπA ring with addition forms a monoid, hence all propositions proven in the ‚Äπmonoid1‚Ä∫ locale
  (defined in the ‚ÄπMonoid_ZF_1‚Ä∫ theory) can be used in the ‚Äπring3‚Ä∫ locale, applied to the 
  additive operation. ‚Ä∫

sublocale ring3 < add_monoid: monoid1 R A ringa ringzero listsum nat_mult
  using ringAssum 
  unfolding IsAring_def IsAgroup_def monoid1_def monoid0_def 
  by auto

text‚ÄπA ring with multiplication forms a monoid, hence all propositions proven in the ‚Äπmonoid1‚Ä∫ locale
  (defined in the ‚ÄπMonoid_ZF_1‚Ä∫ theory) can be used in the ‚Äπring3‚Ä∫ locale, applied to the
  multiplicative operation. ‚Ä∫

sublocale ring3 < mul_monoid: monoid1 R M ringm ringone listprod pow
  using ringAssum 
  unfolding IsAring_def IsAgroup_def monoid1_def monoid0_def 
  by auto

text‚Äπ$0\cdot x = 0$ and $x^0=1$. It is a bit surprising that we do not need to assume
  that $x\in R$ (i.e. $x$ is an element of the ring). These properties are really proven in the ‚ÄπMonoid_ZF_1‚Ä∫ 
  theory where there is no assumption that $x$ is an element of the monoid. ‚Ä∫

lemma (in ring3) mult_pow_zero: shows "0\<nm>x = ùü¨" and "pow(0,x) = ùü≠"
  using add_monoid.nat_mult_zero mul_monoid.nat_mult_zero by simp_all

text‚ÄπNatural multiple and power of a ring element is a ring element.‚Ä∫

lemma (in ring3) mult_pow_type: assumes "n‚àànat" "x‚ààR"
  shows "n\<nm>x ‚àà R" and "pow(n,x) ‚àà R"
  using assms add_monoid.nat_mult_type mul_monoid.nat_mult_type 
  by simp_all

text‚ÄπThe usual properties of multiples and powers: $(n+1)x = nx+x$ and 
  $x^n+1=x^n x$. These are just versions of ‚Äπnat_mult_add_one‚Ä∫ from ‚ÄπMonoid_ZF_1‚Ä∫
  writtent in the notation defined in the ‚Äπring3‚Ä∫ locale.‚Ä∫

lemma (in ring3) nat_mult_pow_add_one: assumes  "n‚àànat" "x‚ààR"
  shows "(n #+ 1)\<nm>x = (n\<nm>x) \<ra> x" and "pow(n #+ 1,x) = pow(n,x)‚ãÖx"
  using assms add_monoid.nat_mult_add_one mul_monoid.nat_mult_add_one 
  by simp_all

text‚ÄπAssociativity for the multiplication by natural number and the ring multiplication:‚Ä∫

lemma (in ring3) nat_mult_assoc: assumes "n‚àànat" "x‚ààR" "y‚ààR"
  shows "n\<nm>x‚ãÖy = n\<nm>(x‚ãÖy)"
proof -
  from assms(1,3) have "n‚àànat" and "0\<nm>x‚ãÖy = 0\<nm>(x‚ãÖy)"
    using mult_pow_zero(1) Ring_ZF_1_L6 by simp_all
  moreover from assms(2,3) have "‚àÄk‚àànat. 
    k\<nm>x‚ãÖy = k\<nm>(x‚ãÖy) ‚ü∂ (k #+ 1)\<nm>x‚ãÖy = (k #+ 1)\<nm>(x‚ãÖy)"
    using nat_mult_pow_add_one(1) mult_pow_type ring_oper_distr(2) Ring_ZF_1_L4(3)
      by simp
  ultimately show ?thesis by (rule ind_on_nat1)
qed

text‚ÄπAddition of natural numbers is distributive with respect to natural multiple.
  This is essentially lemma ‚Äπnat_mult_add‚Ä∫ from ‚ÄπMonoid_ZF_1.thy‚Ä∫, just transferred
  to the ‚Äπring3‚Ä∫ locale.‚Ä∫

lemma (in ring3) nat_add_mult_distrib: assumes "n‚àànat" "m‚àànat" "x‚ààR"
  shows "(n #+ m)\<nm>x = n\<nm>x \<ra> m\<nm>x"
  using assms add_monoid.nat_mult_add by simp

text‚ÄπAssociativity for the multiplication by natural number and the ring multiplication
  extended to three elements of the ring:‚Ä∫

lemma (in ring3) nat_mult_assoc1: assumes "n‚àànat" "x‚ààR" "y‚ààR" "z‚ààR" 
  shows "n\<nm>x‚ãÖy‚ãÖz = n\<nm>(x‚ãÖy‚ãÖz)"
  using assms Ring_ZF_1_L4(3) nat_mult_assoc by simp

text‚ÄπWhen we multiply an expression whose value belongs to a ring by a ring element 
  and we get an expression whose value belongs to a ring.‚Ä∫

lemma (in ring3) mult_elem_ring_type: 
  assumes "n‚àànat" "x‚ààR" and "‚àÄk‚ààn. q(k) ‚àà R" 
  shows "‚àÄk‚ààn. q(k)‚ãÖx ‚àà R" and "(‚àë{‚ü®k,q(k)‚ãÖx‚ü©. k‚ààn}) ‚àà R"
  using assms Ring_ZF_1_L4(3) add_monoid.sum_in_mono by simp_all

text‚ÄπThe sum of expressions whose values belong to a ring is an expression
  whose value belongs to a ring. ‚Ä∫

lemma (in ring3) sum_expr_ring_type: 
  assumes "n‚àànat" "‚àÄk‚ààn. q(k) ‚àà R" "‚àÄk‚ààn. p(k) ‚àà R"
  shows "‚àÄk‚ààn. q(k)\<ra>p(k) ‚àà R" and "(‚àë{‚ü®k,q(k)\<ra>p(k)‚ü©. k‚ààn}) ‚àà R"
  using assms Ring_ZF_1_L4(1) add_monoid.sum_in_mono by simp_all

text‚ÄπCombining ‚Äπmult_elem_ring_type‚Ä∫ and ‚Äπsum_expr_ring_type‚Ä∫ we obtain that
  a (kind of) linear combination of expressions whose values belong to a ring
  belongs to the ring. ‚Ä∫

lemma (in ring3) lin_comb_expr_ring_type:
  assumes "n‚àànat" "x‚ààR"  "y‚ààR" "‚àÄk‚ààn. q(k) ‚àà R" "‚àÄk‚ààn. p(k) ‚àà R"
  shows "‚àÄk‚ààn. q(k)‚ãÖx\<ra>p(k)‚ãÖy ‚àà R" and 
    "(‚àë{‚ü®k,q(k)‚ãÖx\<ra>p(k)‚ãÖy‚ü©. k‚ààn}) ‚àà R"
proof -
  from assms have "‚àÄk‚ààn. q(k)‚ãÖx ‚àà R" and "‚àÄk‚ààn. p(k)‚ãÖy ‚àà R"
    using mult_elem_ring_type(1) by simp_all
  with assms(1) show "‚àÄk‚ààn. q(k)‚ãÖx\<ra>p(k)‚ãÖy ‚àà R" 
    using sum_expr_ring_type(1) by simp
  with assms(1) show "(‚àë{‚ü®k,q(k)‚ãÖx\<ra>p(k)‚ãÖy‚ü©. k‚ààn}) ‚àà R" 
    using add_monoid.sum_in_mono by simp
qed

text‚ÄπA ‚Äπring3‚Ä∫ version of ‚Äπseq_sum_pull_one_elem‚Ä∫ from ‚ÄπMonoid_ZF_1‚Ä∫: ‚Ä∫

lemma (in ring3) rng_seq_sum_pull_one_elem:
  assumes "j ‚àà nat" "‚àÄk‚ààj #+ 1. q(k) ‚àà R"
  shows
    "(‚àë{‚ü®k,q(k)‚ü©. k‚ààj #+ 1}) = q(0)\<ra>(‚àë{‚ü®k,q(k #+ 1)‚ü©. k‚ààj})"
    "(‚àë{‚ü®k,q(k)‚ü©. k‚ààj #+ 1}) = (‚àë{‚ü®k,q(k)‚ü©. k‚ààj})\<ra> q(j)"
  using assms add_monoid.seq_sum_pull_one_elem by simp_all

text‚ÄπDistributive laws for finite sums in a ring: 
  $(\sum_{k=0}^{n-1}q(k))\cdot x = \sum_{k=0}^{n-1}q(k)\cdot x$ and 
  $x\cdot (\sum_{k=0}^{n-1}q(k)) = \sum_{k=0}^{n-1}x\cdot q(k)$. ‚Ä∫

theorem (in ring3) fin_sum_distrib: 
  assumes "x‚ààR"  "n‚àànat" "‚àÄk‚ààn. q(k) ‚àà R" 
  shows 
    "(‚àë{‚ü®k,q(k)‚ü©. k‚ààn})‚ãÖx = ‚àë{‚ü®k,q(k)‚ãÖx‚ü©. k‚ààn}"
    "x‚ãÖ(‚àë{‚ü®k,q(k)‚ü©. k‚ààn}) = ‚àë{‚ü®k,x‚ãÖq(k)‚ü©. k‚ààn}"
proof -
  from assms(1,2) have "n‚àànat" and 
    "(‚àë{‚ü®k,q(k)‚ü©. k‚àà0})‚ãÖx = ‚àë{‚ü®k,q(k)‚ãÖx‚ü©. k‚àà0}"
    using add_monoid.sum_empty Ring_ZF_1_L6(1) by simp_all
  moreover have 
    "‚àÄj‚ààn. (‚àë{‚ü®k,q(k)‚ü©. k‚ààj})‚ãÖx = (‚àë{‚ü®k,q(k)‚ãÖx‚ü©. k‚ààj})
    ‚ü∂ (‚àë{‚ü®k,q(k)‚ü©. k‚ààj #+ 1})‚ãÖx = ‚àë{‚ü®k,q(k)‚ãÖx‚ü©. k‚ààj #+ 1}"
  proof -
    { fix j assume "j‚ààn" and 
        I: "(‚àë{‚ü®k,q(k)‚ü©. k‚ààj})‚ãÖx = (‚àë{‚ü®k,q(k)‚ãÖx‚ü©. k‚ààj})" 
      from assms(2) ‚Äπj‚ààn‚Ä∫ have "j‚àànat" using elem_nat_is_nat(2) 
        by simp
      moreover from assms(2,3) ‚Äπj‚ààn‚Ä∫ have II: "‚àÄk‚ààj #+ 1. q(k) ‚àà R"
        using mem_add_one_subset by blast
      ultimately have     
        "(‚àë{‚ü®k,q(k)‚ü©. k‚ààj #+ 1}) =  (‚àë{‚ü®k,q(k)‚ü©. k‚ààj}) \<ra> q(j)"
        using add_monoid.seq_sum_pull_one_elem(2) by simp
      hence 
        "(‚àë{‚ü®k,q(k)‚ü©. k‚ààj #+ 1})‚ãÖx = ((‚àë{‚ü®k,q(k)‚ü©. k‚ààj}) \<ra> q(j))‚ãÖx"
        by simp
      moreover from assms(1) ‚Äπj‚àànat‚Ä∫ II have
        "(‚àë{‚ü®k,q(k)‚ü©. k‚ààj}) ‚àà R" "q(j) ‚àà R" and "x‚ààR" 
        using add_monoid.sum_in_mono by simp_all
      ultimately have 
        "(‚àë{‚ü®k,q(k)‚ü©. k‚ààj #+ 1})‚ãÖx = (‚àë{‚ü®k,q(k)‚ü©. k‚ààj})‚ãÖx \<ra> q(j)‚ãÖx"
        using ring_oper_distr(2) by simp
      with I have 
        "(‚àë{‚ü®k,q(k)‚ü©. k‚ààj #+ 1})‚ãÖx = (‚àë{‚ü®k,q(k)‚ãÖx‚ü©. k‚ààj})  \<ra> q(j)‚ãÖx" 
        by simp
      moreover 
      from assms(1) II have "‚àÄk‚ààj #+ 1. q(k)‚ãÖx ‚àà R"
        using Ring_ZF_1_L4(3) by simp
      with ‚Äπj‚àànat‚Ä∫ have 
        "(‚àë{‚ü®k,q(k)‚ãÖx‚ü©. k‚ààj #+ 1}) = (‚àë{‚ü®k,q(k)‚ãÖx‚ü©. k‚ààj}) \<ra> q(j)‚ãÖx"
        using add_monoid.seq_sum_pull_one_elem(2) by simp
      ultimately have 
        "(‚àë{‚ü®k,q(k)‚ü©. k‚ààj #+ 1})‚ãÖx = (‚àë{‚ü®k,q(k)‚ãÖx‚ü©. k‚ààj #+ 1})"
        by simp
    } thus ?thesis by simp
  qed
  ultimately show "(‚àë{‚ü®k,q(k)‚ü©. k‚ààn})‚ãÖx = ‚àë{‚ü®k,q(k)‚ãÖx‚ü©. k‚ààn}"
    by (rule fin_nat_ind1)
  from assms(1,2) have "n‚àànat" and 
    "x‚ãÖ(‚àë{‚ü®k,q(k)‚ü©. k‚àà0}) = ‚àë{‚ü®k,x‚ãÖq(k)‚ü©. k‚àà0}"
    using add_monoid.sum_empty Ring_ZF_1_L6(2) by simp_all
  moreover have 
    "‚àÄj‚ààn. x‚ãÖ(‚àë{‚ü®k,q(k)‚ü©. k‚ààj}) = (‚àë{‚ü®k,x‚ãÖq(k)‚ü©. k‚ààj})
    ‚ü∂ x‚ãÖ(‚àë{‚ü®k,q(k)‚ü©. k‚ààj #+ 1}) = ‚àë{‚ü®k,x‚ãÖq(k)‚ü©. k‚ààj #+ 1}" 
  proof -
    { fix j assume "j‚ààn" and 
        I: "x‚ãÖ(‚àë{‚ü®k,q(k)‚ü©. k‚ààj}) = (‚àë{‚ü®k,x‚ãÖq(k)‚ü©. k‚ààj})"
      from assms(2) ‚Äπj‚ààn‚Ä∫ have "j‚àànat" using elem_nat_is_nat(2) 
        by simp
      moreover from assms(2,3) ‚Äπj‚ààn‚Ä∫ have II: "‚àÄk‚ààj #+ 1. q(k) ‚àà R"
        using mem_add_one_subset by blast
      ultimately have     
        "(‚àë{‚ü®k,q(k)‚ü©. k‚ààj #+ 1}) =  (‚àë{‚ü®k,q(k)‚ü©. k‚ààj}) \<ra> q(j)"
        using add_monoid.seq_sum_pull_one_elem(2) by simp
      hence 
        "x‚ãÖ(‚àë{‚ü®k,q(k)‚ü©. k‚ààj #+ 1}) = x‚ãÖ((‚àë{‚ü®k,q(k)‚ü©. k‚ààj}) \<ra> q(j))"
        by simp
      moreover from assms(1) ‚Äπj‚àànat‚Ä∫ II have
        "(‚àë{‚ü®k,q(k)‚ü©. k‚ààj}) ‚àà R" "q(j) ‚àà R" and "x‚ààR" 
        using add_monoid.sum_in_mono by simp_all
       ultimately have 
        "x‚ãÖ(‚àë{‚ü®k,q(k)‚ü©. k‚ààj #+ 1}) = x‚ãÖ(‚àë{‚ü®k,q(k)‚ü©. k‚ààj}) \<ra> x‚ãÖq(j)"
        using ring_oper_distr(1) by simp
       with I have 
        "x‚ãÖ(‚àë{‚ü®k,q(k)‚ü©. k‚ààj #+ 1}) =  (‚àë{‚ü®k,x‚ãÖq(k)‚ü©. k‚ààj})  \<ra> x‚ãÖq(j)" 
        by simp
      moreover 
      from assms(1) II have "‚àÄk‚ààj #+ 1. x‚ãÖq(k) ‚àà R"
        using Ring_ZF_1_L4(3) by simp
      with ‚Äπj‚àànat‚Ä∫ have 
        "(‚àë{‚ü®k,x‚ãÖq(k)‚ü©. k‚ààj #+ 1}) = (‚àë{‚ü®k,x‚ãÖq(k)‚ü©. k‚ààj}) \<ra> x‚ãÖq(j)"
        using add_monoid.seq_sum_pull_one_elem(2) by simp
      ultimately have 
        "x‚ãÖ(‚àë{‚ü®k,q(k)‚ü©. k‚ààj #+ 1}) = (‚àë{‚ü®k,x‚ãÖq(k)‚ü©. k‚ààj #+ 1})"
        by simp
    } thus ?thesis by simp
  qed
  ultimately show  "x‚ãÖ(‚àë{‚ü®k,q(k)‚ü©. k‚ààn}) = ‚àë{‚ü®k,x‚ãÖq(k)‚ü©. k‚ààn}"
    by (rule fin_nat_ind1)
qed

text‚ÄπIn rings we have 
  $\sum_{k=0}^{n-1}q(k) + p(k) = (\sum_{k=0}^{n-1} p(k)) + (\sum_{k=0}^{n-1} q(k))$. 
  This is the same as theorem ‚Äπsum_comm_distrib‚Ä∫ in ‚ÄπMonoid_ZF_1.thy‚Ä∫, except that
  we do not need the assumption about commutativity of the operation as addition in rings
  is always commutative. ‚Ä∫

lemma (in ring3) sum_ring_distrib: 
  assumes "n‚àànat" and  "‚àÄk‚ààn. p(k) ‚àà R" "‚àÄk‚ààn. q(k) ‚àà R"
  shows
    "(‚àë{‚ü®k,p(k)\<ra>q(k)‚ü©. k‚ààn}) = (‚àë{‚ü®k,p(k)‚ü©. k‚ààn}) \<ra> (‚àë{‚ü®k,q(k)‚ü©. k‚ààn})"
  using assms Ring_ZF_1_L1(3) add_monoid.sum_comm_distrib by simp

text‚ÄπTo shorten the notation in the proof of the binomial theorem we give a name to the
  binomial term ${n \choose k} x^{n-k} y^k$.‚Ä∫

definition (in ring3) BT where
  "BT(n,k,x,y) ‚â° Binom(n,k)\<nm>pow(n #- k,x)‚ãÖpow(k,y)"

text‚ÄπIf $n,k$ are natural numbers and $x,y$ are ring elements then the binomial term is 
  an element of the ring. ‚Ä∫

lemma (in ring3) bt_type: assumes "n‚àànat" "k‚àànat" "x‚ààR" "y‚ààR" 
  shows "BT(n,k,x,y) ‚àà R"
  using assms mult_pow_type binom_in_nat Ring_ZF_1_L4(3)
  unfolding BT_def by simp

text‚ÄπThe binomial term is $1$ when the $n=0$ and $k=0$. 
  Somehow we do not need the assumption that $x,y$ are ring elements. ‚Ä∫

lemma (in ring3) bt_at_zero: shows "BT(0,0,x,y) = ùü≠"
  using binom_zero_zero mult_pow_zero(2) add_monoid.nat_mult_one 
        Ring_ZF_1_L2(2) Ring_ZF_1_L3(5)
  unfolding BT_def by simp

text‚ÄπThe binomial term is $x^n$ when $k=0$. ‚Ä∫

lemma (in ring3) bt_at_zero1: assumes "n‚àànat" "x‚ààR"
  shows "BT(n,0,x,y) = pow(n,x)" 
  unfolding BT_def using assms mult_pow_zero(2) binom_left_boundary
    mult_pow_type(2) add_monoid.nat_mult_one Ring_ZF_1_L3(5) 
    by simp

text‚ÄπWhen $k=0$ multiplying the binomial term by $x$ is the same as adding one to $n$. ‚Ä∫

lemma (in ring3) bt_at_zero2: assumes "n‚àànat" "x‚ààR"
  shows "BT(n,0,x,y)‚ãÖx = BT(n #+ 1,0,x,y)"
  using assms bt_at_zero1 nat_mult_pow_add_one(2) by simp

text‚ÄπThe binomial term is $y^n$ when $k=n$.‚Ä∫

lemma (in ring3) bt_at_right: assumes "n‚àànat" "y‚ààR"
  shows "BT(n,n,x,y) = pow(n,y)" 
  unfolding BT_def using assms binom_right_boundary mult_pow_zero(2)
    add_monoid.nat_mult_one Ring_ZF_1_L2(2) mult_pow_type(2) Ring_ZF_1_L3(6)
  by simp

text‚ÄπWhen $k=n$ multiplying the binomial term by $x$ is the same as adding one to $n$. ‚Ä∫

lemma (in ring3) bt_at_right1: assumes "n‚àànat" "y‚ààR"
  shows "BT(n,n,x,y)‚ãÖy = BT(n #+ 1,n #+ 1,x,y)"
  using assms bt_at_right nat_mult_pow_add_one(2) by simp

text‚ÄπA key identity for binomial terms needed for the proof of the binomial theorem:‚Ä∫

lemma (in ring3) bt_rec_identity: 
  assumes "M {is commutative on} R" "j‚àànat" "k‚ààj" "x‚ààR" "y‚ààR"
  shows
