section ‹Direct product of groups›

theory DirectProduct_ZF_1 imports DirectProduct_ZF Group_ZF

begin

text‹This theory proves that the direct product of two groups is a group.
  The result lives in the ‹direct0› locale, which already establishes the
  direct product as a binary operation and proves it inherits associativity and
  commutativity.  Here we add the group structure.›

subsection‹Neutral element and inverse of the direct product›

text‹The neutral element of the direct product is the pair of neutral elements.›

lemma (in direct0) direct_product_neutral:
  assumes GP: "IsAgroup(G,P)" and HQ: "IsAgroup(H,Q)"
  shows "TheNeutralElement(G×H,R) =
    ⟨TheNeutralElement(G,P), TheNeutralElement(H,Q)⟩"
proof -
  let ?eG = "TheNeutralElement(G,P)"
  let ?eH = "TheNeutralElement(H,Q)"
  interpret gG: group0 G P ?eG "λx y. P`⟨x,y⟩" "λx. GroupInv(G,P)`x"
    "λs. Fold(P,?eG,s)" "λn x. Fold(P,?eG,{⟨k,x⟩. k∈n})"
    using GP unfolding group0_def by auto
  interpret gH: group0 H Q ?eH "λx y. Q`⟨x,y⟩" "λx. GroupInv(H,Q)`x"
    "λs. Fold(Q,?eH,s)" "λn x. Fold(Q,?eH,{⟨k,x⟩. k∈n})"
    using HQ unfolding group0_def by auto
  from gG.group0_2_L2_1 have eG_in: "?eG ∈ G" by assumption 
  from gG.group0_2_L2_2 gG.group0_2_L2_3 have eG_neut:"∀g∈G. P`⟨?eG,g⟩ = g ∧ P`⟨g,?eG⟩ = g"
    by blast
  from gH.group0_2_L2_1 have eH_in: "?eH ∈ H" by assumption
    from gH.group0_2_L2_2 gH.group0_2_L2_3 have eH_neut: "∀h∈H. Q`⟨?eH,h⟩ = h ∧ Q`⟨h,?eH⟩ = h" by blast
  have e_in: "⟨?eG,?eH⟩ ∈ G×H" using eG_in eH_in by safe
  have neut_prop: "∀p∈G×H. R`⟨⟨?eG,?eH⟩,p⟩ = p ∧ R`⟨p,⟨?eG,?eH⟩⟩ = p"
  proof
    fix p assume "p ∈ G×H"
    then obtain g h where gr: "g ∈ G" and hr: "h ∈ H" and peq: "p = ⟨g,h⟩" by auto
    from gr hr e_in peq show "R`⟨⟨?eG,?eH⟩,p⟩ = p ∧ R`⟨p,⟨?eG,?eH⟩⟩ = p"
      using DirectProduct_ZF_1_L2 eG_neut eH_neut by (simp only:fst_conv snd_conv)
  qed
  have monoidR: "IsAmonoid(G×H,R)"
    using GP HQ e_in neut_prop DirectProduct_ZF_2_L2
    unfolding IsAgroup_def IsAmonoid_def by blast
  interpret pR: monoid0 "G×H" R "λa b. R`⟨a,b⟩"
    using monoidR unfolding monoid0_def by auto
  have "⟨?eG,?eH⟩ = TheNeutralElement(G×H,R)"
    using pR.group0_1_L4 e_in neut_prop by blast
  thus ?thesis by (simp only: sym)
qed

subsection‹The direct product of groups is a group›

text‹The direct product of two groups is a group.›

theorem (in direct0) direct_product_group:
  assumes GP: "IsAgroup(G,P)" and HQ: "IsAgroup(H,Q)"
  shows "IsAgroup(G×H, R)"
proof (rule definition_of_group)
  let ?eG = "TheNeutralElement(G,P)"
  let ?eH = "TheNeutralElement(H,Q)"
  interpret gG: group0 G P ?eG "λx y. P`⟨x,y⟩" "λx. GroupInv(G,P)`x"
    "λs. Fold(P,?eG,s)" "λn x. Fold(P,?eG,{⟨k,x⟩. k∈n})"
    using GP unfolding group0_def by auto
  interpret gH: group0 H Q ?eH "λx y. Q`⟨x,y⟩" "λx. GroupInv(H,Q)`x"
    "λs. Fold(Q,?eH,s)" "λn x. Fold(Q,?eH,{⟨k,x⟩. k∈n})"
    using HQ unfolding group0_def by auto
  from gG.group0_2_L2_1 have eG_in: "?eG ∈ G" by assumption 
  from gG.group0_2_L2_2 gG.group0_2_L2_3 have eG_neut:"∀g∈G. P`⟨?eG,g⟩ = g ∧ P`⟨g,?eG⟩ = g"
    by blast
  from gH.group0_2_L2_1 have eH_in: "?eH ∈ H" by assumption
    from gH.group0_2_L2_2 gH.group0_2_L2_3 have eH_neut: "∀h∈H. Q`⟨?eH,h⟩ = h ∧ Q`⟨h,?eH⟩ = h" by blast
  have e_in: "⟨?eG,?eH⟩ ∈ G×H" using eG_in eH_in by safe
  have neut_prop: "∀p∈G×H. R`⟨⟨?eG,?eH⟩,p⟩ = p ∧ R`⟨p,⟨?eG,?eH⟩⟩ = p"
  proof
    fix p assume "p ∈ G×H"
    then obtain g h where gr: "g ∈ G" and hr: "h ∈ H" and peq: "p = ⟨g,h⟩" by auto
    from gr hr e_in peq show "R`⟨⟨?eG,?eH⟩,p⟩ = p ∧ R`⟨p,⟨?eG,?eH⟩⟩ = p"
      using DirectProduct_ZF_1_L2 eG_neut eH_neut by (simp only:fst_conv snd_conv)
  qed
  show monoidR: "IsAmonoid(G×H,R)"
    using GP HQ e_in neut_prop DirectProduct_ZF_2_L2
    unfolding IsAgroup_def IsAmonoid_def by blast
  show "∀p∈G×H. ∃b∈G×H. R`⟨p,b⟩ = TheNeutralElement(G×H,R)"
  proof
    fix p assume "p ∈ G×H"
    then obtain g h where gr: "g ∈ G" and hr: "h ∈ H" and peq: "p = ⟨g,h⟩" by auto
    from gr have invG: "GroupInv(G,P)`g ∈ G" using gG.inverse_in_group by blast
    from hr have invH: "GroupInv(H,Q)`h ∈ H" using gH.inverse_in_group by blast
    have b_in: "⟨GroupInv(G,P)`g, GroupInv(H,Q)`h⟩ ∈ G×H" using invG invH by blast
    have "R`⟨p, ⟨GroupInv(G,P)`g, GroupInv(H,Q)`h⟩⟩ = TheNeutralElement(G×H,R)"
      using gr invG hr invH peq DirectProduct_ZF_1_L2
        gG.group0_2_L6[OF gr] gH.group0_2_L6[OF hr]
        direct_product_neutral[OF GP HQ] by (simp only:fst_conv snd_conv)
    with b_in show "∃b∈G×H. R`⟨p,b⟩ = TheNeutralElement(G×H,R)" by blast
  qed
qed

text‹The group inverse in the direct product is computed componentwise.›

lemma (in direct0) direct_product_inv:
  assumes GP: "IsAgroup(G,P)" and HQ: "IsAgroup(H,Q)"
    and gr: "g ∈ G" and hr: "h ∈ H"
  shows "GroupInv(G×H,R)`⟨g,h⟩ = ⟨GroupInv(G,P)`g, GroupInv(H,Q)`h⟩"
proof -
  let ?eG = "TheNeutralElement(G,P)"
  let ?eH = "TheNeutralElement(H,Q)"
  let ?eGH = "TheNeutralElement(G×H,R)"
  interpret gG: group0 G P ?eG "λx y. P`⟨x,y⟩" "λx. GroupInv(G,P)`x"
    "λs. Fold(P,?eG,s)" "λn x. Fold(P,?eG,{⟨k,x⟩. k∈n})"
    using GP  unfolding group0_def by auto
  interpret gH: group0 H Q ?eH "λx y. Q`⟨x,y⟩" "λx. GroupInv(H,Q)`x"
    "λs. Fold(Q,?eH,s)" "λn x. Fold(Q,?eH,{⟨k,x⟩. k∈n})"
    using HQ unfolding group0_def by auto
  interpret pGH: group0 "G×H" R ?eGH "λx y. R`⟨x,y⟩" "λx. GroupInv(G×H,R)`x"
    "λs. Fold(R,?eGH,s)" "λn x. Fold(R,?eGH,{⟨k,x⟩. k∈n})"
    using direct_product_group[OF GP HQ] unfolding group0_def by auto
  from gr have invG: "GroupInv(G,P)`g ∈ G" using gG.inverse_in_group by blast
  from hr have invH: "GroupInv(H,Q)`h ∈ H" using gH.inverse_in_group by blast
  have gh_in: "⟨g,h⟩ ∈ G×H" using gr hr by simp
  have b_in: "⟨GroupInv(G,P)`g, GroupInv(H,Q)`h⟩ ∈ G×H" using invG invH by blast
  have step: "R`⟨⟨g,h⟩, ⟨GroupInv(G,P)`g, GroupInv(H,Q)`h⟩⟩ = ?eGH"
    using gr invG hr invH DirectProduct_ZF_1_L2
      gG.group0_2_L6[OF gr] gH.group0_2_L6[OF hr]
      direct_product_neutral[OF GP HQ] by (simp only:fst_conv snd_conv)
  from gh_in b_in step have
    "⟨GroupInv(G,P)`g, GroupInv(H,Q)`h⟩ = GroupInv(G×H,R)`⟨g,h⟩"
    using pGH.group0_2_L9 by (simp only:)
  thus ?thesis by (simp only:sym)
qed

end
