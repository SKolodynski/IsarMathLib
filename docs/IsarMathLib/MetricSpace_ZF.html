section ‚Äπ Metric spaces ‚Ä∫

theory MetricSpace_ZF imports Topology_ZF_1 OrderedLoop_ZF Lattice_ZF
begin

text‚ÄπA metric space is a set on which a distance between points is defined as a function
  $d:X \times X \to [0,\infty)$. With this definition each metric space is a topological space 
  which is paracompact and Hausdorff ($T_2$), hence normal (in fact even perfectly normal).‚Ä∫

subsection‚Äπ Pseudometric - definition and basic properties ‚Ä∫

text‚ÄπA metric on $X$ is usually defined as a function $d:X \times X \to [0,\infty)$ that satisfies
  the conditions $d(x,x) = 0$, $d(x, y) = 0 \Rightarrow  x = y$ (identity of indiscernibles), 
  $d(x, y)  = d(y, x)$ (symmetry) and $d(x, y) \le d(x, z) + d(z, y)$ (triangle inequality) 
  for all $x,y \in X$.  Here we are going to be a bit more general and define metric and 
  pseudo-metric as a function valued in an ordered loop. ‚Ä∫

text‚Äπ First we define a pseudo-metric, which has the axioms of a metric, but without the second part
  of the identity of indiscernibles. In our definition ‚ÄπIsApseudoMetric‚Ä∫ is a predicate on five sets: the function $d$, 
  the set $X$ on which the metric is defined, the loop carrier $G$, the loop operation $A$ 
  and the order $r$ on $G$.‚Ä∫

definition 
  "IsApseudoMetric(d,X,G,A,r) ‚â° d:X√óX ‚Üí Nonnegative(G,A,r) 
    ‚àß (‚àÄx‚ààX. d`‚ü®x,x‚ü© = TheNeutralElement(G,A))
    ‚àß (‚àÄx‚ààX.‚àÄy‚ààX. d`‚ü®x,y‚ü© = d`‚ü®y,x‚ü©)
    ‚àß (‚àÄx‚ààX.‚àÄy‚ààX.‚àÄz‚ààX. ‚ü®d`‚ü®x,z‚ü©, A`‚ü®d`‚ü®x,y‚ü©,d`‚ü®y,z‚ü©‚ü©‚ü© ‚àà r)"

text‚ÄπWe add the full axiom of identity of indiscernibles to the definition of a pseudometric
  to get the definition of metric.‚Ä∫

definition 
  "IsAmetric(d,X,G,A,r) ‚â° 
    IsApseudoMetric(d,X,G,A,r) ‚àß (‚àÄx‚ààX.‚àÄy‚ààX. d`‚ü®x,y‚ü© = TheNeutralElement(G,A) ‚ü∂ x=y)"

text‚ÄπA disk is defined as set of points located less than the radius from the center.‚Ä∫

definition "Disk(X,d,r,c,R) ‚â° {x‚ààX. ‚ü®d`‚ü®c,x‚ü©,R‚ü© ‚àà StrictVersion(r)}"


text‚ÄπNext we define notation for metric spaces. We will reuse the additive notation defined in 
  the ‚Äπloop1‚Ä∫ locale adding only the assumption about $d$ being a pseudometric and notation
  for a disk centered at $c$ with radius $R$.
  Since for many theorems it is sufficient to assume the pseudometric axioms we will
  assume in this context that the sets $d,X,L,A,r$ form a pseudometric raher than a metric.‚Ä∫

locale pmetric_space =  loop1 +
  fixes d and X 

  assumes pmetricAssum: "IsApseudoMetric(d,X,L,A,r)"
  
  fixes disk
  defines disk_def [simp]: "disk(c,R) ‚â° Disk(X,d,r,c,R)"


text‚Äπ The next lemma shows the definition of the pseudometric in the notation used in the 
  ‚Äπmetric_space‚Ä∫ context.‚Ä∫

lemma (in pmetric_space) pmetric_properties: shows 
  "d: X√óX ‚Üí L‚áß+"
  "‚àÄx‚ààX. d`‚ü®x,x‚ü© = ùü¨"
  "‚àÄx‚ààX.‚àÄy‚ààX. d`‚ü®x,y‚ü© = d`‚ü®y,x‚ü©"
  "‚àÄx‚ààX.‚àÄy‚ààX.‚àÄz‚ààX. d`‚ü®x,z‚ü© \<lsq> d`‚ü®x,y‚ü© \<ra> d`‚ü®y,z‚ü©"
  using pmetricAssum unfolding IsApseudoMetric_def by auto

text‚ÄπThe values of the metric are in the loop.‚Ä∫

lemma (in pmetric_space) pmetric_loop_valued: assumes "x‚ààX" "y‚ààX"
  shows "d`‚ü®x,y‚ü© ‚àà L‚áß+" "d`‚ü®x,y‚ü© ‚àà L"

    by simp
  then show "d`‚ü®x,y‚ü© ‚àà L" using Nonnegative_def by auto
qed
  
text‚ÄπThe definition of the disk in the notation used in the ‚Äπpmetric_space‚Ä∫ context:‚Ä∫

lemma (in pmetric_space) disk_definition: shows "disk(c,R) = {x‚ààX. d`‚ü®c,x‚ü© \<ls> R}"
proof -
  have "disk(c,R) = Disk(X,d,r,c,R)" by simp
  then have "disk(c,R) = {x‚ààX. ‚ü®d`‚ü®c,x‚ü©,R‚ü© ‚àà StrictVersion(r)}" unfolding Disk_def by simp
  moreover have "‚àÄx‚ààX. ‚ü®d`‚ü®c,x‚ü©,R‚ü© ‚àà StrictVersion(r) ‚ü∑ d`‚ü®c,x‚ü© \<ls> R"
    using def_of_strict_ver by simp
  ultimately show ?thesis by auto
qed

text‚ÄπIf the radius is positive then the center is in disk.‚Ä∫

lemma (in pmetric_space) center_in_disk: assumes "c‚ààX" and "R‚ààL‚á©+" shows "c ‚àà disk(c,R)"
  using pmetricAssum assms IsApseudoMetric_def PositiveSet_def disk_definition by simp
  
text‚ÄπA technical lemma that allows us to shorten some proofs: ‚Ä∫

lemma (in pmetric_space) radius_in_loop: assumes "c‚ààX" and "x ‚àà disk(c,R)"
  shows "R‚ààL" "ùü¨\<ls>R" "R‚ààL‚á©+" "(\<rm>d`‚ü®c,x‚ü© \<ad> R) ‚àà L‚á©+"
proof -
  from assms(2) have "x‚ààX" and "d`‚ü®c,x‚ü© \<ls> R" using disk_definition by auto
  with assms(1) show "ùü¨\<ls>R" using pmetric_properties(1) apply_funtype 
      nonneg_definition loop_strict_ord_trans by blast
  then show "R‚ààL" and "R‚ààL‚á©+" using posset_definition PositiveSet_def by auto
  from ‚Äπd`‚ü®c,x‚ü© \<ls> R‚Ä∫ show "(\<rm>d`‚ü®c,x‚ü© \<ad> R) ‚àà L‚á©+"
    using ls_other_side(2) by simp
qed

text‚ÄπIf a point $x$ is inside a disk $B$ and $m\leq R-d(c,x)$ then the disk centered 
  at the point $x$ and with radius $m$ is contained in the disk $B$. ‚Ä∫

lemma (in pmetric_space) disk_in_disk: 
  assumes "c‚ààX"  and "x ‚àà disk(c,R)" and "m \<lsq> (\<rm>d`‚ü®c,x‚ü© \<ad> R)"
  shows "disk(x,m) ‚äÜ disk(c,R)"
proof
  fix y assume "y ‚àà disk(x,m)"
  then have "d`‚ü®x,y‚ü© \<ls> m" using disk_definition by simp
  from assms(1,2) ‚Äπy ‚àà disk(x,m)‚Ä∫ have "R‚ààL" "x‚ààX" "y‚ààX" 
    using radius_in_loop(1) disk_definition by auto
  with assms(1) have "d`‚ü®c,y‚ü© \<lsq> d`‚ü®c,x‚ü© \<ra> d`‚ü®x,y‚ü©" using pmetric_properties(4) by simp
  from assms(1) ‚Äπx‚ààX‚Ä∫ have "d`‚ü®c,x‚ü© ‚àà L" 
    using pmetric_properties(1) apply_funtype nonneg_subset by auto
  with ‚Äπd`‚ü®x,y‚ü© \<ls> m‚Ä∫ assms(3) have "d`‚ü®c,x‚ü© \<ra> d`‚ü®x,y‚ü© \<ls> d`‚ü®c,x‚ü© \<ra> (\<rm>d`‚ü®c,x‚ü© \<ad> R)"
    using loop_strict_ord_trans1 strict_ord_trans_inv(2) by blast
  with ‚Äπd`‚ü®c,x‚ü© ‚àà L‚Ä∫ ‚ÄπR‚ààL‚Ä∫ ‚Äπd`‚ü®c,y‚ü© \<lsq> d`‚ü®c,x‚ü© \<ra> d`‚ü®x,y‚ü©‚Ä∫ ‚Äπy‚ààX‚Ä∫ show "y ‚àà disk(c,R)"
    using lrdiv_props(6) loop_strict_ord_trans disk_definition by simp
qed

text‚Äπ If we assume that the order on the group makes the positive set a meet semi-lattice (i.e.
  every two-element subset of $G_+$ has a greatest lower bound) then 
  the collection of disks centered at points of the space and with radii in the positive set 
  of the group satisfies the base condition. The meet semi-lattice assumption can be weakened 
  to "each two-element subset of $G_+$ has a lower bound in $G_+$", but we don't do that here. ‚Ä∫

lemma (in pmetric_space) disks_form_base: 
  assumes "r {down-directs} L‚á©+"
  defines "B ‚â° ‚ãÉc‚ààX. {disk(c,R). R‚ààL‚á©+}"
  shows "B {satisfies the base condition}"
proof -
  { fix U V assume "U‚ààB" "V‚ààB"
    fix x assume "x‚ààU‚à©V"
    have "‚àÉW‚ààB. x‚ààW ‚àß W‚äÜU‚à©V"
    proof -
      from assms(2) ‚ÄπU‚ààB‚Ä∫ ‚ÄπV‚ààB‚Ä∫ obtain c‚á©U  c‚á©V R‚á©U  R‚á©V 
        where "c‚á©U ‚àà X" "R‚á©U ‚àà L‚á©+" "c‚á©V ‚àà X" "R‚á©V ‚àà L‚á©+" "U = disk(c‚á©U,R‚á©U)" "V = disk(c‚á©V,R‚á©V)"
        by auto
      with ‚Äπx‚ààU‚à©V‚Ä∫ have "x ‚àà disk(c‚á©U,R‚á©U)" and "x ‚àà disk(c‚á©V,R‚á©V)" by auto
      then have "x‚ààX" "d`‚ü®c‚á©U,x‚ü© \<ls> R‚á©U" "d`‚ü®c‚á©V,x‚ü© \<ls> R‚á©V" using disk_definition by auto
      let ?m‚á©U = "\<rm> d`‚ü®c‚á©U,x‚ü© \<ad> R‚á©U"
      let ?m‚á©V = "\<rm> d`‚ü®c‚á©V,x‚ü© \<ad> R‚á©V"
      from ‚Äπc‚á©U‚ààX‚Ä∫ ‚Äπx‚ààdisk(c‚á©U,R‚á©U)‚Ä∫ ‚Äπc‚á©V‚ààX‚Ä∫ ‚Äπx‚ààdisk(c‚á©V,R‚á©V)‚Ä∫ have "?m‚á©U‚ààL‚á©+" and "?m‚á©V‚ààL‚á©+" 
        using radius_in_loop(4) by auto
      with assms(1) obtain m where "m ‚àà L‚á©+" "m \<lsq> ?m‚á©U" "m \<lsq> ?m‚á©V"
        unfolding DownDirects_def by auto
      let ?W = "disk(x,m)"
      from ‚Äπm ‚àà L‚á©+‚Ä∫ ‚Äπm \<lsq> ?m‚á©U‚Ä∫ ‚Äπm \<lsq> ?m‚á©V‚Ä∫ 
        ‚Äπc‚á©U ‚àà X‚Ä∫ ‚Äπx ‚àà disk(c‚á©U,R‚á©U)‚Ä∫ ‚Äπc‚á©V ‚àà X‚Ä∫ ‚Äπx ‚àà disk(c‚á©V,R‚á©V)‚Ä∫ 
        ‚ÄπU = disk(c‚á©U,R‚á©U)‚Ä∫ ‚ÄπV = disk(c‚á©V,R‚á©V)‚Ä∫
        have "?W ‚äÜ U‚à©V" using disk_in_disk by blast
      moreover from assms(2) ‚Äπx‚ààX‚Ä∫ ‚Äπm ‚àà L‚á©+‚Ä∫ have "?W ‚àà B" and "x‚àà?W" using center_in_disk
        by auto
      ultimately show ?thesis by auto
    qed      
  } then show ?thesis unfolding SatisfiesBaseCondition_def by auto
qed

text‚ÄπDisks centered at points farther away than the sum of radii do not overlap. ‚Ä∫
lemma (in pmetric_space) far_disks: 
  assumes "x‚ààX" "y‚ààX"  "r‚á©x\<ra>r‚á©y \<lsq> d`‚ü®x,y‚ü©"
  shows "disk(x,r‚á©x)‚à©disk(y,r‚á©y) = 0"
proof -
  { assume "disk(x,r‚á©x)‚à©disk(y,r‚á©y) ‚â† 0"
    then obtain z where "z ‚àà disk(x,r‚á©x)‚à©disk(y,r‚á©y)" by auto
    then have "z‚ààX" and "d`‚ü®x,z‚ü© \<ra> d`‚ü®y,z‚ü© \<ls> r‚á©x\<ra>r‚á©y"
      using disk_definition add_ineq_strict by auto
    moreover from assms(1,2) ‚Äπz‚ààX‚Ä∫ have "d`‚ü®x,y‚ü© \<lsq> d`‚ü®x,z‚ü© \<ra> d`‚ü®y,z‚ü©"
      using pmetric_properties(3,4) by auto
    ultimately have "d`‚ü®x,y‚ü© \<ls> r‚á©x\<ra>r‚á©y" using loop_strict_ord_trans 
      by simp
    with assms(3) have False using loop_strict_ord_trans by auto
  } thus ?thesis by auto
qed

text‚Äπ If we have a loop element that is smaller than the distance between two points, then
  we can separate these points with disks.‚Ä∫

lemma (in pmetric_space) disjoint_disks:
  assumes "x‚ààX" "y‚ààX" "r‚á©x\<ls>d`‚ü®x,y‚ü©"
  shows "(\<rm>r‚á©x\<ad>(d`‚ü®x,y‚ü©)) ‚àà L‚á©+" and "disk(x,r‚á©x)‚à©disk(y,\<rm>r‚á©x\<ad>(d`‚ü®x,y‚ü©)) = 0"
proof -
  from assms(3) show "(\<rm>r‚á©x\<ad>(d`‚ü®x,y‚ü©)) ‚àà L‚á©+"
    using ls_other_side posset_definition1 by simp
  from assms(1,2,3) have "r‚á©x‚ààL" and "d`‚ü®x,y‚ü© ‚àà L" 
    using less_members(1) pmetric_loop_valued(2) by auto
  then have "r‚á©x\<ra>(\<rm>r‚á©x\<ad>(d`‚ü®x,y‚ü©)) = d`‚ü®x,y‚ü©" using lrdiv_props(6) by simp
  with assms(1,2) ‚Äπd`‚ü®x,y‚ü© ‚àà L‚Ä∫ show "disk(x,r‚á©x)‚à©disk(y,\<rm>r‚á©x\<ad>(d`‚ü®x,y‚ü©)) = 0"
    using loop_ord_refl far_disks by simp
qed

text‚ÄπUnions of disks form a topology, hence (pseudo)metric spaces are topological spaces. ‚Ä∫

theorem (in pmetric_space) pmetric_is_top: 
  assumes  "r {down-directs} L‚á©+" 
  defines "B ‚â° ‚ãÉc‚ààX. {disk(c,R). R‚ààL‚á©+}" 
  defines "T ‚â° {‚ãÉA. A ‚àà Pow(B)}"
  shows "T {is a topology}"  "B {is a base for} T"  "‚ãÉT = X"
proof -
  from assms  show  "T {is a topology}"  "B {is a base for} T" 
    using disks_form_base Top_1_2_T1 by auto
  then have "‚ãÉT = ‚ãÉB" using Top_1_2_L5 by simp
  moreover have "‚ãÉB = X"
  proof
    from assms(2) show "‚ãÉB ‚äÜ X" using disk_definition by auto
    { fix x assume "x‚ààX"
      from assms(1) obtain R where "R‚ààL‚á©+" unfolding DownDirects_def by blast
      with assms(2) ‚Äπx‚ààX‚Ä∫ have "x ‚àà ‚ãÉB" using center_in_disk by auto
    } thus "X ‚äÜ ‚ãÉB" by auto
  qed 
  ultimately show "‚ãÉT = X" by simp
qed

text‚ÄπDistance of different points is greater than zero. ‚Ä∫

lemma (in metric_space) dist_pos: assumes "x‚ààX" "y‚ààX" "x‚â†y"
  shows "ùü¨\<ls>d`‚ü®x,y‚ü©" "d`‚ü®x,y‚ü© ‚àà L‚á©+"

text‚ÄπAn ordered loop valued metric space is $T_2$ (i.e. Hausdorff).‚Ä∫

theorem (in metric_space) metric_space_T2:
    assumes "r {down-directs} L‚á©+"
    defines "B ‚â° ‚ãÉc‚ààX. {disk(c,R). R‚ààL‚á©+}" 
    defines "T ‚â° {‚ãÉA. A ‚àà Pow(B)}"
    shows "T {is T‚á©2}"
proof -
  { fix x y assume "x‚àà‚ãÉT" "y‚àà‚ãÉT" "x‚â†y"
    from assms have "B‚äÜT" using pmetric_is_top(2) base_sets_open by auto
    moreover have "‚àÉU‚ààB. ‚àÉV‚ààB. x‚ààU ‚àß y‚ààV ‚àß U‚à©V = 0"
    proof -
      let ?R = "d`‚ü®x,y‚ü©"
      from assms have "‚ãÉT = X" using pmetric_is_top(3) by simp
      with ‚Äπx‚àà‚ãÉT‚Ä∫ ‚Äπy‚àà‚ãÉT‚Ä∫ have "x‚ààX" "y‚ààX" by auto
      with ‚Äπx‚â†y‚Ä∫ have "?R‚ààL‚á©+" using dist_pos by simp
      with assms(2) ‚Äπx‚ààX‚Ä∫ ‚Äπy‚ààX‚Ä∫ have "disk(x,?R) ‚àà B" and "disk(y,?R) ‚àà B"
        by auto
      { assume "disk(x,?R) ‚à© disk(y,?R) = 0"
        moreover from assms(2) ‚Äπx‚ààX‚Ä∫ ‚Äπy‚ààX‚Ä∫ ‚Äπ?R‚ààL‚á©+‚Ä∫ have 
            "disk(x,?R)‚ààB" "disk(y,?R)‚ààB" "x‚ààdisk(x,?R)" "y‚ààdisk(y,?R)"
          using center_in_disk by auto
        ultimately have "‚àÉU‚ààB. ‚àÉV‚ààB. x‚ààU ‚àß y‚ààV ‚àß U‚à©V = 0" by auto
      }
      moreover
      { assume "disk(x,?R) ‚à© disk(y,?R) ‚â† 0"
        then obtain z where "z ‚àà disk(x,?R)" and "z ‚àà disk(y,?R)" 
          by auto
        then have "d`‚ü®x,z‚ü© \<ls> ?R" using disk_definition by simp
        then have "ùü¨ \<ls> \<rm>d`‚ü®x,z‚ü©\<ad>?R" using ls_other_side(1) by simp
        let ?r = "\<rm>d`‚ü®x,z‚ü©\<ad>?R"
        have "?r\<ls>?R"
        proof -
          from ‚Äπz ‚àà disk(y,?R)‚Ä∫ ‚Äπx‚ààX‚Ä∫ ‚Äπy‚ààX‚Ä∫ have "z‚ààX" "x‚â†z"
            using disk_definition pmetric_properties(3) by auto
          with ‚Äπx‚ààX‚Ä∫ ‚Äπy‚ààX‚Ä∫ ‚Äπz‚ààX‚Ä∫ show ?thesis
            using pmetric_loop_valued dist_pos(1) subtract_pos(2) by simp 
        qed
        with ‚Äπx‚ààX‚Ä∫ ‚Äπy‚ààX‚Ä∫ have "disk(x,?r)‚à©disk(y,\<rm>?r\<ad>?R) = 0"
          by (rule disjoint_disks)
        moreover 
        from ‚Äπùü¨\<ls>?r‚Ä∫ ‚Äπ?r\<ls>?R‚Ä∫ have "?r‚ààL‚á©+" "(\<rm>?r\<ad>?R) ‚àà L‚á©+"
          using ls_other_side posset_definition1 by auto
        with assms(2) ‚Äπx‚ààX‚Ä∫ ‚Äπy‚ààX‚Ä∫ have 
            "disk(x,?r)‚ààB" "disk(y,\<rm>?r\<ad>(d`‚ü®x,y‚ü©))‚ààB" and
            "x‚ààdisk(x,?r)" "y‚ààdisk(y,\<rm>?r\<ad>(d`‚ü®x,y‚ü©))"
          using center_in_disk by auto
