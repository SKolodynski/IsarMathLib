section ‹Rings - Direct Product and Chinese Remainder Theorem›

theory Ring_ZF_5 imports Ring_ZF_4 Ring_ZF_3 DirectProduct_ZF

begin

text‹This section develops the direct product of two rings and proves the
  Chinese Remainder Theorem: for comaximal ideals $I$ and $J$ in a commutative
  ring $R$ (meaning $I + J = R$), there is a ring isomorphism
  $R/(I \cap J) \cong R/I \times R/J$.›

subsection‹Direct product of rings›

text‹The direct product of two rings is a ring with componentwise operations.›

lemma direct_product_ring:
  assumes ringR: "IsAring(R,A,M)" and ringS: "IsAring(S,U,V)"
  shows "IsAring(R×S, DirectProduct(A,U,R,S), DirectProduct(M,V,R,S))"
proof -
  let ?DA = "DirectProduct(A,U,R,S)"
  let ?DM = "DirectProduct(M,V,R,S)"
  interpret r1: ring0 R A M "λ x y. A`⟨x,y⟩" "λ x. GroupInv(R,A)`x"
    "λx y. A`⟨x,GroupInv(R,A)`y⟩" "λx y. M`⟨x,y⟩" "TheNeutralElement(R,A)"
    "TheNeutralElement(R,M)" "A`⟨TheNeutralElement(R,M),TheNeutralElement(R,M)⟩"
    "λx. M`⟨x,x⟩" "λs. Fold(A,TheNeutralElement(R,A),s)" "λn x. Fold(A,TheNeutralElement(R,A),{⟨k,x⟩. k∈n})"
    "λs. Fold(M,TheNeutralElement(R,M),s)" "λn x. Fold(M,TheNeutralElement(R,M),{⟨k,x⟩. k∈n})"
   using ringR unfolding ring0_def by auto
  interpret r2: ring0 S U V "λ x y. U`⟨x,y⟩" "λ x. GroupInv(S,U)`x"
    "λx y. U`⟨x,GroupInv(S,U)`y⟩" "λx y. V`⟨x,y⟩" "TheNeutralElement(S,U)"
    "TheNeutralElement(S,V)" "U`⟨TheNeutralElement(S,V),TheNeutralElement(S,V)⟩"
    "λx. V`⟨x,x⟩" "λs. Fold(U,TheNeutralElement(S,U),s)" "λn x. Fold(U,TheNeutralElement(S,U),{⟨k,x⟩. k∈n})"
    "λs. Fold(V,TheNeutralElement(S,V),s)" "λn x. Fold(V,TheNeutralElement(S,V),{⟨k,x⟩. k∈n})"
    using ringS unfolding ring0_def by auto
  have A_fun: "A : R×R→R" using r1.add_group.group_oper_fun by simp
  have U_fun: "U : S×S→S" using r2.add_group.group_oper_fun by simp
  have M_fun: "M : R×R→R" using r1.mult_monoid.monoid_oper_fun by simp
  have V_fun: "V : S×S→S" using r2.mult_monoid.monoid_oper_fun by simp
  interpret dpA: direct0 A U R S "DirectProduct(A,U,R,S)" using A_fun U_fun unfolding direct0_def by auto
  interpret dpM: direct0 M V R S "DirectProduct(M,V,R,S)" using M_fun V_fun unfolding direct0_def by auto
  have DA_fun: "?DA : (R×S)×(R×S)→R×S" using dpA.DirectProduct_ZF_1_L1 .
  have DM_fun: "?DM : (R×S)×(R×S)→R×S" using dpM.DirectProduct_ZF_1_L1 .
  have DA_val: "∀x∈R×S. ∀y∈R×S.
    ?DA`⟨x,y⟩ = ⟨A`⟨fst(x),fst(y)⟩, U`⟨snd(x),snd(y)⟩⟩"
    using dpA.DirectProduct_ZF_1_L2 .
  have DM_val: "∀x∈R×S. ∀y∈R×S.
    ?DM`⟨x,y⟩ = ⟨M`⟨fst(x),fst(y)⟩, V`⟨snd(x),snd(y)⟩⟩"
    using dpM.DirectProduct_ZF_1_L2 .
  have A_assoc: "A {is associative on} R"
    using r1.add_group.groupAssum unfolding IsAgroup_def IsAmonoid_def by blast
  have U_assoc: "U {is associative on} S"
    using r2.add_group.groupAssum unfolding IsAgroup_def IsAmonoid_def by blast
  have M_assoc: "M {is associative on} R"
    using r1.mult_monoid.monoidAssum unfolding IsAmonoid_def by blast
  have V_assoc: "V {is associative on} S"
    using r2.mult_monoid.monoidAssum unfolding IsAmonoid_def by blast
  have DA_assoc: "?DA {is associative on} R×S"
    using dpA.DirectProduct_ZF_2_L2[OF A_assoc U_assoc] by blast
  have DM_assoc: "?DM {is associative on} R×S"
    using dpM.DirectProduct_ZF_2_L2[OF M_assoc V_assoc] by blast
  let ?zRS = "⟨TheNeutralElement(R,A), TheNeutralElement(S,U)⟩"
  let ?oRS = "⟨TheNeutralElement(R,M), TheNeutralElement(S,V)⟩"
  have fstz:"fst(?zRS) = TheNeutralElement(R,A)" using fst_conv by blast
  have sndz:"snd(?zRS) = TheNeutralElement(S,U)" using snd_conv by blast
  have fsto:"fst(?oRS) = TheNeutralElement(R,M)" using fst_conv by blast
  have sndo:"snd(?oRS) = TheNeutralElement(S,V)" using snd_conv by blast
  have zRS_in: "?zRS ∈ R×S"
    using r1.Ring_ZF_1_L2(1) r2.Ring_ZF_1_L2(1) by blast
  have oRS_in: "?oRS ∈ R×S"
    using r1.Ring_ZF_1_L2(2) r2.Ring_ZF_1_L2(2) by blast
  have DA_neut_prop: "∀p∈R×S.
    ?DA`⟨?zRS,p⟩ = p ∧ ?DA`⟨p,?zRS⟩ = p"
  proof
    fix p assume p: "p ∈ R×S"
    have pr: "fst(p) ∈ R" and ps: "snd(p) ∈ S" using p by auto
    have "?DA`⟨?zRS,p⟩ =
      ⟨A`⟨fst(?zRS),fst(p)⟩, U`⟨snd(?zRS),snd(p)⟩⟩"
      using DA_val zRS_in p by blast
    also have "… = ⟨A`⟨TheNeutralElement(R,A),fst(p)⟩,U`⟨TheNeutralElement(S,U),snd(p)⟩⟩"
      using fstz sndz by (simp only:)
    also have "… = ⟨fst(p), snd(p)⟩"
      using r1.Ring_ZF_1_L3(4)[OF pr] r2.Ring_ZF_1_L3(4)[OF ps] by (simp only:)
    also have "… = p" using p by auto
    finally have left: "?DA`⟨?zRS,p⟩ = p" .
    have "?DA`⟨p,?zRS⟩ =
      ⟨A`⟨fst(p),fst(?zRS)⟩, U`⟨snd(p),snd(?zRS)⟩⟩"
      using DA_val zRS_in p by blast
    also have "… = ⟨A`⟨fst(p),TheNeutralElement(R,A)⟩,U`⟨snd(p),TheNeutralElement(S,U)⟩⟩" 
      using sndz fstz by (simp only:)
    also have "… = ⟨fst(p), snd(p)⟩"
      using r1.Ring_ZF_1_L3(3)[OF pr] r2.Ring_ZF_1_L3(3)[OF ps] by (simp only:)
    also have "… = p" using p by auto
    finally have right: "?DA`⟨p,?zRS⟩ = p" .
    from left right show "?DA`⟨?zRS,p⟩ = p ∧ ?DA`⟨p,?zRS⟩ = p" by blast
  qed
  have DM_neut_prop: "∀p∈R×S.
    ?DM`⟨?oRS,p⟩ = p ∧ ?DM`⟨p,?oRS⟩ = p"
  proof
    fix p assume p: "p ∈ R×S"
    have pr: "fst(p) ∈ R" and ps: "snd(p) ∈ S" using p by auto
    have "?DM`⟨p,?oRS⟩ =
      ⟨M`⟨fst(p),fst(?oRS)⟩, V`⟨snd(p),snd(?oRS)⟩⟩"
      using DM_val oRS_in p by blast
    also have "… = ⟨M`⟨fst(p),TheNeutralElement(R,M)⟩,V`⟨snd(p),TheNeutralElement(S,V)⟩⟩"
      by (simp only:fsto sndo)
    also have "… = ⟨fst(p), snd(p)⟩"
      using r1.Ring_ZF_1_L3(5)[OF pr] r2.Ring_ZF_1_L3(5)[OF ps] by (simp only:)
    also have "… = p" using p by auto
    finally have left: "?DM`⟨p,?oRS⟩ = p" .
    have "?DM`⟨?oRS,p⟩ =
      ⟨M`⟨TheNeutralElement(R,M),fst(p)⟩, V`⟨TheNeutralElement(S,V),snd(p)⟩⟩"
      using DM_val p oRS_in by (simp only: fsto sndo)
    also have "… = ⟨fst(p), snd(p)⟩"
      using r1.Ring_ZF_1_L3(6)[OF pr] r2.Ring_ZF_1_L3(6)[OF ps] by (simp only:)
    also have "… = p" using p by auto
    finally have right: "?DM`⟨?oRS,p⟩ = p" .
    from left right show "?DM`⟨?oRS,p⟩ = p ∧ ?DM`⟨p,?oRS⟩ = p" by (simp only:)
  qed
  have DA_monoid: "IsAmonoid(R×S, ?DA)"
    using DA_assoc zRS_in DA_neut_prop unfolding IsAmonoid_def by blast
  have DM_monoid: "IsAmonoid(R×S, ?DM)"
    using DM_assoc oRS_in DM_neut_prop unfolding IsAmonoid_def by blast
  have neut_DA: "TheNeutralElement(R×S,?DA) = ?zRS"
  proof -
    interpret DA_mon: monoid0 "R×S" ?DA "λx y.  DirectProduct(A, U, R, S) `⟨x,y⟩" 
      using DA_monoid unfolding monoid0_def by blast
    show ?thesis apply (rule DA_mon.group0_1_L4[of ?zRS,THEN sym])
      apply (simp only:zRS_in) apply safe
      by (simp only:DA_neut_prop)+
  qed
  have DA_inv: "∀p∈R×S. ∃b∈R×S. ?DA`⟨p,b⟩ = TheNeutralElement(R×S,?DA)"
  proof
    fix p assume p: "p ∈ R×S"
    have pr: "fst(p) ∈ R" and ps: "snd(p) ∈ S" using p by auto
    let ?b = "⟨GroupInv(R,A)`fst(p), GroupInv(S,U)`snd(p)⟩"
    have b_in: "?b ∈ R×S"
      using r1.add_group.inverse_in_group[OF pr]
            r2.add_group.inverse_in_group[OF ps] by blast
    have "?DA`⟨p,?b⟩ =
      ⟨A`⟨fst(p), GroupInv(R,A)`fst(p)⟩, U`⟨snd(p), GroupInv(S,U)`snd(p)⟩⟩"
      using DA_val p b_in by (simp only:fst_conv snd_conv)
    also have "… = ?zRS"
      using r1.add_group.group0_2_L6[OF pr] r2.add_group.group0_2_L6[OF ps] by (simp only:)
    also have "… = TheNeutralElement(R×S,?DA)" using neut_DA by (simp only:)
    finally have "?DA`⟨p,?b⟩ = TheNeutralElement(R×S,?DA)" .
    with b_in show "∃b∈R×S. ?DA`⟨p,b⟩ = TheNeutralElement(R×S,?DA)" by blast
  qed
  have DA_group: "IsAgroup(R×S, ?DA)"
    using definition_of_group DA_monoid DA_inv by blast
  have DA_comm: "?DA {is commutative on} R×S"
  proof -
    from r1.ringAssum have "A {is commutative on} R" unfolding IsAring_def by auto
    from r2.ringAssum have "U {is commutative on} S" unfolding IsAring_def by auto
    with ‹A {is commutative on} R› show ?thesis
      using dpA.DirectProduct_ZF_2_L1 by blast
  qed
  have distrib: "IsDistributive(R×S, ?DA, ?DM)"
  proof -
    from r1.ringAssum have distR: "IsDistributive(R,A,M)" unfolding IsAring_def by auto
    from r2.ringAssum have distS: "IsDistributive(S,U,V)" unfolding IsAring_def by auto
    {
      fix x y z assume xyz: "x∈R×S" "y∈R×S" "z∈R×S"
      have xr: "fst(x)∈R" and xs: "snd(x)∈S" using xyz(1) by auto
      have yr: "fst(y)∈R" and ys: "snd(y)∈S" using xyz(2) by auto
      have zr: "fst(z)∈R" and zs: "snd(z)∈S" using xyz(3) by auto
      have yz:"?DA`⟨y,z⟩∈R×S" using DA_group group0.group_op_closed
        unfolding group0_def by (simp only:xyz(2,3))
      have "M`⟨fst(x),fst(y)⟩∈R" "V`⟨snd(x),snd(y)⟩∈S" "M`⟨fst(x),fst(z)⟩∈R"
        "V`⟨snd(x),snd(z)⟩∈S" using xr xs yr ys zr zs r1.Ring_ZF_1_L4(3) r2.Ring_ZF_1_L4(3)
        by fast+
      then have els:"⟨M`⟨fst(x),fst(y)⟩,V`⟨snd(x),snd(y)⟩⟩∈(R×S)"  "⟨M`⟨fst(x),fst(z)⟩,V`⟨snd(x),snd(z)⟩⟩∈(R×S)"
        by safe
      have "M`⟨fst(y),fst(x)⟩∈R" "V`⟨snd(y),snd(x)⟩∈S" "M`⟨fst(z),fst(x)⟩∈R"
        "V`⟨snd(z),snd(x)⟩∈S" using xr xs yr ys zr zs r1.Ring_ZF_1_L4(3) r2.Ring_ZF_1_L4(3)
        by fast+
      then have els2:"⟨M`⟨fst(y),fst(x)⟩,V`⟨snd(y),snd(x)⟩⟩∈(R×S)"  "⟨M`⟨fst(z),fst(x)⟩,V`⟨snd(z),snd(x)⟩⟩∈(R×S)"
        by safe
      have left: "?DM`⟨x,?DA`⟨y,z⟩⟩ = ?DA`⟨?DM`⟨x,y⟩,?DM`⟨x,z⟩⟩"
      proof -
        have "?DM`⟨x,?DA`⟨y,z⟩⟩ =
          ⟨M`⟨fst(x), fst(?DA`⟨y,z⟩)⟩, V`⟨snd(x), snd(?DA`⟨y,z⟩)⟩⟩"
          using DM_val yz xyz(1) by (simp only:)
        also have "… = ⟨M`⟨fst(x), A`⟨fst(y),fst(z)⟩⟩, V`⟨snd(x), U`⟨snd(y),snd(z)⟩⟩⟩"
          using DA_val xyz(2,3) by (simp only:fst_conv snd_conv)
        also have "… =
          ⟨A`⟨M`⟨fst(x),fst(y)⟩, M`⟨fst(x),fst(z)⟩⟩,
           U`⟨V`⟨snd(x),snd(y)⟩, V`⟨snd(x),snd(z)⟩⟩⟩"
          using distR distS xr yr zr xs ys zs
          unfolding IsDistributive_def by (simp only:)
        also have "… = ?DA`⟨⟨M`⟨fst(x),fst(y)⟩,V`⟨snd(x),snd(y)⟩⟩,⟨M`⟨fst(x),fst(z)⟩,V`⟨snd(x),snd(z)⟩⟩⟩"
          using DA_val els by (simp only:fst_conv snd_conv)
        also have "… = ?DA`⟨?DM`⟨x,y⟩,?DM`⟨x,z⟩⟩"
          using DA_val DM_val xyz by (simp only:fst_conv snd_conv)
        finally show ?thesis .
      qed
      have right: "?DM`⟨?DA`⟨y,z⟩,x⟩ = ?DA`⟨?DM`⟨y,x⟩,?DM`⟨z,x⟩⟩"
      proof -
        have "?DM`⟨?DA`⟨y,z⟩,x⟩ =
          ⟨M`⟨fst(?DA`⟨y,z⟩),fst(x)⟩, V`⟨snd(?DA`⟨y,z⟩),snd(x)⟩⟩"
          using DM_val DA_val xyz(1) yz by (simp only:)
        also have "… = ⟨M`⟨A`⟨fst(y),fst(z)⟩,fst(x)⟩, V`⟨U`⟨snd(y),snd(z)⟩,snd(x)⟩⟩"
          using DA_val xyz(2,3) by (simp only:fst_conv snd_conv)
        also have "… =
          ⟨A`⟨M`⟨fst(y),fst(x)⟩, M`⟨fst(z),fst(x)⟩⟩,
           U`⟨V`⟨snd(y),snd(x)⟩, V`⟨snd(z),snd(x)⟩⟩⟩"
          using distR distS xr yr zr xs ys zs
          unfolding IsDistributive_def by (simp only:)
        also have "… = ?DA`⟨⟨M`⟨fst(y),fst(x)⟩,V`⟨snd(y),snd(x)⟩⟩,⟨M`⟨fst(z),fst(x)⟩,V`⟨snd(z),snd(x)⟩⟩⟩"
          using DA_val els2 by (simp only:fst_conv snd_conv)
        also have "… = ?DA`⟨?DM`⟨y,x⟩,?DM`⟨z,x⟩⟩"
          using DA_val DM_val xyz by (simp only:fst_conv snd_conv)
        finally show ?thesis .
      qed
      from left right have
        "?DM`⟨x,?DA`⟨y,z⟩⟩ = ?DA`⟨?DM`⟨x,y⟩,?DM`⟨x,z⟩⟩ ∧
         ?DM`⟨?DA`⟨y,z⟩,x⟩ = ?DA`⟨?DM`⟨y,x⟩,?DM`⟨z,x⟩⟩" by  (simp only:)
    }
    thus ?thesis unfolding IsDistributive_def by blast
  qed
  from DA_group DA_comm DM_monoid distrib show ?thesis
    unfolding IsAring_def by blast
qed

text‹The additive neutral element of the direct product ring is the pair
  of additive neutral elements.›

lemma direct_product_neutral:
  assumes ringR: "IsAring(R,A,M)" and ringS: "IsAring(S,U,V)"
  shows "TheNeutralElement(R×S, DirectProduct(A,U,R,S)) =
         ⟨TheNeutralElement(R,A), TheNeutralElement(S,U)⟩"
proof -
  let ?DA = "DirectProduct(A,U,R,S)"
  interpret r1: ring0 R A M "λ x y. A`⟨x,y⟩" "λ x. GroupInv(R,A)`x"
    "λx y. A`⟨x,GroupInv(R,A)`y⟩" "λx y. M`⟨x,y⟩" "TheNeutralElement(R,A)"
    "TheNeutralElement(R,M)" "A`⟨TheNeutralElement(R,M),TheNeutralElement(R,M)⟩"
    "λx. M`⟨x,x⟩" "λs. Fold(A,TheNeutralElement(R,A),s)" "λn x. Fold(A,TheNeutralElement(R,A),{⟨k,x⟩. k∈n})"
    "λs. Fold(M,TheNeutralElement(R,M),s)" "λn x. Fold(M,TheNeutralElement(R,M),{⟨k,x⟩. k∈n})"
   using ringR unfolding ring0_def by auto
  interpret r2: ring0 S U V "λ x y. U`⟨x,y⟩" "λ x. GroupInv(S,U)`x"
    "λx y. U`⟨x,GroupInv(S,U)`y⟩" "λx y. V`⟨x,y⟩" "TheNeutralElement(S,U)"
    "TheNeutralElement(S,V)" "U`⟨TheNeutralElement(S,V),TheNeutralElement(S,V)⟩"
    "λx. V`⟨x,x⟩" "λs. Fold(U,TheNeutralElement(S,U),s)" "λn x. Fold(U,TheNeutralElement(S,U),{⟨k,x⟩. k∈n})"
    "λs. Fold(V,TheNeutralElement(S,V),s)" "λn x. Fold(V,TheNeutralElement(S,V),{⟨k,x⟩. k∈n})"
    using ringS unfolding ring0_def by auto
  have A_fun: "A : R×R→R" using r1.add_group.group_oper_fun by simp
  have U_fun: "U : S×S→S" using r2.add_group.group_oper_fun by simp
  interpret dpA: direct0 A U R S "DirectProduct(A,U,R,S)" using A_fun U_fun unfolding direct0_def by auto
  have DA_val: "∀x∈R×S. ∀y∈R×S.
    ?DA`⟨x,y⟩ = ⟨A`⟨fst(x),fst(y)⟩, U`⟨snd(x),snd(y)⟩⟩"
    using dpA.DirectProduct_ZF_1_L2.
  have A_assoc: "A {is associative on} R"
    using r1.add_group.groupAssum unfolding IsAgroup_def IsAmonoid_def by  (simp only:)
  have U_assoc: "U {is associative on} S"
    using r2.add_group.groupAssum unfolding IsAgroup_def IsAmonoid_def by  (simp only:)
  have DA_assoc: "?DA {is associative on} R×S"
    using dpA.DirectProduct_ZF_2_L2[OF A_assoc U_assoc] by  (simp only:)
  let ?zRS = "⟨TheNeutralElement(R,A), TheNeutralElement(S,U)⟩"
  have zRS_in: "?zRS ∈ R×S"
    using r1.Ring_ZF_1_L2(1) r2.Ring_ZF_1_L2(1) by (simp only:)
  have DA_neut_prop: "∀p∈R×S.
    ?DA`⟨?zRS,p⟩ = p ∧ ?DA`⟨p,?zRS⟩ = p"
  proof
    fix p assume p: "p ∈ R×S"
    have pr: "fst(p) ∈ R" and ps: "snd(p) ∈ S" using p by auto
    have "?DA`⟨?zRS,p⟩ =
      ⟨A`⟨fst(?zRS),fst(p)⟩, U`⟨snd(?zRS),snd(p)⟩⟩"

  of multiplicative neutral elements.›

lemma direct_product_neutral_mult:

  shows "TheNeutralElement(R×S, DirectProduct(M,V,R,S)) =

  let ?zRS = "⟨TheNeutralElement(R,M), TheNeutralElement(S,V)⟩"

  have DA_neut_prop: "∀p∈R×S.

      by (simp only:fst_conv snd_conv)
    also have "… = ⟨fst(p), snd(p)⟩"
      using r1.Ring_ZF_1_L3(6)[OF pr] r2.Ring_ZF_1_L3(6)[OF ps] by (simp only:)
    also have "… = p" using p by auto
    finally have left: "?DM`⟨?zRS,p⟩ = p" .

    also have "… = p" using p by auto
    finally have right: "?DM`⟨p,?zRS⟩ = p" .

     TheNeutralElement(S, V)⟩ =

      DirectProduct(M, V, R, S))" by blast
  then show ?thesis by (simp only:sym)

subsection‹Chinese Remainder Theorem›

text‹The Chinese Remainder Theorem: if $I$ and $J$ are comaximal ideals of
  a commutative ring $R$ (i.e.\ $I + J = R$), then $R/(I \cap J) \cong R/I \times R/J$
  as rings. The proof constructs the canonical map $\varphi(r) = ([r]_I, [r]_J)$,
  shows it is a surjective ring homomorphism with kernel $I \cap J$, and applies the
  first isomorphism theorem.›

theorem (in commutative_ring) chinese_remainder:
  assumes idealI: "I◃R" and idealJ: "J◃R"
    and comaximal: "I+⇩IJ = R"
  shows "∃φ. IsRingHomomor(φ,
           QuotientBy(I∩J),
           QuotientGroupOp(R,A,I∩J),
           ProjFun2(R,QuotientGroupRel(R,A,I∩J),M),
           QuotientBy(I)×QuotientBy(J),
           DirectProduct(QuotientGroupOp(R,A,I),QuotientGroupOp(R,A,J),
                         QuotientBy(I),QuotientBy(J)),
           DirectProduct(ProjFun2(R,QuotientGroupRel(R,A,I),M),
                         ProjFun2(R,QuotientGroupRel(R,A,J),M),
                         QuotientBy(I),QuotientBy(J)))
           ∧ φ ∈ bij(QuotientBy(I∩J),
                       QuotientBy(I)×QuotientBy(J))"
proof -
  let ?rI = "QuotientGroupRel(R,A,I)"
  let ?rJ = "QuotientGroupRel(R,A,J)"
  let ?qI = "QuotientGroupOp(R,A,I)"
  let ?qJ = "QuotientGroupOp(R,A,J)"
  let ?mI = "ProjFun2(R,?rI,M)"
  let ?mJ = "ProjFun2(R,?rJ,M)"
  let ?QI = "QuotientBy(I)"
  let ?QJ = "QuotientBy(J)"
  let ?DA = "DirectProduct(?qI,?qJ,?QI,?QJ)"
  let ?DM = "DirectProduct(?mI,?mJ,?QI,?QJ)"
  let ?IJ = "I∩J"
  let ?φ = "λr∈R. ⟨?rI``{r}, ?rJ``{r}⟩"
  have idealIJ: "?IJ◃R" using intersection_ideals[of "{I,J}"] idealI idealJ by simp
  have QI_ring: "IsAring(?QI,?qI,?mI)" using quotientBy_is_ring[OF idealI] by simp
  have QJ_ring: "IsAring(?QJ,?qJ,?mJ)" using quotientBy_is_ring[OF idealJ] by simp
  have QIJ_ring: "IsAring(?QI×?QJ,?DA,?DM)"
    using direct_product_ring[OF QI_ring QJ_ring] by simp
  have φ_fun: "?φ : R→?QI×?QJ"
  proof -
    have eqI: "equiv(R,?rI)" using ideal_equiv_rel[OF idealI] by simp
    have eqJ: "equiv(R,?rJ)" using ideal_equiv_rel[OF idealJ] by simp
    {
      fix r assume r: "r∈R"
      have "?rI``{r} ∈ ?QI" using eqI r quotientI
        unfolding QuotientBy_def[OF idealI] by auto
      moreover have "?rJ``{r} ∈ ?QJ" using eqJ r quotientI
        unfolding QuotientBy_def[OF idealJ] by auto
      ultimately have "⟨?rI``{r},?rJ``{r}⟩ ∈ ?QI×?QJ" by auto
    }
    then show ?thesis using ZF_fun_from_total lam_type by auto
  qed
  have φ_val: "∀r∈R. ?φ`r = ⟨?rI``{r},?rJ``{r}⟩"
    using φ_fun ZF_fun_from_tot_val by auto
  have eqI: "equiv(R,?rI)" using ideal_equiv_rel[OF idealI] by simp
  have eqJ: "equiv(R,?rJ)" using ideal_equiv_rel[OF idealJ] by simp
  have congI_A: "Congruent2(?rI,A)"
    using Group_ZF_2_4_L5A[OF add_group.groupAssum ideal_normal_add_subgroup[OF idealI]]
    by simp
  have congJ_A: "Congruent2(?rJ,A)"
    using Group_ZF_2_4_L5A[OF add_group.groupAssum ideal_normal_add_subgroup[OF idealJ]]
    by simp
  have congI_M: "Congruent2(?rI,M)" using quotientBy_mul_monoid(1)[OF idealI] by simp
  have congJ_M: "Congruent2(?rJ,M)" using quotientBy_mul_monoid(1)[OF idealJ] by simp
  have qI_val: "∀r⇩1∈R. ∀r⇩2∈R.
    ?qI`⟨?rI``{r⇩1},?rI``{r⇩2}⟩ = ?rI``{A`⟨r⇩1,r⇩2⟩}"
    using EquivClass_1_L10[OF eqI congI_A]
    unfolding QuotientGroupOp_def by auto
  have qJ_val: "∀r⇩1∈R. ∀r⇩2∈R.
    ?qJ`⟨?rJ``{r⇩1},?rJ``{r⇩2}⟩ = ?rJ``{A`⟨r⇩1,r⇩2⟩}"
    using EquivClass_1_L10[OF eqJ congJ_A]
    unfolding QuotientGroupOp_def by auto
  have mI_val: "∀r⇩1∈R. ∀r⇩2∈R.
    ?mI`⟨?rI``{r⇩1},?rI``{r⇩2}⟩ = ?rI``{M`⟨r⇩1,r⇩2⟩}"
    using EquivClass_1_L10[OF eqI congI_M] by simp
  have mJ_val: "∀r⇩1∈R. ∀r⇩2∈R.
    ?mJ`⟨?rJ``{r⇩1},?rJ``{r⇩2}⟩ = ?rJ``{M`⟨r⇩1,r⇩2⟩}"
    using EquivClass_1_L10[OF eqJ congJ_M] by simp
  have QI_fun: "?qI : ?QI×?QI→?QI"
    using QI_ring unfolding IsAring_def IsAgroup_def IsAmonoid_def IsAssociative_def by auto
  have QJ_fun: "?qJ : ?QJ×?QJ→?QJ"
    using QJ_ring unfolding IsAring_def IsAgroup_def IsAmonoid_def IsAssociative_def by auto
  have mI_fun: "?mI : ?QI×?QI→?QI"
    using QI_ring unfolding IsAring_def IsAmonoid_def IsAssociative_def by auto
  have mJ_fun: "?mJ : ?QJ×?QJ→?QJ"
    using QJ_ring unfolding IsAring_def IsAmonoid_def IsAssociative_def by auto
  have DA_val: "∀x∈?QI×?QJ. ∀y∈?QI×?QJ.
    ?DA`⟨x,y⟩ = ⟨?qI`⟨fst(x),fst(y)⟩, ?qJ`⟨snd(x),snd(y)⟩⟩"
  proof -
    interpret dpDA: direct0 ?qI ?qJ ?QI ?QJ ?DA
      using QI_fun QJ_fun unfolding direct0_def by auto
    from dpDA.DirectProduct_ZF_1_L2 show ?thesis .
  qed
  have DM_val: "∀x∈?QI×?QJ. ∀y∈?QI×?QJ.
    ?DM`⟨x,y⟩ = ⟨?mI`⟨fst(x),fst(y)⟩, ?mJ`⟨snd(x),snd(y)⟩⟩"
  proof -
    interpret dpDM: direct0 ?mI ?mJ ?QI ?QJ ?DM
      using mI_fun mJ_fun unfolding direct0_def by auto
    from dpDM.DirectProduct_ZF_1_L2 show ?thesis .
  qed
  have φ_add: "IsMorphism(R,A,?DA,?φ)"
    unfolding IsMorphism_def
  proof (intro ballI)
    fix r⇩1 r⇩2 assume r1: "r⇩1∈R" and r2: "r⇩2∈R"
    have sum_in: "A`⟨r⇩1,r⇩2⟩∈R" using Ring_ZF_1_L4(1)[OF r1 r2] by simp
    have φr1_in: "?φ`r⇩1 ∈ ?QI×?QJ" using φ_fun r1 apply_type by (simp only:)
    have φr2_in: "?φ`r⇩2 ∈ ?QI×?QJ" using φ_fun r2 apply_type by (simp only:)
    have els:"?rI``{r⇩1}∈?QI" "?rJ``{r⇩1}∈?QJ" "?rI``{r⇩2}∈?QI" "?rJ``{r⇩2}∈?QJ"
      using eqI r1 r2 quotientI
      unfolding QuotientBy_def[OF idealI] QuotientBy_def[OF idealJ] by auto
    then have elsx:"⟨?rI``{r⇩1},?rJ``{r⇩1}⟩∈?QI×?QJ" "⟨?rI``{r⇩2},?rJ``{r⇩2}⟩∈?QI×?QJ"
      by auto
    have "?φ`(A`⟨r⇩1,r⇩2⟩) = ⟨?rI``{A`⟨r⇩1,r⇩2⟩}, ?rJ``{A`⟨r⇩1,r⇩2⟩}⟩"
      using φ_val sum_in by auto
    also have "… = ⟨?qI`⟨?rI``{r⇩1},?rI``{r⇩2}⟩, ?qJ`⟨?rJ``{r⇩1},?rJ``{r⇩2}⟩⟩"
      using qI_val r1 r2 qJ_val by auto
    also have "… = ?DA`⟨⟨?rI``{r⇩1},?rJ``{r⇩1}⟩,⟨?rI``{r⇩2},?rJ``{r⇩2}⟩⟩"
      using DA_val elsx by (simp only: sym fst_conv snd_conv)
    also have "… = ?DA`⟨?φ`r⇩1,?φ`r⇩2⟩"
      using φ_val r1 r2 by (simp only:)
    finally show "?φ`(A`⟨r⇩1,r⇩2⟩) = ?DA`⟨?φ`r⇩1,?φ`r⇩2⟩" .
  qed
  have φ_mult: "IsMorphism(R,M,?DM,?φ)"
    unfolding IsMorphism_def
  proof (intro ballI)
    fix r⇩1 r⇩2 assume r1: "r⇩1∈R" and r2: "r⇩2∈R"
    have prod_in: "M`⟨r⇩1,r⇩2⟩∈R" using Ring_ZF_1_L4(3)[OF r1 r2] by simp
    have φr1_in: "?φ`r⇩1 ∈ ?QI×?QJ" using φ_fun r1 apply_type by (simp only:)
    have φr2_in: "?φ`r⇩2 ∈ ?QI×?QJ" using φ_fun r2 apply_type by (simp only:)
    have "?φ`(M`⟨r⇩1,r⇩2⟩) = ⟨?rI``{M`⟨r⇩1,r⇩2⟩}, ?rJ``{M`⟨r⇩1,r⇩2⟩}⟩"
      using φ_val prod_in by auto
    also have "… = ⟨?mI`⟨?rI``{r⇩1},?rI``{r⇩2}⟩, ?mJ`⟨?rJ``{r⇩1},?rJ``{r⇩2}⟩⟩"
      using mI_val r1 r2 mJ_val by auto
    also have "… = ?DM`⟨⟨?rI``{r⇩1},?rJ``{r⇩1}⟩,⟨?rI``{r⇩2},?rJ``{r⇩2}⟩⟩"
      using DM_val φr1_in φr2_in φ_val r1 r2 by auto
    also have "… = ?DM`⟨?φ`r⇩1,?φ`r⇩2⟩"
      using φ_val r1 r2 by auto
    finally show "?φ`(M`⟨r⇩1,r⇩2⟩) = ?DM`⟨?φ`r⇩1,?φ`r⇩2⟩" .
  qed
  have φ_one: "?φ`𝟭 = TheNeutralElement(?QI×?QJ,?DM)"
  proof -
    have "?φ`𝟭 = ⟨?rI``{𝟭},?rJ``{𝟭}⟩"
      using φ_val Ring_ZF_1_L2(2) by auto
    also have "… = ⟨TheNeutralElement(?QI,?mI), TheNeutralElement(?QJ,?mJ)⟩"
      using one_quotient[OF idealI] one_quotient[OF idealJ] by auto
    also have "… = TheNeutralElement(?QI×?QJ,?DM)"
      using direct_product_neutral_mult[OF QI_ring QJ_ring] by simp
    finally show ?thesis .
  qed
  have φ_hom: "IsRingHomomor(?φ,R,A,M,?QI×?QJ,?DA,?DM)"
    using φ_fun φ_add φ_mult φ_one unfolding IsRingHomomor_def by simp
  have neut_DA: "TheNeutralElement(?QI×?QJ,?DA) =
      ⟨TheNeutralElement(?QI,?qI), TheNeutralElement(?QJ,?qJ)⟩"
    using direct_product_neutral[OF QI_ring QJ_ring] by simp
  have neut_QI: "TheNeutralElement(?QI,?qI) = ?rI``{𝟬}"
    using neutral_quotient[OF idealI] by simp
  have neut_QJ: "TheNeutralElement(?QJ,?qJ) = ?rJ``{𝟬}"
    using neutral_quotient[OF idealJ] by simp
  have norm_I: "IsAnormalSubgroup(R,A,I)" using ideal_normal_add_subgroup[OF idealI] by simp
  have norm_J: "IsAnormalSubgroup(R,A,J)" using ideal_normal_add_subgroup[OF idealJ] by simp
  have ker_eq: "?φ-``{TheNeutralElement(?QI×?QJ,?DA)} = ?IJ"
  proof
    show "?φ-``{TheNeutralElement(?QI×?QJ,?DA)} ⊆ ?IJ"
    proof
      fix r assume r_in_ker: "r ∈ ?φ-``{TheNeutralElement(?QI×?QJ,?DA)}"
      then have r: "r∈R" and φr: "?φ`r = TheNeutralElement(?QI×?QJ,?DA)"
        using func1_1_L15 by auto
      from φr have vals: "?rI``{r} = ?rI``{𝟬} ∧ ?rJ``{r} = ?rJ``{𝟬}"
        using φ_val r neut_DA neut_QI neut_QJ by auto
      have rI: "r∈I"
        using add_group.Group_ZF_2_4_L5E norm_I r neut_QI vals
        unfolding QuotientBy_def[OF idealI] by auto

    show "?IJ ⊆ ?φ-``{TheNeutralElement(?QI×?QJ,?DA)}"
    proof
      fix r assume r_in: "r ∈ ?IJ"

        using add_group.Group_ZF_2_4_L5E norm_I r neut_QI rI
        unfolding QuotientBy_def[OF idealI] by auto
      moreover have "?rJ``{r} = ?rJ``{𝟬}"
        using add_group.Group_ZF_2_4_L5E norm_J r neut_QJ rJ
        unfolding QuotientBy_def[OF idealJ] by auto
      ultimately have "?φ`r = TheNeutralElement(?QI×?QJ,?DA)"
        using φ_val r neut_DA neut_QI neut_QJ by auto
      with r φ_fun show "r ∈ ?φ-``{TheNeutralElement(?QI×?QJ,?DA)}"
        using func1_1_L15 by auto
    qed
  qed
  have φ_surj: "?φ ∈ surj(R,?QI×?QJ)"
    unfolding surj_def
  proof (safe)
    from φ_fun show "?φ : R→?QI×?QJ" .
