section‹Uniformity defined by a collection of pseudometrics›

theory MetricUniform_ZF imports FinOrd_ZF_1 MetricSpace_ZF UniformityLattice_ZF

begin

text‹In the ‹MetricSpace_ZF› we show how a single (ordered loop valued) pseudometric
  defines a uniformity. In this theory we extend this to the situation where we have
  an arbitrary collection of pseudometrics, all defined on the same set $X$ and valued
  in an ordered loop $L$. Since real numbers form an ordered loop all results proven in
  this theory are true for the standard real-valued pseudometrics. ›

subsection‹Uniformity defined by a collection of pseudometrics›

text‹Suppose $\mathcal{M}$ is a collection of (an ordered loop valued) pseudometrics on $X$,
  i.e. $d:X\times X\rightarrow L^+$ is a pseudometric for every $d\in \mathcal{M}$.
  Then, for each $d\in \mathcal{M}$ the collection of sets 
  $\mathfrak{B} = \{ d^{-1}(\{c\in L^+: c\leq b\}): b \in L_+ \}$
  forms a fundamental system of entourages. Consequently, by taking supersets of 
  $\mathfrak{B}$ we can get a uniformity defined by the pseudometric 
  (see theorem ‹metric_gauge_base› in the ‹MetricSpace_ZF› theory. 
  Repeating this process for each $d\in \mathcal{M}$ we obtain a collection of uniformities
  on $X$. Since all uniformities on $X$ ordered by inclusion form a complete lattice
  (see theorem ‹uniformities_compl_latt› in the ‹UniformityLattice_ZF›) there exists the smallest 
  (coarsest) uniformity that contains all uniformities defined by the 
  pseudometrics in $\mathcal{M}$. ›

text‹To shorten the main definition let's define the uniformity derived from a single
  pseudometric as ‹UnifFromPseudometric›. In the definition below $X$ is the underlying set,
  $L,A,r$ are the carrier, binary operation and the order relation, resp., of the ordered
  loop where the pseudometrics takes its values and $d$ is the pseudometrics.›

definition
  "UnifFromPseudometric(X,L,A,r,d) ≡ Supersets(X×X,UniformGauge(X,L,A,r,d))"

text‹We construct the uniformity from a collection of pseudometrics $\mathcal{M}$ by taking the
  supremumum (i.e. the least upper upper bound) of the collection of
  uniformities defined by each pseudometric from $\mathcal{M}$. ›

definition 
  "UnifFromPseudometrics(X,L,A,r,ℳ) ≡ LUB_Unif(X,{UnifFromPseudometric(X,L,A,r,d). d∈ℳ})"

text‹The context ‹muliple_pmetric› is very similar to the ‹pmetric_space› context
  except that rather than fixing a single pseudometric $d$ we fix a 
  nonempty collection of pseudometrics $\mathcal{M}$. That forces the notation for
  ‹disk›, topology, interior and closure to depend on the pseudometric $d$.
  The $\mathcal{U}$ symbol will denote the collection of uniformities
  generated by $\mathcal{M}$. We also assume that the positive cone $L_+$ 
  of the loop is nonempty, $r$ down-directs $L_+$ (see ‹Order_ZF› for definition)
  and that the (positive cone of the) ordered loop is halfable (see ‹MetricSpace_ZF)›). 
  In short, we assume what we need for ordered loop valued pseudometrics to generate uniformities.›

locale muliple_pmetric = loop1 +
  fixes ℳ and X
  assumes nemptyX: "X≠∅"
  assumes nemptyLp: "L⇩+≠∅"
  assumes downdir: "r {down-directs} L⇩+"
  assumes halfable: "IsHalfable(L,A,r)"
  assumes nemptyM: "ℳ≠∅"
  assumes mpmetricAssm: "∀d∈ℳ. IsApseudoMetric(d,X,L,A,r)"
  fixes unifs ("𝒰")
  defines unifs_def [simp]: "𝒰 ≡ {UnifFromPseudometric(X,L,A,r,d). d∈ℳ}"
  fixes disk
  defines disk_def [simp]: "disk(d,c,R) ≡ Disk(X,d,r,c,R)"
  fixes pmettop ("τ") 
  defines pmettop [simp]: "τ(d) ≡ MetricTopology(X,L,A,r,d)"
  fixes interior ("int")
  defines interior_def [simp]: "int(d,D) ≡ Interior(D,τ(d))"
  fixes cl
  defines cl_def [simp]: "cl(d,D) ≡ Closure(D,τ(d))"

text‹If $d$ is one of the pseudometrics from $\mathcal{M}$ then theorems
  proven in the ‹pmetric_space› context are valid as applied to $d$.›

lemma (in muliple_pmetric) pmetric_space_valid_in_mpm: 
  assumes "d∈ℳ" shows "pmetric_space(L,A,r,d,X)" 
proof
  from assms mpmetricAssm show "IsApseudoMetric(d,X,L,A,r)" by simp
qed 

text‹In the ‹muliple_pmetric› context each element of $\mathcal{U}$ is a uniformity.›

lemma (in muliple_pmetric) each_gen_unif_unif: shows "𝒰 ⊆ Uniformities(X)"
proof
  fix Φ assume "Φ∈𝒰"
  then obtain d where "d∈ℳ" and "Φ = UnifFromPseudometric(X,L,A,r,d)"
    by auto
  with nemptyX nemptyLp downdir halfable show "Φ ∈ Uniformities(X)"
    using pmetric_space_valid_in_mpm pmetric_space.metric_gauge_base(2) unif_in_unifs
    unfolding UnifFromPseudometric_def by simp
qed

text‹The uniformity generated by a family of pseudometrics $\mathcal{M}$ is indeed a uniformity
  which is the supremum of uniformities in $\mathcal{U}$.›

lemma (in muliple_pmetric) gen_unif_unif:
  shows 
    "UnifFromPseudometrics(X,L,A,r,ℳ) {is a uniformity on} X"
    "UnifFromPseudometrics(X,L,A,r,ℳ) = Supremum(OrderOnUniformities(X),𝒰)"
  using nemptyX nemptyM each_gen_unif_unif lub_unif_sup(2,3)
  unfolding UnifFromPseudometrics_def by simp_all

text‹The uniformity generated by a family of pseudometrics contains all uniformities
  generated by the pseudometrics in $\mathcal{M}$.›

lemma (in muliple_pmetric) gen_unif_contains_unifs: assumes "Φ∈𝒰" 
  shows "Φ ⊆ UnifFromPseudometrics(X,L,A,r,ℳ)"
  using nemptyX assms each_gen_unif_unif lub_unif_upper_bound
  unfolding OrderOnUniformities_def InclusionOn_def UnifFromPseudometrics_def
  by simp

text‹If a uniformity contains all uniformities generated by the pseudometrics in $\mathcal{M}$
  then it contains the uniformity generated by that family of pseudometrics.›

lemma (in muliple_pmetric) gen_unif_LUB: 
  assumes "Ψ {is a uniformity on} X" and "∀Φ∈𝒰. Φ⊆Ψ"
  shows "UnifFromPseudometrics(X,L,A,r,ℳ) ⊆ Ψ"
proof -
  from assms have "∀Φ∈𝒰. ⟨Φ,Ψ⟩ ∈ OrderOnUniformities(X)"
    using each_gen_unif_unif unif_in_unifs
    unfolding OrderOnUniformities_def InclusionOn_def
    by auto
  with nemptyX nemptyM show ?thesis
    using each_gen_unif_unif lub_unif_lub
    unfolding OrderOnUniformities_def InclusionOn_def UnifFromPseudometrics_def
    by simp
qed

text‹The uniformity generated by the collection of pseudometrics $\mathcal{M}$
  contains all inverse images of the initial segments of the positive
  cone of $L$, (i.e. sets of the form $d^{-1}(\{c\in L: 0 \leq c \leq y \})$ for 
  $d\in \mathcal{M}$ and $y$ from the positive cone $L_+$ of $L$).›

lemma (in muliple_pmetric) gen_unif_contains_gauges: assumes "d∈ℳ" "y∈L⇩+"
  shows "d-``({c∈L⇧+. ⟨c,y⟩∈r}) ∈ UnifFromPseudometrics(X,L,A,r,ℳ)"
proof -
  let ?g = "d-``({c∈L⇧+. ⟨c,y⟩∈r})"
  from assms(2) have "?g ∈ UniformGauge(X,L,A,r,d)"
    unfolding UniformGauge_def by auto
  with mpmetricAssm assms(1) have "?g ∈ UnifFromPseudometric(X,L,A,r,d)"
    unfolding IsApseudoMetric_def UnifFromPseudometric_def
    using func1_1_L3 superset_gen by auto
  with assms(1) show ?thesis using gen_unif_contains_unifs by auto
qed

text‹The uniformity generated by the collection of pseudometrics $\mathcal{M}$
  is the coarsest uniformity that contains all inverse images of the initial 
  segments of the positive cone of $L$ with respect to all $d\in\mathcal{M}$.›

theorem (in muliple_pmetric) gen_unif_corsest: 
  assumes "Ψ {is a uniformity on} X" and "∀d∈ℳ. ∀y∈L⇩+. d-``({c∈L⇧+. ⟨c,y⟩∈r}) ∈ Ψ"
  shows "UnifFromPseudometrics(X,L,A,r,ℳ) ⊆ Ψ"
proof -
  { fix d assume "d∈ℳ"
    with assms(2) have "UniformGauge(X,L,A,r,d) ⊆ Ψ"
      unfolding UniformGauge_def by auto
    with assms(1) have "UnifFromPseudometric(X,L,A,r,d) ⊆ Ψ"
      unfolding IsUniformity_def UnifFromPseudometric_def 
      using filter_superset_closed by simp
  } hence "∀Φ∈𝒰. Φ⊆Ψ" by simp
  with assms(1) show ?thesis using gen_unif_LUB by simp
qed

subsection‹Uniformity defined by a collection of pseudometrics - an alternative approach›

text‹This section presents an alternative, more direct approach to defining a uniformity generated
  by a collection of pseudometric. This approach is probably equivalent to the one presented
  in the previous section, but more complicated and hence obsolete.›

text‹The next two definitions describe the way a common fundamental system of entourages 
  for $\mathcal{M}$ is constructed. First we take finite subset $M$ of $\mathcal{M}$.
  Then we choose $f:M\rightarrow L_+$. This way for each $d\in M$
  the value $f(d)$ is a positive element of $L$ and 
  $\{d^{-1}(\{c\in L^+: c\leq f(d)\}): d\in M\}$ is a finite collection of 
  subsets of $X\times X$. Then we take intersections of such finite
  collections as $M$ varies over $\mathcal{M}$ and $f$ varies over all possible functions mapping
  $M$ to $L_+$. To simplify notation for this construction we split it into two steps. 
  In the first step we define a collection of finite intersections resulting from 
  choosing a finite set of pseudometrics $M$, $f:M\rightarrow L_+$ and varying 
  the selector function $f$ over the space of functions mapping $M$ to the set of 
  positive elements of $L$. ›

definition
  "UniformGaugeSets(X,L,A,r,M) ≡ 
  {(⋂d∈M. d-``({c∈Nonnegative(L,A,r). ⟨c,f`(d)⟩ ∈ r})). f∈M→PositiveSet(L,A,r)}"
  
text‹In the second step we collect all uniform gauge sets defined above 
  as parameter $M$ vary over all nonempty finite subsets of $\mathcal{M}$. 
  This is the collection of sets that we will show forms a fundamental system of entourages.›

definition
  "UniformGauges(X,L,A,r,ℳ) ≡ ⋃M∈FinPow(ℳ)∖{∅}. UniformGaugeSets(X,L,A,r,M)"

text‹Analogously what is done in the ‹pmetric_space› context 
  we will write ‹UniformGauges(X,L,A,r,ℳ)› as ‹𝔅› in the ‹muliple_pmetric› context.›

abbreviation (in muliple_pmetric) mgauge ("𝔅") where "𝔅 ≡ UniformGauges(X,L,A,r,ℳ)"

text‹The next lemma just shows the definition of $\mathfrak{B}$ in notation 
  used in the ‹muliple_pmetric›. ›

lemma (in muliple_pmetric) mgauge_def_alt: shows 
  "𝔅 = (⋃M∈FinPow(ℳ)∖{∅}. {(⋂d∈M. d-``({c∈L⇧+. c\<lsq>f`(d)})). f∈M→L⇩+})"
  unfolding UniformGaugeSets_def UniformGauges_def by simp

text‹$\mathfrak{B}$ consists of (finite) intersections of sets of the
  form $d^{-1}(\{c\in L^+:c\leq f(d)\})$ where $f:M\rightarrow L_+$
  some finite subset $M\subseteq \mathcal{M}$.
  More precisely, if $M$ is a nonempty finite subset of $\mathcal{M}$ and $f$ maps 
  $M$ to the positive set of the loop $L$, then the set 
  $\bigcap_{d\in M} d^{-1}(\{c\in L^+:c\leq f(d)\}$ is in $\mathfrak{B}$.›

lemma (in muliple_pmetric) mgauge_finset_fun:
  assumes "M∈FinPow(ℳ)" "M≠∅" "f:M→L⇩+"
  shows "(⋂d∈M. d-``({c∈L⇧+. c\<lsq>f`(d)})) ∈ 𝔅"
  using assms mgauge_def_alt by auto

text‹If $d$ is member of any finite subset of $\mathcal{M}$ then 
  $d$ maps $X\times X$ to the nonnegative set of the ordered loop $L$.›

lemma (in muliple_pmetric) each_pmetric_map:
  assumes "M∈FinPow(ℳ)" and "d∈M" shows "d: X×X → L⇧+"
    using assms pmetric_space_valid_in_mpm pmetric_space.pmetric_properties(1)
    unfolding FinPow_def by auto

text‹Members of the uniform gauge defined by multiple pseudometrics 
  are subsets of $X\times X$ i.e. relations on $X$. ›

lemma (in muliple_pmetric) muniform_gauge_relations:
  assumes "B∈𝔅" shows "B⊆X×X"
proof -
  from assms obtain M f where "M∈FinPow(ℳ)" and
    I: "B = (⋂d∈M. d-``({c∈L⇧+. c\<lsq>f`(d)}))"
    using mgauge_def_alt by auto
  { fix d assume "d∈M"
    with ‹M∈FinPow(ℳ)› have "d: X×X → L⇧+"
      using each_pmetric_map by simp
    then have "d-``({c∈L⇧+. c\<lsq>f`(d)}) ⊆ X×X" using func1_1_L15 by auto
  } with I show ?thesis using inter_subsets_subset by simp
qed

text‹Suppose $M_1$ is a subset of $M$ and $\mathcal{M}$.
  $f_1,f$ map $M_1$ and $M$, resp. to $L_+$ and $f\leq f_1$ on $M_1$.
  Then the set $\bigcap_{d\in M} d^{-1}(\{y \in L_+: y\leq f(d)\})$
  is included in the set $\bigcap_{d\in M_1} d^{-1}(\{y \in L_+: y\leq f_1(d)\})$. ›

lemma (in muliple_pmetric) fun_inter_vimage_mono:
  assumes "M⇩1⊆ℳ" "M⇩1⊆M" "M⇩1≠∅" "f⇩1:M⇩1→L⇩+" "f:M→L⇩+" and 
    "∀d∈M⇩1. f`(d)\<lsq>f⇩1`(d)"
  shows
     "(⋂d∈M. d-``({c∈L⇧+. c\<lsq>f`(d)})) ⊆ (⋂d∈M⇩1. d-``({c∈L⇧+. c\<lsq>f⇩1`(d)}))"
proof -
  let ?L = "(⋂d∈M. d-``({c∈L⇧+. c\<lsq>f`(d)}))"
  let ?R = "(⋂d∈M⇩1. d-``({c∈L⇧+. c\<lsq>f⇩1`(d)}))"
  from assms(2,3) have I: "?L ⊆ (⋂d∈M⇩1. d-``({c∈L⇧+. c\<lsq>f`(d)}))"
    using inter_index_mono by simp
  from assms(1,6) have 
    "∀d∈M⇩1. (d-``({c∈L⇧+. c\<lsq>f`(d)})) ⊆ d-``({c∈L⇧+. c\<lsq>f⇩1`(d)})" 
    using pmetric_space_valid_in_mpm pmetric_space.uniform_gauge_mono
      by force
  with assms(2) have  "(⋂d∈M⇩1. d-``({c∈L⇧+. c\<lsq>f`(d)})) ⊆ ?R" by force
  with I show "?L⊆?R" by (rule subset_trans)
qed

text‹For any two sets $B_1,B_2$ in $\mathfrak{B}$ there exist a third one
  that is contained in both. ›

lemma (in muliple_pmetric) mgauge_1st_cond:
  assumes "B⇩1∈𝔅" "B⇩2∈𝔅"
  shows "∃B∈𝔅. B⊆B⇩1∩B⇩2" 
proof -
  from assms(1,2) obtain M⇩1 f⇩1 M⇩2 f⇩2 where "M⇩1≠∅" "M⇩2≠∅" and
  I: "M⇩1∈FinPow(ℳ)" "M⇩2∈FinPow(ℳ)" "f⇩1:M⇩1→L⇩+" "f⇩2:M⇩2→L⇩+" and  
  II: "B⇩1 = (⋂d∈M⇩1. d-``({c∈L⇧+. c\<lsq>f⇩1`(d)}))" "B⇩2 = (⋂d∈M⇩2. d-``({c∈L⇧+. c\<lsq>f⇩2`(d)}))"
    using mgauge_def_alt by auto
  let ?M⇩3 = "M⇩1∪M⇩2"  
  from downdir have "IsDownDirectedSet(L⇩+,r)" 
    using down_directs_directed by simp
  with I have 
    "∃f⇩3∈?M⇩3→L⇩+. (∀d∈M⇩1. ⟨f⇩3`(d),f⇩1`(d)⟩∈r) ∧ (∀d∈M⇩2. ⟨f⇩3`(d),f⇩2`(d)⟩∈r)"
    using two_fun_low_bound by simp
  then obtain f⇩3 where "f⇩3:?M⇩3→L⇩+" and
    III: "∀d∈M⇩1. f⇩3`(d)\<lsq>f⇩1`(d)" "∀d∈M⇩2. f⇩3`(d)\<lsq>f⇩2`(d)"
      by auto
  let ?B⇩3 = "⋂d∈?M⇩3. d-``({c∈L⇧+. c\<lsq>f⇩3`(d)})"
  from I(1,2) ‹M⇩1≠∅› ‹f⇩3:?M⇩3→L⇩+› have "?B⇩3∈𝔅"
    using union_finpow mgauge_def_alt by auto
  moreover have "?B⇩3⊆B⇩1∩B⇩2"
  proof - 
    from I(1,2) have "M⇩1⊆ℳ" "M⇩1⊆?M⇩3" "M⇩2⊆ℳ" "M⇩2⊆?M⇩3"
      unfolding FinPow_def by auto
    with ‹M⇩1≠∅› ‹M⇩2≠∅› I(3,4) II ‹f⇩3:?M⇩3→L⇩+› III  
    have "?B⇩3⊆B⇩1" and "?B⇩3⊆B⇩2" using fun_inter_vimage_mono by simp_all
    thus "?B⇩3⊆B⇩1∩B⇩2" by auto
  qed
  ultimately show "∃B∈𝔅. B⊆B⇩1∩B⇩2" by (rule witness_exists)
qed

text‹Sets in $\mathfrak{B}$ contain the diagonal and are symmetric,
  hence contain a symmetric subset from $\mathfrak{B}$.›

lemma (in muliple_pmetric) mgauge_2nd_and_3rd_cond: assumes "B∈𝔅" 
  shows "id(X)⊆B" "B = converse(B)" "∃B⇩2∈𝔅. B⇩2 ⊆ converse(B)" 
proof -
  from assms obtain M f where "M∈FinPow(ℳ)" "M≠∅" "f:M→L⇩+" and
    I: "B = (⋂d∈M. d-``({c∈L⇧+. c\<lsq>f`(d)}))"
    using mgauge_def_alt by auto
  { fix d assume "d∈M"
    let ?B⇩d = "d-``({c∈L⇧+. c\<lsq>f`(d)})"
    from ‹d∈M› ‹f:M→L⇩+› ‹M∈FinPow(ℳ)› have
      "pmetric_space(L,A,r,d,X)" and "?B⇩d ∈ UniformGauge(X,L,A,r,d)"
      unfolding FinPow_def UniformGauge_def 
      using apply_funtype pmetric_space_valid_in_mpm by auto
    then have "id(X)⊆?B⇩d" and "?B⇩d = converse(?B⇩d)" 
      using pmetric_space.gauge_2nd_cond pmetric_space.gauge_symmetric 
      by simp_all
  } with I ‹M≠∅› show "id(X)⊆B" and "B = converse(B)" by auto
  from assms ‹B = converse(B)› show "∃B⇩2∈𝔅. B⇩2 ⊆ converse(B)"
    by auto
qed

text‹$\mathfrak{B}$ is a subset of the power set of $X\times X$.›

lemma (in muliple_pmetric) mgauge_5thCond: shows "𝔅⊆Pow(X×X)"
  using muniform_gauge_relations by auto

text‹If $\mathcal{M}$ and $L_+$ are nonempty then $\mathfrak{B}$ is also nonempty.›

lemma (in muliple_pmetric) mgauge_6thCond: shows "𝔅≠∅" 
proof -
  from nemptyLp nemptyM obtain M y where "M∈FinPow(ℳ)" "M≠∅" and "y∈L⇩+"
    using finpow_nempty_nempty by blast
  from ‹y∈L⇩+› have "ConstantFunction(M,y):M→L⇩+" using func1_3_L1 by simp
  with ‹M∈FinPow(ℳ)› ‹M≠∅› show "𝔅≠∅" using mgauge_finset_fun by auto
qed

text‹If the loop order is halfable then for every set $B_1\in \mathfrak{B}$
  there is another set $B_2\in \mathfrak{B}$ such that $B_2\circ B_2 \subseteq B_1$.›

lemma (in muliple_pmetric) mgauge_4thCond: 
  assumes "B⇩1∈𝔅" shows "∃B⇩2∈𝔅. B⇩2 O B⇩2 ⊆ B⇩1"
proof -
  from assms(1) obtain M f⇩1 where "M∈FinPow(ℳ)" "M≠∅" "f⇩1:M→L⇩+" and
    I: "B⇩1 = (⋂d∈M. d-``({c∈L⇧+. c\<lsq>f⇩1`(d)}))"
    using mgauge_def_alt by auto
  from halfable ‹f⇩1:M→L⇩+› have "∀d∈M. ∃b⇩2∈L⇩+. b⇩2\<ra>b⇩2 \<lsq> f⇩1`(d)"
    using apply_funtype unfolding IsHalfable_def by simp
  with ‹M∈FinPow(ℳ)› have "∃f⇩2∈M→L⇩+. ∀d∈M. f⇩2`(d) \<ra> f⇩2`(d) \<lsq> f⇩1`(d)"
    unfolding FinPow_def using finite_choice_fun by auto
  then obtain f⇩2 where "f⇩2∈M→L⇩+" and II: "∀d∈M. f⇩2`(d) \<ra> f⇩2`(d) \<lsq> f⇩1`(d)"
    by auto
  let ?B⇩2 = "⋂d∈M. d-``({c∈L⇧+. c\<lsq>f⇩2`(d)})" 
  { fix d assume "d∈M"
    let ?A⇩2 = "d-``({c∈L⇧+. c\<lsq>f⇩2`(d)})"
    from ‹d∈M› ‹M∈FinPow(ℳ)› have "pmetric_space(L,A,r,d,X)"

subsection‹Halving sequences of entourages›

text‹The notion of uniformity can be defined in several ways. Our primary definition
  in the ‹UniformSpace_ZF› theory is based on the concept of entourages. 
  In ‹UniformSpace_ZF_1› we consider an alternative definition based on uniform covers.
  The third possible definition is based on families of pseudometrics.
 "The significance of defining a uniformity by means of a family 
  of pseudometrics lies in the fact that all uniformities can be so obtained." 
  (from Bourbaki: General Topology Chapter IX par. 1.4).
  In this section we formalize a part of the material that is needed to prove 
  this statement that can be done without the Axiom of Choice and in the general setting 
  of ordered-loop valued pseudometrics.›

text‹The definition of a uniformity requires that for each entourage $U$ 
  there is another one $V$ such that $V\circ V\subseteq U$. Furthermore, 
  lemma ‹half_size_symm› shows that we can assume $V$ is symmetric.
  In some propositions in this section we will assume that we are given a function say $h$ 
  that provides those half-size entourages, i.e. maps a uniformity $\Phi$ into itself and 
  for each entourage $U\in\Phi$ we have $h(U)\circ h(U)\subseteq V$. 

  The existence of such function follows from the Axiom of Choice, but we don't want to use AC
  in this theory.

  The next definition lists the desired properties of such halving function for shorter
  references in theorem assumptions.›

definition
  IsHalvingFunction ("_ {is a halving function for} _") where
    "h {is a halving function for} Φ ≡ 
      h:Φ→Φ ∧ (∀U∈Φ. h`(U) = converse(h`(U)) ∧ h`(U) O h`(U) ⊆ U)"

text‹Let $\Phi$ be a uniformity on $X$, $U\in\Phi$ and $h$ is a halving function for
  $\Phi$. This function then defines inductively a sequence $\{ H_n\}_{n\in\mathbb{N}}$ of entourages 
  such that $H_0 = U$ and $H_{n+1}\circ H_{n+1}\subseteq H_n$. The next lemma shows
  that $H$ is a $\Phi$-valued sequence which starts at $U$. ›

lemma halving_seq_start: 
  assumes "U∈Φ" "h {is a halving function for} Φ"
  defines "H ≡ InductiveSequence(U,h)"
  shows "H:nat→Φ" and "H`(0) = U"
  using assms indseq_seq indseq_valat0 unfolding IsHalvingFunction_def by auto

text‹For the inductively defined sequence of halving entourages $H$ starting from $U$
   and a natural number $n$ we indeed have $H_{n+1} \circ H_{n+1} \subseteq H_n$. ›

lemma halving_seq_halves: assumes "U∈Φ" "h {is a halving function for} Φ" "n∈nat"
  defines "H ≡ InductiveSequence(U,h)"
  shows "H`(n #+ 1) = converse(H`(n #+ 1))" and "H`(n #+ 1) O H`(n #+ 1) ⊆ H`(n)"
proof -
  from assms have "H`(n #+ 1) =  h`(H`(n))"
    unfolding IsHalvingFunction_def using indseq_vals succ_add_one(1) by auto
  from assms have "H`(n)∈Φ" using halving_seq_start(1) apply_funtype by force
  with assms(2) ‹H`(n #+ 1) =  h`(H`(n))› show
    "H`(n #+ 1) = converse(H`(n #+ 1))" and "H`(n #+ 1) O H`(n #+ 1) ⊆ H`(n)"
    unfolding IsHalvingFunction_def by simp_all
qed

text‹A halving sequence in $\Phi$ is decreasing in the inclusion order on $\Phi$,
  hence the inclusion order on $\Phi$ is total on the sequence's image 
  of the natural numbers, hence total on the sequence's image of positive
  natural numbers. We need that fact about positive natural numbers
  because we plan to prove that the image of the sequence on the positive
  natural numbers forms a fundamental system of symmetric entourages.
  and the first element of the sequence (at index $0$) does not have to be symmetric.›

lemma halving_seq_decr: 
  assumes "Φ {is a uniformity on} X" "U∈Φ" "h {is a halving function for} Φ"
  defines "H ≡ InductiveSequence(U,h)"
  shows 
    "IsDecreasingSeq(Φ,InclusionOn(Φ),H)"
    "InclusionOn(Φ) {is total on} (H``(nat))"
    "InclusionOn(Φ) {is total on} (H``(nat∖{0}))"
proof -
  from assms(2,3,4) have "H:nat→Φ" using halving_seq_start(1) 
    by simp
  let ?r = "InclusionOn(Φ)"
  { fix n assume "n∈nat"
    with ‹H:nat→Φ› have "H`(n) ∈ Φ" and "H`(n #+ 1) ∈ Φ" 
      using apply_funtype by simp_all
    with assms(1) have "H`(n #+ 1) ⊆ X×X" and "id(X) ⊆ H`(n #+ 1)"
      using uni_domain(1) unfolding IsUniformity_def by simp_all
    then have "H`(n #+ 1) ⊆ H`(n #+ 1) O H`(n #+ 1)"
      using refl_square_greater by simp
    with assms(2,3,4) ‹n∈nat› ‹H`(n #+ 1) ∈ Φ› ‹H`(n) ∈ Φ›
    have "⟨H`(n #+ 1),H`(n)⟩ ∈ ?r"
      using halving_seq_halves(2) unfolding InclusionOn_def 
      by force
  } hence "∀n∈nat. ⟨H`(n #+ 1),H`(n)⟩ ∈ ?r" by simp
  with ‹H:nat→Φ› show "IsDecreasingSeq(Φ,InclusionOn(Φ),H)" 
    unfolding IsDecreasingSeq_def by simp
  then show "?r {is total on} H``(nat)"
    using incl_is_partorder decr_seq_total 
    unfolding IsPartOrder_def IsPreorder_def  by blast
  then show "InclusionOn(Φ) {is total on} (H``(nat∖{0}))"
    unfolding IsTotal_def by auto
qed

text‹We aim at showing that a halving sequence image is a fundamental system of entourages.
  The next lemma shows the first condition in the uniform base definition:
  for two sets in a halving sequence image there is a third one that is contained in both. ›

lemma halving_seq_base_1st_cond: 
  assumes "Φ {is a uniformity on} X" "U∈Φ" "h {is a halving function for} Φ"
  defines "H ≡ InductiveSequence(U,h)" 
  assumes "B⇩1∈H``(nat∖{0})" "B⇩2∈H``(nat∖{0})"
  shows "∃B∈H``(nat∖{0}). B⊆B⇩1∩B⇩2"
proof -
  let ?r = "InclusionOn(Φ)"
  from assms have "⟨B⇩1,B⇩2⟩ ∈ ?r ∨ ⟨B⇩2,B⇩1⟩ ∈ ?r"
    using halving_seq_decr(3) unfolding IsTotal_def by simp
  with assms(5,6) show ?thesis unfolding InclusionOn_def 
    by auto
qed

  are symmetric and for every such set $B$ there is another one contained in the converse of the
  first one and another one $B_2$ such that $B_2\circ B_2\subseteq B_1$.

  system of entourages, but it's good to have as we plan to construct a pseudometric
  that generates that halving sequence image of positive naturals.›

text‹ The next goal we want to achieve is to show that if a uniformity
  has an enumerable base consisting of symmetric entourages $\{ V_n\}_{n\in\mathbb{N}}$
  then it can be represented by a pseudometric. In the Bourbaki's proof
  the first step is to inductively define a sequence $\{ V_n\}_{n\in\mathbb{N}}$ 
  of entourages with the property that $U_1 \subseteq V_1$ and for all natural
  $n\geq 1$ we have $U_{n+1}\circ U_{n+1}\circ U_{n+1} \subseteq U_n\cap V_n$.
  The existence of such sequence for every uniformity with countable base
  follows from theorem ‹ustex3sym› in ‹UniformSpace_ZF› theory and the Axiom of Choice.
  Here we construct this sequence from an analogue of the halving function but 
  with the property the $t(U)\circ t(u)\circ t(U)\subseteq U$ using the notion of 
  inductive sequence with changing generation function.
  The ‹IsDiv3Function(Φ,t)› predicate defines an analogue of the halving function but 
  with the property the $t(U)\circ t(u)\circ t(U)\subseteq U$ for all $U\in\Phi$. ›

definition "IsDiv3Function(Φ,t) ≡ t:Φ→Φ ∧ 
  (∀U∈Φ. t`(U) = converse(t`(U)) ∧ t`(U) O (t`(U) O t`(U)) ⊆ U)" 

text‹Given a function $t$ that maps a uniformity $\Phi$ into itself 
  and a sequence of entourages $\mathcal{V}:mathbb{N}\rightarrow \Phi$
  we define a sequence of functions called ‹SeqInter› such that for every $n\in\mathbb{N}$
  the ‹SeqInter(n)› maps $\Phi$ into itself and ‹SeqInter(n)›$(U) = t(U \cap \mathcal{V}_n$. 
  In actual theorems we will assume that the function $t$ satisfies the 
  ‹IsDiv3Function› predicate and the ‹SeqInter› will be a generating sequence of functions
  in an inductive definition.›

definition 
  "SeqInter(Φ,𝒱,t) ≡ {⟨n,{⟨U,t`(U ∩ 𝒱`(n))⟩. U∈Φ}⟩. n∈nat}"

text‹Suppose $\Phi$ is a uniformity on $X$, $t$ maps $\Phi$ into itself and
  $\mathcal{V}:\mathbb{N}\rightarrow\Phi$ is a sequence of entourages of $\Phi$.
  Then ‹SeqInter(Φ,𝒱,t)› is a sequence of functions mapping $\Phi$ into itself.›

definition "Div3Seq(X,Φ,𝒱,t) ≡ InductiveSeqVarF(X×X,Φ,SeqInter(Φ,𝒱,t))"

text‹A couple of other properties of the thirding sequence in addition to what is shown
  in ‹seq_div_cube›: 
  Under the same assumptions and additionaly that $n$ is a natural number 
  and denoting the thirding sequence as $T$ the composition cube of the $(n+1)$'th element of the 
  thirding sequence and the $(n+1)$'th element itself are contained in the $n$'th element, 
  and also the the $(n+1)$'th element is contained in $\mathcal{V}_n$. ›

lemma div3_seq_props:
  assumes "Φ {is a uniformity on} X" "IsDiv3Function(Φ,t)" "𝒱:nat→Φ" "n∈nat"
  defines "T ≡ Div3Seq(X,Φ,𝒱,t)"
  shows "T`(n #+ 1) ⊆ T`(n)" "T`(n #+ 1) ⊆ 𝒱`(n)" and 
    "T`(n #+ 1) O T`(n #+ 1) O T`(n #+ 1) ⊆ T`(n)"
proof -
  from assms have 
    I: "T`(n #+ 1) O T`(n #+ 1) O T`(n #+ 1) ⊆ T`(n)∩𝒱`(n)"
    using seq_div_cube(3) unfolding Div3Seq_def by simp
  thus II: "T`(n #+ 1) O T`(n #+ 1) O T`(n #+ 1) ⊆ T`(n)" by auto
  from assms have "T:nat→Φ" using seq_div_cube(1)
    unfolding Div3Seq_def by simp
  with assms(4) have "T`(n #+ 1) ∈ Φ" using apply_funtype 
    by simp
  with assms(1) have "T`(n #+ 1) ⊆ X×X" and "id(X) ⊆ T`(n #+ 1)"
    using uni_domain(1) unfolding IsUniformity_def by simp_all
  then have "T`(n #+ 1) ⊆ T`(n #+ 1) O T`(n #+ 1) O T`(n #+ 1)"
    using refl_cube_greater by simp
  with I II show "T`(n #+ 1) ⊆ T`(n)" and "T`(n #+ 1) ⊆ 𝒱`(n)"
    by auto
qed

text‹If $\Phi$ is a uniformity on $X$, $t$ maps $\Phi$ into itself so that
  $t(U)$ is symmetric and $t(U)\circ t(U)\circ t(U)\subseteq U$ for all $U\in\Phi$, 
  and $\mathcal(V):\mathbb{N}\rightarrow\Phi$ is a sequence of entourages whose image is
  a fundamental system of entourages of $\Phi$, then the image of the thirding sequence 
  $T:\mathbb{N}\rightarrow \Phi$ defined by $t$ and $\mathcal{V}$ is a fundamental 
  system of entourages (a uniform base) of $\Phi$.›

theorem div3_seq_base_of: 
  assumes "Φ {is a uniformity on} X" "IsDiv3Function(Φ,t)" and 
    "𝒱:nat→Φ" "(𝒱``(nat)) {is a uniform base of} Φ"
  defines "T ≡ Div3Seq(X,Φ,𝒱,t)"
  shows "(T``(nat)) {is a uniform base of} Φ"
proof -
  let ?𝔅 = "T``(nat)"
  from assms(1,2,3,5) have "T:nat→Φ"
    unfolding Div3Seq_def using seq_div_cube(1) by simp
  then have "?𝔅 ⊆ Φ" using func1_1_L6 by simp
  { fix U assume "U∈Φ"
    with assms(4) obtain B where "B∈𝒱``(nat)" and "B⊆U"
      unfolding IsUniformityBase_def by blast
    from assms(3) ‹B∈𝒱``(nat)› obtain n where "n∈nat" and "B=𝒱`(n)"
      using func_imagedef by auto
    with assms ‹B⊆U› ‹T:nat→Φ› have "∃V∈?𝔅. V⊆U"
      using div3_seq_props(2) func_imagedef by force
  } hence "∀U∈Φ. ∃V∈?𝔅. V⊆U" by simp
  with ‹?𝔅 ⊆ Φ› show "(T``(nat)) {is a uniform base of} Φ"
    unfolding IsUniformityBase_def by simp
qed

text‹The rest of this section aims at showing that With the assumptions of ‹seq_div_cube› 
  (i.e. when $\mathcal{V}$ is any sequence of entourages of $\Phi$) 
  the image of the thirding sequence is a fundamental system of entourages. 
  This is a mildly interesting fact proven by a mistake that is unlikely to be ever used,
  but will stay here as a curiosity.
  We start with the claim that the thirding sequence is decreasing in the inclusion order 
  on $\Phi$ hence the inclusion order on $\Phi$ is total on 
  the sequence's image of the natural numbers.›

lemma div3_seq_decr: 
  assumes "Φ {is a uniformity on} X" "IsDiv3Function(Φ,t)" "𝒱:nat→Φ"
  shows 
    "IsDecreasingSeq(Φ,InclusionOn(Φ),Div3Seq(X,Φ,𝒱,t))"
    "InclusionOn(Φ) {is total on} (Div3Seq(X,Φ,𝒱,t)``(nat))"
proof -
  let ?T = "Div3Seq(X,Φ,𝒱,t)"
  let ?r = "InclusionOn(Φ)"
  from assms(1,2,3) have "?T:nat→Φ" using seq_div_cube(1)
    unfolding Div3Seq_def by simp
  with assms(1,2,3) show "IsDecreasingSeq(Φ,InclusionOn(Φ),?T)"
    using apply_funtype div3_seq_props(1) 
    unfolding InclusionOn_def IsDecreasingSeq_def by simp
  then show "?r {is total on} ?T``(nat)"
    using incl_is_partorder decr_seq_total 
    unfolding IsPartOrder_def IsPreorder_def by blast
qed

text‹The next lemma shows the first condition we need:
  for two sets in a thirding sequence image there is a third one that is contained in both.›

lemma div3_seq_base_1st_cond: 
  assumes "Φ {is a uniformity on} X" "IsDiv3Function(Φ,t)" "𝒱:nat→Φ"
  defines "T ≡ Div3Seq(X,Φ,𝒱,t)"
  assumes "B⇩1∈T``(nat)" "B⇩2∈T``(nat)"
  shows "∃B∈T``(nat). B⊆B⇩1∩B⇩2"
proof -
  let ?r = "InclusionOn(Φ)"
  from assms have "⟨B⇩1,B⇩2⟩ ∈ ?r ∨ ⟨B⇩2,B⇩1⟩ ∈ ?r"
    using div3_seq_decr(2) unfolding IsTotal_def by simp
  with assms(5,6) show ?thesis unfolding InclusionOn_def 
    by auto
qed

text‹Sets in the thirding sequence image of natural numbers contain the diagonal, 
  are symmetric and for every such set $B$ there is another one contained in the converse of the
  first one and another one $B_2$ such that $B_2\circ B_2\subseteq B_1$.›

lemma div3_seq_base_2_3_4_conds: 
  assumes "Φ {is a uniformity on} X" "IsDiv3Function(Φ,t)" "𝒱:nat→Φ"
  defines "T ≡ Div3Seq(X,Φ,𝒱,t)"
  assumes "B∈T``(nat)"
  shows 
    "id(X)⊆B" "B = converse(B)" 
    "∃B⇩1∈T``(nat). B⇩1 ⊆ converse(B)" 
    "∃B⇩2∈T``(nat). B⇩2 O B⇩2 ⊆ B"
proof -
  from assms have "T:nat→Φ"
    unfolding Div3Seq_def using seq_div_cube(1) by simp
  with assms(1,5) show "id(X)⊆B" using entourage_props(2) func1_1_L6(2)
    by blast
  from assms(5) ‹T:nat→Φ› obtain n where "n∈nat" and "B=T`(n)"
    using func_imagedef by auto
  with assms(1,2,3,4) show "B = converse(B)"
    using seq_div_cube(3) unfolding Div3Seq_def by simp
  with assms(5) show "∃B⇩1∈T``(nat). B⇩1 ⊆ converse(B)" by auto
  from assms ‹n∈nat› ‹T:nat→Φ› have 
    I: "T`(n #+ 1) O T`(n #+ 1) O T`(n #+ 1) ⊆ T`(n)"
    and "T`(n #+ 1)∈T``(nat)" "T`(n #+ 1) ∈ Φ"
    using div3_seq_props(3) func_imagedef apply_funtype 
    unfolding Div3Seq_def by auto
  from assms(1) I ‹T`(n #+ 1) ∈ Φ› have 
    "T`(n #+ 1) O T`(n #+ 1) ⊆ T`(n)"
    using entourage_props(1,2) refl_cube_greater_square by blast
  with ‹T`(n #+ 1)∈T``(nat)› ‹B=T`(n)› show "∃B⇩2∈T``(nat). B⇩2 O B⇩2 ⊆ B"
    by auto
qed

text‹The thirding sequence image of natural numbers is contained 
  in the powerset of $X\times X$ and is not empty.›

lemma div3_seq_base_5_and_6th_cond: 
  assumes "Φ {is a uniformity on} X" "IsDiv3Function(Φ,t)" "𝒱:nat→Φ"
  defines "T ≡ Div3Seq(X,Φ,𝒱,t)"
  shows "T``(nat) ⊆ Pow(X×X)" and "T``(nat) ≠ ∅"
proof -
  from assms have "T:nat→Φ"
    unfolding Div3Seq_def using seq_div_cube(1) by simp
  then have "T``(nat) ⊆ Φ" using func1_1_L6(2) by simp
  with assms(1) show "T``(nat) ⊆ Pow(X×X)" 
    using unif_filter unfolding IsFilter_def by blast
  from ‹T:nat→Φ› show "T``(nat) ≠ ∅" using func_imagedef
    by auto
qed

text‹If $\Phi$ is a uniformity on $X$, $t$ maps $\Phi$ into itself so that
  $t(U)$ is symmetric and $t(U)\circ t(U)\circ T(U)\subseteq U$ for all $U\in\Phi$, 
  and $\mathcal(V):\mathbb{N}\rightarrow\Phi$ is any sequence of entourages from $\Phi$
  then the image of the sequence defined inductively by 
  $T_0 = X\times X,\  T_{n+1} = t(T_n\cap\mathcal{V}_n)$ is a fundamental system of entourages.›

theorem div3_seq_base:
  assumes "Φ {is a uniformity on} X" "IsDiv3Function(Φ,t)" "𝒱:nat→Φ"
  defines "T ≡ Div3Seq(X,Φ,𝒱,t)"
  shows "(T``(nat)) {is a uniform base on} X"
  using assms div3_seq_base_1st_cond div3_seq_base_2_3_4_conds
    div3_seq_base_5_and_6th_cond
  unfolding IsUniformityBaseOn_def by simp
end
