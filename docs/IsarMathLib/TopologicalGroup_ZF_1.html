theorem(in topgroup) cl_point:
  assumes "x‚ààG"
  shows "cl({x}) = (‚ãÇH‚ààùí©‚á©0. x\<ltr>H)"
proof-
  {
    have c:"cl({x}) = (‚ãÇH‚ààùí©‚á©0. {x}\<sad>H)" using cl_topgroup assms by auto
    {
      fix H
      assume "H‚ààùí©‚á©0"
      then have "{x}\<sad>H=x\<ltr> H" using interval_add(3) assms
        by auto
      with ‚ÄπH‚ààùí©‚á©0‚Ä∫ have "{x}\<sad>H‚àà{x\<ltr>H. H‚ààùí©‚á©0}" by auto
    }
    then have "{{x}\<sad>H. H‚ààùí©‚á©0}‚äÜ{x\<ltr>H. H‚ààùí©‚á©0}" by auto
    moreover
    {
      fix H
      assume "H‚ààùí©‚á©0"
      then have "{x}\<sad>H=x\<ltr> H" using interval_add(3) assms
        by auto
      with ‚ÄπH‚ààùí©‚á©0‚Ä∫ have "x\<ltr> H‚àà{{x}\<sad>H. H‚ààùí©‚á©0}" by auto
    }
    then have "{x\<ltr>H. H‚ààùí©‚á©0}‚äÜ{{x}\<sad>H. H‚ààùí©‚á©0}" by auto
    ultimately have "{{x}\<sad>H. H‚ààùí©‚á©0}={x\<ltr>H. H‚ààùí©‚á©0}" by auto
    then have "(‚ãÇH‚ààùí©‚á©0. {x}\<sad>H) = (‚ãÇH‚ààùí©‚á©0. x\<ltr>H)" by auto
    with c show "cl({x})=(‚ãÇH‚ààùí©‚á©0. x\<ltr>H)" by auto
  }
qed

theorem (in topgroup) neu_closed_imp_T1:
  assumes "{ùü¨}{is closed in}T"
  shows "T{is T‚á©1}"
proof-
  {
    fix x z assume xG:"x‚ààG" and zG:"z‚ààG" and dis:"x‚â†z"
    then have clx:"cl({x})=(‚ãÇH‚ààùí©‚á©0. x\<ltr>H)" using cl_point by auto

      fix y
      assume "y‚ààcl({x})"

      then have t:"‚àÄH‚ààùí©‚á©0. y‚ààx\<ltr>H" by auto
      from ‚Äπy‚ààcl({x})‚Ä∫ xG have yG:"y‚ààG" using Top_3_L11(1) G_def by auto
      {
        fix H
        assume HNeig:"H‚ààùí©‚á©0"
        with t have "y‚ààx\<ltr>H" by auto
        then obtain n where "y=x\<ra>n" and "n‚ààH" unfolding ltrans_def grop_def LeftTranslation_def by auto
        with HNeig have nG:"n‚ààG" unfolding zerohoods_def by auto
        from ‚Äπy=x\<ra>n‚Ä∫ and ‚Äπn‚ààH‚Ä∫ have "(\<rm>x)\<ra>y‚ààH" using group0.group0_2_L18(2) group0_valid_in_tgroup xG nG yG unfolding grinv_def grop_def
          by auto
      }
      then have el:"(\<rm>x)\<ra>y‚àà(‚ãÇùí©‚á©0)" using zneigh_not_empty by auto
      have "cl({ùü¨})=(‚ãÇH‚ààùí©‚á©0. ùü¨\<ltr>H)" using cl_point zero_in_tgroup by auto
      moreover
      {
        fix H  assume "H‚ààùí©‚á©0"
        then have "H‚äÜG" unfolding zerohoods_def by auto
        then have "ùü¨\<ltr>H=H" using image_id_same  group0.trans_neutral(2) group0_valid_in_tgroup unfolding gzero_def ltrans_def
          by auto
        with ‚ÄπH‚ààùí©‚á©0‚Ä∫ have "ùü¨\<ltr>H‚ààùí©‚á©0" "H‚àà{ùü¨\<ltr>H. H‚ààùí©‚á©0}" by auto
      }
      then have "{ùü¨\<ltr>H. H‚ààùí©‚á©0}=ùí©‚á©0" by blast
      ultimately have "cl({ùü¨})=(‚ãÇùí©‚á©0)" by auto
      with el have "(\<rm>x)\<ra>y‚ààcl({ùü¨})" by auto
      then have "(\<rm>x)\<ra>y‚àà{ùü¨}" using assms Top_3_L8 G_def zero_in_tgroup by auto
      then have "(\<rm>x)\<ra>y=ùü¨" by auto
      then have "y=\<rm>(\<rm>x)" using group0.group0_2_L9(2) group0_valid_in_tgroup neg_in_tgroup xG yG unfolding grop_def grinv_def by auto
      then have "y=x" using group0.group_inv_of_inv group0_valid_in_tgroup xG unfolding grinv_def by auto
    }
    then have "cl({x})‚äÜ{x}" by auto
    then have "cl({x})={x}" using xG cl_contains_set G_def by blast
    then have "{x}{is closed in}T" using Top_3_L8 xG G_def by auto
    then have "(‚ãÉT)-{x}‚ààT" using IsClosed_def by auto moreover
    from dis zG G_def have "z‚àà((‚ãÉT)-{x}) ‚àß x‚àâ((‚ãÉT)-{x})" by auto
    ultimately have "‚àÉV‚ààT. z‚ààV‚àßx‚àâV" by(safe,auto)
  }
  then show "T{is T‚á©1}" using isT1_def by auto
qed

theorem (in topgroup) T0_imp_neu_closed:

  shows "{ùü¨}{is closed in}T"

lemma (in topgroup) exist_basehoods_closed:
  assumes "U‚ààùí©‚á©0"
  shows "‚àÉV‚ààùí©‚á©0. cl(V)‚äÜU"
proof-
  from assms obtain V where "V‚ààùí©‚á©0" "V‚äÜU" "(V\<sad>V)‚äÜU" "(\<sm>V)=V" using exists_procls_zerohood by blast
  have inv_fun:"GroupInv(G,f)‚ààG‚ÜíG" using group0_2_T2 Ggroup by auto
  have f_fun:"f‚ààG√óG‚ÜíG" using group0.group_oper_fun group0_valid_in_tgroup by auto
  {
    fix x assume "x‚ààcl(V)"
    with ‚ÄπV‚ààùí©‚á©0‚Ä∫ have "x‚àà‚ãÉT" "V‚äÜ‚ãÉT" using Top_3_L11(1) unfolding zerohoods_def G_def by blast+
    with ‚ÄπV‚ààùí©‚á©0‚Ä∫ have "x‚ààint(x\<ltr>V)" using elem_in_int_ltrans G_def by auto
    with ‚ÄπV‚äÜ‚ãÉT‚Ä∫‚Äπx‚ààcl(V)‚Ä∫ have "int(x\<ltr>V)‚à©V‚â†0" using cl_inter_neigh Top_2_L2 by blast
    then have "(x\<ltr>V)‚à©V‚â†0" using Top_2_L1 by blast
    then obtain q where "q‚àà(x\<ltr>V)" and "q‚ààV" by blast
    with ‚ÄπV‚äÜ‚ãÉT‚Ä∫‚Äπx‚àà‚ãÉT‚Ä∫ obtain v where "q=x\<ra>v" "v‚ààV" unfolding ltrans_def grop_def using group0.ltrans_image
      group0_valid_in_tgroup unfolding G_def by auto
    from ‚ÄπV‚äÜ‚ãÉT‚Ä∫ ‚Äπv‚ààV‚Ä∫‚Äπq‚ààV‚Ä∫ have "v‚àà‚ãÉT" "q‚àà‚ãÉT" by auto
    with ‚Äπq=x\<ra>v‚Ä∫‚Äπx‚àà‚ãÉT‚Ä∫ have "q\<rs>v=x" using group0.group0_2_L18(1) group0_valid_in_tgroup unfolding G_def
        unfolding grsub_def grinv_def grop_def by auto moreover
    from ‚Äπv‚ààV‚Ä∫ have "(\<rm>v)‚àà(\<sm>V)" unfolding setninv_def grinv_def using func_imagedef inv_fun ‚ÄπV‚äÜ‚ãÉT‚Ä∫ G_def by auto
    then have "(\<rm>v)‚ààV" using ‚Äπ(\<sm>V)=V‚Ä∫ by auto
    with ‚Äπq‚ààV‚Ä∫ have "‚ü®q,\<rm>v‚ü©‚ààV√óV" by auto
    then have "f`‚ü®q,\<rm>v‚ü©‚ààV\<sad>V" using lift_subset_suff f_fun ‚ÄπV‚äÜ‚ãÉT‚Ä∫ unfolding setadd_def by auto
    with ‚ÄπV\<sad>V‚äÜU‚Ä∫ have "q\<rs>v‚ààU" unfolding grsub_def grop_def by auto
    with ‚Äπq\<rs>v=x‚Ä∫ have "x‚ààU" by auto
  }
  then have "cl(V)‚äÜU" by auto
  with ‚ÄπV‚ààùí©‚á©0‚Ä∫ show ?thesis by auto

theorem(in topgroup) T1_imp_T2:

  shows "T{is T‚á©2}"
proof-
  {
    fix x y assume ass:"x‚àà‚ãÉT" "y‚àà‚ãÉT" "x‚â†y"
    {
      assume "(\<rm>y)\<ra>x=ùü¨"
      with ass(1,2) have "y=x" using group0.group0_2_L11[where a="y" and b="x"] group0_valid_in_tgroup by auto (*cannot be erased.*)
      with ass(3) have "False" by auto
    }
    then have "(\<rm>y)\<ra>x‚â†ùü¨" by auto
    then have "ùü¨‚â†(\<rm>y)\<ra>x" by auto
    from ‚Äπy‚àà‚ãÉT‚Ä∫ have "(\<rm>y)‚àà‚ãÉT" using neg_in_tgroup G_def by auto

      G_def by auto

      by auto

        unfolding G_def by auto

      with ‚Äπx=y\<ra>u‚Ä∫ ‚Äπy‚àà‚ãÉT‚Ä∫ ‚Äπx‚àà‚ãÉT‚Ä∫ ‚ÄπQ‚äÜG‚Ä∫ have "(\<rm>y)\<ra>x=u" using group0.group0_2_L18(2) group0_valid_in_tgroup unfolding G_def
        unfolding grsub_def grinv_def grop_def by auto
      with ‚Äπu‚ààQ‚Ä∫ have "(\<rm>y)\<ra>x‚ààQ" by auto
      then have "False" using ‚Äπ(\<rm>y)\<ra>x‚àâQ‚Ä∫ by auto
    }

    {
      assume "y‚ààx\<ltr>Q"
      with ‚ÄπQ‚äÜG‚Ä∫ ‚Äπx‚àà‚ãÉT‚Ä∫ obtain u where "u‚ààQ" and "y=x\<ra>u" unfolding ltrans_def grop_def using group0.ltrans_image group0_valid_in_tgroup
        unfolding G_def by auto

      with ‚Äπy=x\<ra>u‚Ä∫ ‚Äπy‚àà‚ãÉT‚Ä∫ ‚Äπx‚àà‚ãÉT‚Ä∫ ‚ÄπQ‚äÜG‚Ä∫ have "(\<rm>x)\<ra>y=u" using group0.group0_2_L18(2) group0_valid_in_tgroup unfolding G_def
