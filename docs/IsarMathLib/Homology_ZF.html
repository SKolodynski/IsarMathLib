section ‹Homology groups›

theory Homology_ZF imports ChainComplex_ZF

begin

text ‹The \emph{cycles} at degree $n$ are the kernel of the boundary map $d_n$,
  i.e.\ the elements of $C_n$ that are sent to zero by $d_n$.
  The \emph{boundaries} at degree $n$ are the image of the boundary map $d_{n+1}$,
  i.e.\ elements of $C_n$ of the form $d_{n+1}(x)$ for some $x \in C_{n+1}$.
  The chain complex condition $d_n \circ d_{n+1} = 0$ implies that every boundary
  is a cycle.  The $n$-th \emph{homology group} is the quotient $Z_n / B_n$ of
  cycles by boundaries.›

abbreviation (in chain_complex) Cycles :: "i ⇒ i" where
  "Cycles(n) ≡ (d(n))-``{zero_C(n \<rs> 𝟭)}"

abbreviation (in chain_complex) Boundaries :: "i ⇒ i" where
  "Boundaries(n) ≡ (d(n \<ra> 𝟭))``(C(n \<ra> 𝟭))"

text ‹We denote the restriction of the group operation $P(n)$ to the cycles by
  ‹CycleOp(n)›.  This is the operation that makes $Z_n$ a group in its own right.›

abbreviation (in chain_complex) CycleOp :: "i ⇒ i" where
  "CycleOp(n) ≡ restrict(P(n), Cycles(n) × Cycles(n))"

text ‹The carrier of the $n$-th homology group is the quotient set
  $Z_n / B_n = Z_n / \!\sim$, where $x \sim y$ iff $x - y \in B_n$.›

definition (in chain_complex) Hn :: "i ⇒ i" where
  "Hn(n) ≡ Cycles(n) // QuotientGroupRel(Cycles(n), CycleOp(n), Boundaries(n))"

definition (in chain_complex) HnOp :: "i ⇒ i" where
  "HnOp(n) ≡ QuotientGroupOp(Cycles(n), CycleOp(n), Boundaries(n))"

text ‹The key structural fact: every boundary is a cycle.  This is a direct
  consequence of the boundary-squared-zero condition in the chain complex.›

lemma (in chain_complex) boundaries_in_cycles:
  assumes "n ∈ ℤ"
  shows "Boundaries(n) ⊆ Cycles(n)"
proof -
  have np: "n \<ra> 𝟭 ∈ ℤ"
    using assms Int_ZF_1_L8A(2) Int_ZF_1_T2(3) group0.group_op_closed by simp
  have arith1: "(n \<ra> 𝟭) \<rs> 𝟭 = n"
    using Int_ZF_1_L8A(2) group0.inv_cancel_two(2)[OF Int_ZF_1_T2(3)] assms by simp
  have arith2: "(n \<ra> 𝟭) \<rs> 𝟭 \<rs> 𝟭 = n \<rs> 𝟭"
    using arith1 by simp
  from bnd_sq_zero[OF np] arith1 arith2 show ?thesis
    by (simp add: Int_ZF_1_L8 Int_ZF_1_L10)
qed

text ‹Cycles at degree $n$ form a subgroup of $C_n$.›

lemma (in chain_complex) cycles_is_subgroup:
  assumes "n ∈ ℤ"
  shows "IsAsubgroup(Cycles(n), P(n))"
proof -
  have pn: "n \<rs> 𝟭 ∈ ℤ"
    using assms Int_ZF_1_L10(1) Int_ZF_1_L8A(2) by simp
  interpret dn: group_homo "C(n)" "P(n)" "C(n \<rs> 𝟭)" "P(n \<rs> 𝟭)" "d(n)"
    "TheNeutralElement(C(n), P(n))" "λx y. P(n)`⟨x, y⟩"
    "λx. GroupInv(C(n), P(n))`x"
    "λs. Fold(P(n), TheNeutralElement(C(n), P(n)), s)"
    "λm x. Fold(P(n), TheNeutralElement(C(n), P(n)), {⟨k, x⟩. k ∈ m})"
    "TheNeutralElement(C(n \<rs> 𝟭), P(n \<rs> 𝟭))" "λx y. P(n \<rs> 𝟭)`⟨x, y⟩"
    "λx. GroupInv(C(n \<rs> 𝟭), P(n \<rs> 𝟭))`x"
    "λs. Fold(P(n \<rs> 𝟭), TheNeutralElement(C(n \<rs> 𝟭), P(n \<rs> 𝟭)), s)"
    "λm x. Fold(P(n \<rs> 𝟭), TheNeutralElement(C(n \<rs> 𝟭), P(n \<rs> 𝟭)), {⟨k, x⟩. k ∈ m})"
    using cplx_group[OF assms] cplx_group[OF pn] cplx_hom[OF assms]
    unfolding group_homo_def by simp
  from dn.kernel_is_subgroup show ?thesis by simp
qed

text ‹Cycles at degree $n$ form a group (with the restricted operation).›

lemma (in chain_complex) cycles_is_group:
  assumes "n ∈ ℤ"
  shows "IsAgroup(Cycles(n), CycleOp(n))"
  using cycles_is_subgroup[OF assms] unfolding IsAsubgroup_def by simp

text ‹The cycle group at degree $n$ is abelian, inheriting commutativity from $C_n$.›

lemma (in chain_complex) cycles_is_abelian:
  assumes "n ∈ ℤ"
  shows "CycleOp(n) {is commutative on} Cycles(n)"
proof -
  have pn_fun: "P(n) : C(n) × C(n) → C(n)"
    using cplx_group[OF assms]
     group0.group_oper_fun unfolding group0_def by auto
  have sub: "Cycles(n) ⊆ C(n)"
    using cycles_is_subgroup[OF assms] cplx_group[OF assms]
    group0.group0_3_L2 unfolding group0_def by auto
  from func_ZF_4_L1[OF pn_fun sub cplx_abelian[OF assms]]
  show ?thesis by simp
qed

text ‹Boundaries at degree $n$ form a subgroup of the ambient group $C_n$.›

lemma (in chain_complex) boundaries_subgroup_ambient:
  assumes "n ∈ ℤ"
  shows "IsAsubgroup(Boundaries(n), P(n))"
proof -
  have np: "n \<ra> 𝟭 ∈ ℤ"
    using assms Int_ZF_1_L8A(2) Int_ZF_1_T2(3) group0.group_op_closed by simp
  have arith1: "(n \<ra> 𝟭) \<rs> 𝟭 = n"
    using Int_ZF_1_L8A(2) group0.inv_cancel_two(2)[OF Int_ZF_1_T2(3)] assms by simp
  have ppn: "(n \<ra> 𝟭) \<rs> 𝟭 ∈ ℤ" using arith1 assms by simp
  interpret dn1: group_homo "C(n \<ra> 𝟭)" "P(n \<ra> 𝟭)"
    "C((n \<ra> 𝟭) \<rs> 𝟭)" "P((n \<ra> 𝟭) \<rs> 𝟭)" "d(n \<ra> 𝟭)"
    "TheNeutralElement(C(n \<ra> 𝟭), P(n \<ra> 𝟭))" "λx y. P(n \<ra> 𝟭)`⟨x, y⟩"
    "λx. GroupInv(C(n \<ra> 𝟭), P(n \<ra> 𝟭))`x"
    "λs. Fold(P(n \<ra> 𝟭), TheNeutralElement(C(n \<ra> 𝟭), P(n \<ra> 𝟭)), s)"
    "λm x. Fold(P(n \<ra> 𝟭), TheNeutralElement(C(n \<ra> 𝟭), P(n \<ra> 𝟭)), {⟨k, x⟩. k ∈ m})"
    "TheNeutralElement(C((n \<ra> 𝟭) \<rs> 𝟭), P((n \<ra> 𝟭) \<rs> 𝟭))"
    "λx y. P((n \<ra> 𝟭) \<rs> 𝟭)`⟨x, y⟩"
    "λx. GroupInv(C((n \<ra> 𝟭) \<rs> 𝟭), P((n \<ra> 𝟭) \<rs> 𝟭))`x"
    "λs. Fold(P((n \<ra> 𝟭) \<rs> 𝟭), TheNeutralElement(C((n \<ra> 𝟭) \<rs> 𝟭), P((n \<ra> 𝟭) \<rs> 𝟭)), s)"
    "λm x. Fold(P((n \<ra> 𝟭) \<rs> 𝟭), TheNeutralElement(C((n \<ra> 𝟭) \<rs> 𝟭), P((n \<ra> 𝟭) \<rs> 𝟭)),
             {⟨k, x⟩. k ∈ m})"
    using cplx_group[OF np] cplx_group[OF ppn] cplx_hom[OF np]
    unfolding group_homo_def by simp
  from dn1.image_is_group have "IsAsubgroup(Boundaries(n), P((n \<ra> 𝟭) \<rs> 𝟭))"
    by simp
  with arith1 show ?thesis by simp
qed

text ‹Boundaries at degree $n$ form a subgroup of the cycle group $Z_n$.›

lemma (in chain_complex) boundaries_subgroup_cycles:
  assumes "n ∈ ℤ"
  shows "IsAsubgroup(Boundaries(n), CycleOp(n))"
proof -
  have bc: "Boundaries(n) ⊆ Cycles(n)"
    using boundaries_in_cycles[OF assms] by simp
  have bsub: "IsAsubgroup(Boundaries(n), P(n))"
    using boundaries_subgroup_ambient[OF assms] by simp
  have inter: "Cycles(n) × Cycles(n) ∩ Boundaries(n) × Boundaries(n) =
               Boundaries(n) × Boundaries(n)"
    using bc by auto
  from bsub inter show ?thesis using restrict_restrict
    unfolding IsAsubgroup_def
    by simp
qed

subsection ‹The $n$-th homology group›

text ‹The $n$-th homology group $H_n = Z_n / B_n$ is a group.›

theorem (in chain_complex) Hn_is_group:
  assumes "n ∈ ℤ"
  shows "IsAgroup(Hn(n), HnOp(n))"
proof -
  have normal: "IsAnormalSubgroup(Cycles(n), CycleOp(n), Boundaries(n))"
    using Group_ZF_2_4_L6(1)[OF cycles_is_group[OF assms]
                               cycles_is_abelian[OF assms]
                               boundaries_subgroup_cycles[OF assms]] by simp
  from Group_ZF_2_4_T1[OF cycles_is_group[OF assms] normal]
  show ?thesis unfolding Hn_def HnOp_def by simp
qed

text ‹The $n$-th homology group is abelian, as a quotient of the abelian cycle group.›

theorem (in chain_complex) Hn_is_abelian:
  assumes "n ∈ ℤ"
  shows "HnOp(n) {is commutative on} Hn(n)"
proof -
  from Group_ZF_2_4_L6(2)[OF cycles_is_group[OF assms]
                             cycles_is_abelian[OF assms]
                             boundaries_subgroup_cycles[OF assms]]
  show ?thesis unfolding HnOp_def Hn_def by simp
qed

subsection ‹Chain maps induce homomorphisms on homology›

text‹A chain map sends cycles to cycles: $d_n(z)=0$ implies $d'_n(f_n(z))=0$.›

lemma (in chain_map) cmap_maps_cycles:
  assumes "n ∈ ints" and "z ∈ src.Cycles(n)"
  shows "f(n)`z ∈ tgt.Cycles(n)"
proof -
  have pn: "n \<rs> 𝟭 ∈ ints"
    using assms(1) Int_ZF_1_L10(1) Int_ZF_1_L8A(2) by simp
  have fn: "f(n) : C(n) → C'(n)"
    using cmap_hom[OF assms(1)] unfolding Homomor_def by simp
  have fnp: "f(n \<rs> 𝟭) : C(n \<rs> 𝟭) → C'(n \<rs> 𝟭)"
    using cmap_hom[OF pn] unfolding Homomor_def by simp
  have dn: "d(n) : C(n) → C(n \<rs> 𝟭)"
    using src.cplx_hom[OF assms(1)] unfolding Homomor_def by simp
  have dn': "d'(n) : C'(n) → C'(n \<rs> 𝟭)"
    using tgt.cplx_hom[OF assms(1)] unfolding Homomor_def by simp
  from assms(2) have zC: "z ∈ C(n)" and dzn: "d(n)`z = src.zero_C(n \<rs> 𝟭)"
    using func1_1_L15[OF dn] by auto
  from fn zC have fzC': "f(n)`z ∈ C'(n)" using apply_type by simp
  interpret fn_m1: group_homo
    "C(n \<rs> 𝟭)" "P(n \<rs> 𝟭)" "C'(n \<rs> 𝟭)" "P'(n \<rs> 𝟭)" "f(n \<rs> 𝟭)"
    "TheNeutralElement(C(n \<rs> 𝟭), P(n \<rs> 𝟭))"
    "λx y. P(n \<rs> 𝟭)`⟨x, y⟩"
    "λx. GroupInv(C(n \<rs> 𝟭), P(n \<rs> 𝟭))`x"
    "λs. Fold(P(n \<rs> 𝟭), TheNeutralElement(C(n \<rs> 𝟭), P(n \<rs> 𝟭)), s)"
    "λm x. Fold(P(n \<rs> 𝟭), TheNeutralElement(C(n \<rs> 𝟭), P(n \<rs> 𝟭)), {⟨k, x⟩. k ∈ m})"
    "TheNeutralElement(C'(n \<rs> 𝟭), P'(n \<rs> 𝟭))"
    "λx y. P'(n \<rs> 𝟭)`⟨x, y⟩"
    "λx. GroupInv(C'(n \<rs> 𝟭), P'(n \<rs> 𝟭))`x"
    "λs. Fold(P'(n \<rs> 𝟭), TheNeutralElement(C'(n \<rs> 𝟭), P'(n \<rs> 𝟭)), s)"
    "λm x. Fold(P'(n \<rs> 𝟭), TheNeutralElement(C'(n \<rs> 𝟭), P'(n \<rs> 𝟭)), {⟨k, x⟩. k ∈ m})"
    using src.cplx_group[OF pn] tgt.cplx_group[OF pn] cmap_hom[OF pn]
    unfolding group_homo_def by simp
  have key: "d'(n)`(f(n)`z) = tgt.zero_C(n \<rs> 𝟭)"
  proof -
    have eq1: "(d'(n) O f(n))`z = d'(n)`(f(n)`z)"
      using comp_fun_apply fn dn' zC by simp
    have eq2: "(f(n \<rs> 𝟭) O d(n))`z = f(n \<rs> 𝟭)`(d(n)`z)"
      using comp_fun_apply dn fnp zC by simp
    have eq3: "d'(n) O f(n) = f(n \<rs> 𝟭) O d(n)"
      using cmap_comm[OF assms(1)] by simp
    from eq1 eq2 eq3 have "d'(n)`(f(n)`z) = f(n \<rs> 𝟭)`(d(n)`z)" by simp
    with dzn fn_m1.f_neutral show ?thesis by simp
  qed
  from fzC' key show ?thesis using func1_1_L15[OF dn'] by auto
qed

text‹A chain map sends boundaries to boundaries: the image of $d_{n+1}$
  maps into the image of $d'_{n+1}$.›

lemma (in chain_map) cmap_maps_boundaries:
  assumes "n ∈ ints" and "z ∈ src.Boundaries(n)"
  shows "f(n)`z ∈ tgt.Boundaries(n)"
proof -
  have np: "n \<ra> 𝟭 ∈ ints"
    using assms(1) Int_ZF_1_L8A(2) Int_ZF_1_T2(3) group0.group_op_closed by simp
  have arith1: "(n \<ra> 𝟭) \<rs> 𝟭 = n"
    using Int_ZF_1_L8A(2) group0.inv_cancel_two(2)[OF Int_ZF_1_T2(3)] assms(1) by simp
  have dnp: "d(n \<ra> 𝟭) : C(n \<ra> 𝟭) → C(n)"
    using src.cplx_hom[OF np] arith1 unfolding Homomor_def by simp
  have dnp': "d'(n \<ra> 𝟭) : C'(n \<ra> 𝟭) → C'(n)"
    using tgt.cplx_hom[OF np] arith1 unfolding Homomor_def by simp
  have fn: "f(n) : C(n) → C'(n)"
    using cmap_hom[OF assms(1)] unfolding Homomor_def by simp
  have fnp: "f(n \<ra> 𝟭) : C(n \<ra> 𝟭) → C'(n \<ra> 𝟭)"
    using cmap_hom[OF np] unfolding Homomor_def by simp
  have comm_np: "d'(n \<ra> 𝟭) O f(n \<ra> 𝟭) = f(n) O d(n \<ra> 𝟭)"
    using cmap_comm[OF np] arith1 by simp
  from assms(2) obtain b where bC: "b ∈ C(n \<ra> 𝟭)" and zb: "z = d(n \<ra> 𝟭)`b"
    using func_imagedef[OF dnp] by auto
  from bC fnp have fbC': "f(n \<ra> 𝟭)`b ∈ C'(n \<ra> 𝟭)" using apply_type by simp
  have key: "f(n)`z = d'(n \<ra> 𝟭)`(f(n \<ra> 𝟭)`b)"
  proof -
    have "f(n)`z = f(n)`(d(n \<ra> 𝟭)`b)" using zb by simp
    also have "… = (f(n) O d(n \<ra> 𝟭))`b"
      using comp_fun_apply dnp fn bC by simp
    also have "… = (d'(n \<ra> 𝟭) O f(n \<ra> 𝟭))`b"
      using comm_np by simp
    also have "… = d'(n \<ra> 𝟭)`(f(n \<ra> 𝟭)`b)"
      using comp_fun_apply fnp dnp' bC by simp
    finally show ?thesis .
  qed
  have img: "(d'(n \<ra> 𝟭))``(C'(n \<ra> 𝟭)) = {d'(n \<ra> 𝟭)`x. x ∈ C'(n \<ra> 𝟭)}"
    using func_imagedef[OF dnp' subset_refl] by simp
  from fbC' key img show ?thesis by auto
qed

text‹The map on homology induced by a chain map $f$: it sends the class $[z]$
  in $H_n(C)$ to the class $[f_n(z)]$ in $H_n(C')$.›

definition (in chain_map) Hn_map :: "i ⇒ i" where
  "Hn_map(n) ≡
    {⟨QuotientGroupRel(src.Cycles(n), src.CycleOp(n), src.Boundaries(n))``{z},
       QuotientGroupRel(tgt.Cycles(n), tgt.CycleOp(n), tgt.Boundaries(n))``{f(n)`z}⟩.
     z ∈ src.Cycles(n)}"

text‹The map induced by a chain map is a group homomorphism $H_n(C) \to H_n(C')$.›

theorem (in chain_map) Hn_map_is_hom:
  assumes "n ∈ ints"
  shows "Homomor(Hn_map(n), src.Hn(n), src.HnOp(n), tgt.Hn(n), tgt.HnOp(n))"
proof -
  let ?Rs = "QuotientGroupRel(src.Cycles(n), src.CycleOp(n), src.Boundaries(n))"
  let ?Rt = "QuotientGroupRel(tgt.Cycles(n), tgt.CycleOp(n), tgt.Boundaries(n))"
  let ?M = "Hn_map(n)"
  have M_eq: "?M = {⟨?Rs``{z}, ?Rt``{f(n)`z}⟩. z ∈ src.Cycles(n)}"
    unfolding Hn_map_def by simp
  have normal_s: "IsAnormalSubgroup(src.Cycles(n), src.CycleOp(n), src.Boundaries(n))"
    using Group_ZF_2_4_L6(1)[OF src.cycles_is_group[OF assms]
                                     src.cycles_is_abelian[OF assms]
                                     src.boundaries_subgroup_cycles[OF assms]] by simp
  have normal_t: "IsAnormalSubgroup(tgt.Cycles(n), tgt.CycleOp(n), tgt.Boundaries(n))"
    using Group_ZF_2_4_L6(1)[OF tgt.cycles_is_group[OF assms]
                                     tgt.cycles_is_abelian[OF assms]
                                     tgt.boundaries_subgroup_cycles[OF assms]] by simp
  have equiv_s: "equiv(src.Cycles(n), ?Rs)"
    using src.cycles_is_subgroup[OF assms] group0.Group_ZF_2_4_L3[OF _ src.boundaries_subgroup_cycles[OF assms],
        of "src.Cycles(n)"]
    unfolding IsAsubgroup_def group0_def by auto
  have equiv_t: "equiv(tgt.Cycles(n), ?Rt)"
    using tgt.cycles_is_subgroup[OF assms] group0.Group_ZF_2_4_L3[OF _ tgt.boundaries_subgroup_cycles[OF assms]]
    unfolding IsAsubgroup_def group0_def by simp
  have cong_s: "Congruent2(?Rs, src.CycleOp(n))"
    using Group_ZF_2_4_L5A src.cycles_is_group[OF assms] normal_s by simp
  have cong_t: "Congruent2(?Rt, tgt.CycleOp(n))"
    using Group_ZF_2_4_L5A tgt.cycles_is_group[OF assms] normal_t by simp
  have fn: "f(n) : C(n) → C'(n)"
    using cmap_hom[OF assms] unfolding Homomor_def by simp
  interpret Cn: group0 "C(n)" "P(n)" "src.zero_C(n)" "λx y. P(n)`⟨x,y⟩"
    "λx. GroupInv(C(n),P(n))`x" "λs. Fold(P(n), src.zero_C(n), s)"
    "λna x. Fold(P(n), src.zero_C(n), {⟨k, x⟩. k ∈ na})"
    using src.cplx_group[OF assms] unfolding group0_def by simp
  interpret Cn': group0 "C'(n)" "P'(n)" "tgt.zero_C(n)" "λx y. P'(n)`⟨x,y⟩"
    "λx. GroupInv(C'(n),P'(n))`x" "λs. Fold(P'(n), tgt.zero_C(n), s)"
    "λna x. Fold(P'(n), tgt.zero_C(n), {⟨k, x⟩. k ∈ na})"
    using tgt.cplx_group[OF assms] unfolding group0_def by simp
  interpret fn_homo: group_homo "C(n)" "P(n)" "C'(n)" "P'(n)" "f(n)"
    "TheNeutralElement(C(n), P(n))"
    "λx y. P(n)`⟨x, y⟩"
    "λx. GroupInv(C(n), P(n))`x"
    "λs. Fold(P(n), TheNeutralElement(C(n), P(n)), s)"
    "λm x. Fold(P(n), TheNeutralElement(C(n), P(n)), {⟨k, x⟩. k ∈ m})"
    "TheNeutralElement(C'(n), P'(n))"
    "λx y. P'(n)`⟨x, y⟩"
    "λx. GroupInv(C'(n), P'(n))`x"
    "λs. Fold(P'(n), TheNeutralElement(C'(n), P'(n)), s)"
    "λm x. Fold(P'(n), TheNeutralElement(C'(n), P'(n)), {⟨k, x⟩. k ∈ m})"
    using src.cplx_group[OF assms] tgt.cplx_group[OF assms] cmap_hom[OF assms]
    unfolding group_homo_def by simp
  have sub_s: "src.Cycles(n) ⊆ C(n)"
    using src.cycles_is_subgroup[OF assms] Cn.group0_3_L2 by simp
  have sub_t: "tgt.Cycles(n) ⊆ C'(n)"
    using tgt.cycles_is_subgroup[OF assms] Cn'.group0_3_L2 by simp
  have inv_in_cycles_s: "∀h ∈ src.Cycles(n). GroupInv(src.Cycles(n), src.CycleOp(n))`h = GroupInv(C(n), P(n))`h"
    using Cn.group0_3_T2[OF src.cycles_is_subgroup[OF assms]] by blast
  have inv_in_cycles_t: "∀h ∈ tgt.Cycles(n). GroupInv(tgt.Cycles(n), tgt.CycleOp(n))`h = GroupInv(C'(n), P'(n))`h"
    using Cn'.group0_3_T2[OF tgt.cycles_is_subgroup[OF assms]] by blast
  have M_subset: "?M ∈ Pow(src.Hn(n) × tgt.Hn(n))"
  proof(safe)
    fix p assume "p ∈ ?M"
    with M_eq obtain z where zZ: "z ∈ src.Cycles(n)"
      and p_eq: "p = ⟨?Rs``{z}, ?Rt``{f(n)`z}⟩" by auto
    from zZ equiv_s have Rs_cls: "?Rs``{z} ∈ src.Hn(n)"
      unfolding src.Hn_def quotient_def by auto
    from cmap_maps_cycles[OF assms zZ] equiv_t have Rt_cls: "?Rt``{f(n)`z} ∈ tgt.Hn(n)"
      unfolding tgt.Hn_def quotient_def by auto
    from p_eq Rs_cls Rt_cls show "p ∈ src.Hn(n) × tgt.Hn(n)" by auto
  qed
  have M_dom: "src.Hn(n) ⊆ domain(?M)"
    using M_eq unfolding src.Hn_def quotient_def domain_def by auto
  have M_sv: "∀A y t. ⟨A,y⟩ ∈ ?M ⟶ ⟨A,t⟩ ∈ ?M ⟶ y = t"
  proof (intro allI impI)
    fix A y t
    assume Ay: "⟨A,y⟩ ∈ ?M" and At: "⟨A,t⟩ ∈ ?M"
    from Ay M_eq obtain z⇩y where zy: "z⇩y ∈ src.Cycles(n)"
      and Ay_eq: "⟨A,y⟩ = ⟨?Rs``{z⇩y}, ?Rt``{f(n)`z⇩y}⟩" by auto
    from At M_eq obtain z⇩t where zt: "z⇩t ∈ src.Cycles(n)"
      and At_eq: "⟨A,t⟩ = ⟨?Rs``{z⇩t}, ?Rt``{f(n)`z⇩t}⟩" by auto
    from Ay_eq At_eq have cls_eq: "?Rs``{z⇩y} = ?Rs``{z⇩t}"
      and y_eq: "y = ?Rt``{f(n)`z⇩y}" and t_eq: "t = ?Rt``{f(n)`z⇩t}" by auto
    from equiv_s zt cls_eq have rel_yz: "⟨z⇩y, z⇩t⟩ ∈ ?Rs"
      using same_image_equiv by simp
    from rel_yz have diff_B:
      "src.CycleOp(n)`⟨z⇩y, GroupInv(src.Cycles(n), src.CycleOp(n))`z⇩t⟩ ∈ src.Boundaries(n)"
      unfolding QuotientGroupRel_def by auto
    from zt inv_in_cycles_s have inv_eq:
      "GroupInv(src.Cycles(n), src.CycleOp(n))`z⇩t = GroupInv(C(n), P(n))`z⇩t" by blast
    from zy zt sub_s have zy_C: "z⇩y ∈ C(n)" and zt_C: "z⇩t ∈ C(n)" by auto
    from zt_C have zinv_C: "GroupInv(C(n), P(n))`z⇩t ∈ C(n)" using Cn.inverse_in_group by blast
    have "GroupInv(src.Cycles(n), src.CycleOp(n))`z⇩t ∈ src.Cycles(n)"
      using group0.inverse_in_group[OF _ zt, of "restrict(P(n), src.Cycles(n) × src.Cycles(n))"]
       src.cycles_is_group[OF assms] unfolding group0_def by auto
    with zy have "⟨z⇩y,GroupInv(src.Cycles(n), src.CycleOp(n))`z⇩t⟩∈src.Cycles(n)×src.Cycles(n)"
      by auto
    then have "(if (⟨z⇩y,GroupInv(src.Cycles(n), src.CycleOp(n))`z⇩t⟩∈src.Cycles(n)×src.Cycles(n)) then  (P(n) `⟨z⇩y,GroupInv(src.Cycles(n), src.CycleOp(n))`z⇩t⟩) else 0) 
      = (P(n)`⟨z⇩y,GroupInv(src.Cycles(n), src.CycleOp(n))`z⇩t⟩)"
      apply (simp only:) by (simp only:if_true)
    then have "restrict(P(n), src.Cycles(n) × src.Cycles(n))`⟨z⇩y,GroupInv(src.Cycles(n), src.CycleOp(n))`z⇩t⟩ = 
      P(n)`⟨z⇩y,GroupInv(src.Cycles(n), src.CycleOp(n))`z⇩t⟩"
      using restrict[of "P(n)" " src.Cycles(n) × src.Cycles(n)" "⟨z⇩y,GroupInv(src.Cycles(n), src.CycleOp(n))`z⇩t⟩"]
      by (simp only:)
    with inv_eq have cycle_op_eq:
      "src.CycleOp(n)`⟨z⇩y, GroupInv(src.Cycles(n), src.CycleOp(n))`z⇩t⟩ =
       P(n)`⟨z⇩y, GroupInv(C(n), P(n))`z⇩t⟩"
      by (simp only:)
    from diff_B cycle_op_eq have diff_C:
      "P(n)`⟨z⇩y, GroupInv(C(n), P(n))`z⇩t⟩ ∈ src.Boundaries(n)" by (simp only:)
    from cmap_maps_boundaries[OF assms diff_C] have f_diff:
      "f(n)`(P(n)`⟨z⇩y, GroupInv(C(n), P(n))`z⇩t⟩) ∈ tgt.Boundaries(n)" .
    from fn_homo.f_hom zy_C zinv_C have f_hom_step:
      "f(n)`(P(n)`⟨z⇩y, GroupInv(C(n), P(n))`z⇩t⟩) = P'(n)`⟨f(n)`z⇩y, f(n)`(GroupInv(C(n), P(n))`z⇩t)⟩"
      by (simp only:)
    from fn_homo.f_inv zt_C have f_inv_step:
      "f(n)`(GroupInv(C(n), P(n))`z⇩t) = GroupInv(C'(n), P'(n))`(f(n)`z⇩t)" by (simp only:)
    from f_diff f_hom_step f_inv_step have Rt_mem:
      "P'(n)`⟨f(n)`z⇩y, GroupInv(C'(n), P'(n))`(f(n)`z⇩t)⟩ ∈ tgt.Boundaries(n)" by (simp only:)
    from cmap_maps_cycles[OF assms zy] have fzy_tC: "f(n)`z⇩y ∈ tgt.Cycles(n)" .
    from cmap_maps_cycles[OF assms zt] have fzt_tC: "f(n)`z⇩t ∈ tgt.Cycles(n)" .
    from fzt_tC inv_in_cycles_t have tgt_inv_eq:
      "GroupInv(tgt.Cycles(n), tgt.CycleOp(n))`(f(n)`z⇩t) = GroupInv(C'(n), P'(n))`(f(n)`z⇩t)" by (simp only:)
    have "GroupInv(tgt.Cycles(n), tgt.CycleOp(n))`(f(n)`z⇩t) ∈ tgt.Cycles(n)"
      using group0.inverse_in_group[OF _ fzt_tC, of "restrict(P'(n), tgt.Cycles(n) × tgt.Cycles(n))"]
       tgt.cycles_is_group[OF assms] unfolding group0_def by (simp only:)
    with fzy_tC have "⟨f(n)`z⇩y,GroupInv(tgt.Cycles(n), tgt.CycleOp(n))`(f(n)`z⇩t)⟩∈tgt.Cycles(n)×tgt.Cycles(n)"
      by auto
    then have "(if (⟨f(n)`z⇩y,GroupInv(tgt.Cycles(n), tgt.CycleOp(n))`(f(n)`z⇩t)⟩∈tgt.Cycles(n)×tgt.Cycles(n)) then  (P'(n) `⟨f(n)`z⇩y,GroupInv(tgt.Cycles(n), tgt.CycleOp(n))`(f(n)`z⇩t)⟩) else 0) 
      = (P'(n)`⟨f(n)`z⇩y,GroupInv(tgt.Cycles(n), tgt.CycleOp(n))`(f(n)`z⇩t)⟩)"
      apply (simp only:) by (simp only:if_true)
    then have "restrict(P'(n), tgt.Cycles(n) × tgt.Cycles(n))`⟨f(n)`z⇩y,GroupInv(tgt.Cycles(n), tgt.CycleOp(n))`(f(n)`z⇩t)⟩ = 
      P'(n)`⟨f(n)`z⇩y,GroupInv(tgt.Cycles(n), tgt.CycleOp(n))`(f(n)`z⇩t)⟩"
      using restrict[of "P'(n)" "tgt.Cycles(n) × tgt.Cycles(n)" "⟨f(n)`z⇩y,GroupInv(tgt.Cycles(n), tgt.CycleOp(n))`(f(n)`z⇩t)⟩"]
      by (simp only:)
    with tgt_inv_eq have tgt_op_eq:
      "tgt.CycleOp(n)`⟨f(n)`z⇩y, GroupInv(tgt.Cycles(n), tgt.CycleOp(n))`(f(n)`z⇩t)⟩ =
       P'(n)`⟨f(n)`z⇩y, GroupInv(C'(n), P'(n))`(f(n)`z⇩t)⟩"
      by (simp only:)
    with Rt_mem have "tgt.CycleOp(n)`⟨f(n)`z⇩y, GroupInv(tgt.Cycles(n), tgt.CycleOp(n))`(f(n)`z⇩t)⟩∈tgt.Boundaries(n)"
      by (simp only:)
    then have Rt_rel: "⟨f(n)`z⇩y, f(n)`z⇩t⟩ ∈ ?Rt"
      using fzy_tC fzt_tC unfolding QuotientGroupRel_def by auto
    from equiv_t fzt_tC Rt_rel have "?Rt``{f(n)`z⇩y} = ?Rt``{f(n)`z⇩t}"
      using equiv_class_eq by simp
    with y_eq t_eq show "y = t" by simp
  qed
  from M_subset M_dom M_sv have M_fun: "?M : src.Hn(n) → tgt.Hn(n)"
    unfolding Pi_def function_def by auto
  have morphism: "IsMorphism(src.Hn(n), src.HnOp(n), tgt.HnOp(n), ?M)"
    unfolding IsMorphism_def
  proof (intro ballI)
    fix A⇩1 A⇩2 assume A1: "A⇩1 ∈ src.Hn(n)" and A2: "A⇩2 ∈ src.Hn(n)"
    show "?M`(src.HnOp(n)`⟨A⇩1, A⇩2⟩) = tgt.HnOp(n)`⟨?M`A⇩1, ?M`A⇩2⟩"
    proof-
      from A1 obtain z⇩1 where z1: "z⇩1 ∈ src.Cycles(n)" and A1_eq: "A⇩1 = ?Rs``{z⇩1}"
        unfolding src.Hn_def quotient_def by auto
      from A2 obtain z⇩2 where z2: "z⇩2 ∈ src.Cycles(n)" and A2_eq: "A⇩2 = ?Rs``{z⇩2}"
        unfolding src.Hn_def quotient_def by auto
      from z1 M_eq A1_eq have pair1: "⟨A⇩1, ?Rt``{f(n)`z⇩1}⟩ ∈ ?M" by auto
      from z2 M_eq A2_eq have pair2: "⟨A⇩2, ?Rt``{f(n)`z⇩2}⟩ ∈ ?M" by auto
      from pair1 M_fun have step1: "?M`A⇩1 = ?Rt``{f(n)`z⇩1}"
        using apply_equality by simp
      from pair2 M_fun have step2: "?M`A⇩2 = ?Rt``{f(n)`z⇩2}"
        using apply_equality by simp
      from z1 z2 sub_s have z1C: "z⇩1 ∈ C(n)" and z2C: "z⇩2 ∈ C(n)" by auto
      have cycle_closed: "src.CycleOp(n)`⟨z⇩1, z⇩2⟩ ∈ src.Cycles(n)"
        using src.cycles_is_group[OF assms] z1 z2
          group0.group_op_closed unfolding group0_def by (simp only:)
      have op_s: "src.HnOp(n)`⟨A⇩1, A⇩2⟩ = ?Rs``{src.CycleOp(n)`⟨z⇩1, z⇩2⟩}"
        using group0.Group_ZF_2_2_L2[OF _ equiv_s cong_s _ z1 z2] A1_eq A2_eq
        src.cycles_is_group[OF assms]
        unfolding src.HnOp_def QuotientGroupOp_def group0_def by auto
      from M_eq have "⋀z. z∈src.Cycles(n) ⟹ ⟨?Rs``{z},?Rt``{f(n)`z}⟩∈?M" by auto
      with cycle_closed have pair_prod:
        "⟨?Rs``{src.CycleOp(n)`⟨z⇩1, z⇩2⟩}, ?Rt``{f(n)`(src.CycleOp(n)`⟨z⇩1, z⇩2⟩)}⟩ ∈ ?M"
        by auto
      with op_s have "⟨src.HnOp(n)`⟨A⇩1, A⇩2⟩, ?Rt``{f(n)`(src.CycleOp(n)`⟨z⇩1, z⇩2⟩)}⟩ ∈ ?M"
        by auto
      with M_fun have step3:
        "?M`(src.HnOp(n)`⟨A⇩1, A⇩2⟩) = ?Rt``{f(n)`(src.CycleOp(n)`⟨z⇩1, z⇩2⟩)}"
        using apply_equality by (simp only:)
      from z1 z2 have "⟨z⇩1,z⇩2⟩∈src.Cycles(n)×src.Cycles(n)" by auto
      then have cycleop_eq: "src.CycleOp(n)`⟨z⇩1, z⇩2⟩ = P(n)`⟨z⇩1, z⇩2⟩"
        using restrict[of "P(n)" "src.Cycles(n)×src.Cycles(n)" "⟨z⇩1,z⇩2⟩"] by (simp only:if_true)
      from fn_homo.f_hom z1C z2C have f_prod:
        "f(n)`(P(n)`⟨z⇩1, z⇩2⟩) = P'(n)`⟨f(n)`z⇩1, f(n)`z⇩2⟩" by (simp only:)
      from cmap_maps_cycles[OF assms z1] cmap_maps_cycles[OF assms z2] have
        fz1_tC: "f(n)`z⇩1 ∈ tgt.Cycles(n)" and fz2_tC: "f(n)`z⇩2 ∈ tgt.Cycles(n)" by simp_all
      from fz1_tC fz2_tC have "⟨f(n)`z⇩1,f(n)`z⇩2⟩∈tgt.Cycles(n)×tgt.Cycles(n)" by auto
      then have tgt_cycleop_eq:
        "tgt.CycleOp(n)`⟨f(n)`z⇩1, f(n)`z⇩2⟩ = P'(n)`⟨f(n)`z⇩1, f(n)`z⇩2⟩"
        using restrict by (simp only:if_true)
      have op_t: "tgt.HnOp(n)`⟨?Rt``{f(n)`z⇩1}, ?Rt``{f(n)`z⇩2}⟩ =
                  ?Rt``{tgt.CycleOp(n)`⟨f(n)`z⇩1, f(n)`z⇩2⟩}"
        using group0.Group_ZF_2_2_L2[OF _ equiv_t cong_t _ fz1_tC fz2_tC]
        tgt.cycles_is_group[OF assms]
        unfolding tgt.HnOp_def QuotientGroupOp_def group0_def by auto
      from step3 cycleop_eq f_prod tgt_cycleop_eq op_t step1 step2
      show "?M`(src.HnOp(n)`⟨A⇩1, A⇩2⟩) = tgt.HnOp(n)`⟨?M`A⇩1, ?M`A⇩2⟩"
        by (simp only:)
    qed
  qed
  from M_fun morphism show ?thesis unfolding Homomor_def by simp
qed
