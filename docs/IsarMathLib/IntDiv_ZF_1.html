section ‹Integer division: three as a modulus›

theory IntDiv_ZF_1 imports IntDiv_ZF_IML

begin

text‹This theory collects arithmetic facts about integer division and remainders
  that are needed for indexing constructions, in particular the properties of $3$
  as a modulus used to interleave three families of groups into a single
  $\mathbb{Z}$-indexed family (the long exact sequence in homology).›

text‹The left summand can be reduced modulo $p$ without changing the remainder
  of the sum: $(n+m) \bmod p = ((n \bmod p) + m) \bmod p$.›

lemma (in int0) zmod_add_left:
  assumes "n∈ℤ" "m∈ℤ" "p∈ℤ"
  shows "(n\<ra>m) zmod p = ((n zmod p)\<ra>m) zmod p"
proof-
  have "(n \<ra> m) zmod p = (n $+ m)zmod p" using Int_ZF_1_L2(1)[OF assms(1,2)]
    by auto
  also have "… = ((n zmod p) $+ m) zmod p" using zmod_zadd_left_eq[of n m p].
  also have "… = ((n zmod p) \<ra> m) zmod p" using Int_ZF_1_L2(1)[OF _ assms(2)]
    by auto
  ultimately show ?thesis by auto
qed

subsection‹Properties of three as a modulus›

text‹Three is a positive integer.›

lemma (in int0) three_pos: shows "𝟯 ∈ ℤ⇩+"
  using int_two_three_are_int int_one_two_are_pos Int_ZF_2_L12B Int_ZF_1_5_L7
  by auto

text‹Three satisfies the positivity preconditions needed for ‹zmod› and ‹zdiv›.›

lemma (in int0) three_zmod_prec: shows "𝟬 \<lsq> 𝟯"  "𝟯 ≠ 𝟬"
  using three_pos Int_ZF_1_5_L3 Int_ZF_2_L16C by auto

text‹For any integer $m$, $m \bmod 3$ is either $0$, $1$, or $2$.›

lemma (in int0) zmod_three_cases:
  assumes "m ∈ ℤ"
  shows "m zmod 𝟯 = 𝟬 ∨ m zmod 𝟯 = 𝟭 ∨ m zmod 𝟯 = 𝟮"
proof -
  let ?r = "m zmod 𝟯"
  have T: "𝟬 ∈ ℤ" "𝟭 ∈ ℤ" "𝟮 ∈ ℤ" "𝟯 ∈ ℤ"
    using int_zero_one_are_int int_two_three_are_int by auto
  have three_rs_one: "𝟯\<rs>𝟭 = 𝟮"
    using T Int_ZF_1_T2 group0.inv_cancel_two by simp
  have rule:"⋀m n. m ∈ ℤ ⟹
    𝟬 \<lsq> n ⟹ n ≠ 𝟬 ⟹ 𝟬 \<lsq> m zmod n"
    and "⋀m n. m ∈ ℤ ⟹
    𝟬 \<lsq> n ⟹ n ≠ 𝟬 ⟹ m zmod n \<lsq> (n\<rs>𝟭)"
    using IntDiv_ZF_1_L2(1,4) by auto
  then have "⋀m. m:ℤ ⟹ 𝟬 \<lsq> m zmod 𝟯"
    and "⋀m n. m ∈ ℤ ⟹ m zmod 𝟯 \<lsq> (𝟯\<rs>𝟭)" using three_zmod_prec
    by auto
  then have bnd: "𝟬 \<lsq> ?r" "?r \<lsq> 𝟮"
    using assms three_rs_one by auto
  have rT: "?r ∈ ℤ"
    using bnd(1) Int_ZF_2_L1A by simp
  from rT T have "?r = 𝟬 ∨ (?r \<lsq> 𝟬\<rs>𝟭) ∨ (𝟬\<ra>𝟭 \<lsq> ?r)"
    using Int_ZF_1_3_L6B by auto
  then show ?thesis
  proof (elim disjE)
    assume "?r = 𝟬" then show ?thesis by simp
  next
    assume A: "?r \<lsq> 𝟬\<rs>𝟭"
    with bnd(1) have "𝟬 \<lsq> 𝟬\<rs>𝟭"
      by (rule Int_order_transitive)
    with T have False using Int_ZF_1_2_L3AA by simp
    then show ?thesis by simp
  next
    assume A: "𝟬\<ra>𝟭 \<lsq> ?r"
    with T have one_le: "𝟭 \<lsq> ?r"
      using Int_ZF_1_1_L4 by simp
    from rT T have "?r = 𝟭 ∨ (?r \<lsq> 𝟭\<rs>𝟭) ∨ (𝟭\<ra>𝟭 \<lsq> ?r)"
      using Int_ZF_1_3_L6B by auto
    then show ?thesis
    proof (elim disjE)
      assume "?r = 𝟭" then show ?thesis by simp
    next
      assume "?r \<lsq> 𝟭\<rs>𝟭"
      with one_le have "𝟭 \<lsq> 𝟭\<rs>𝟭"
        by (rule Int_order_transitive)
      with T have False using Int_ZF_1_2_L3AA by simp
      then show ?thesis by simp
    next
      assume "𝟭\<ra>𝟭 \<lsq> ?r"
      with bnd(2) have "?r = 𝟮"
        using Int_ZF_2_L3 by simp
      then show ?thesis by simp
    qed
  qed
qed

text‹If $m \bmod 3 = 0$ then $(m-1) \bmod 3 = 2$.›

corollary (in int0) zmod_0_minus_one:
  assumes "m ∈ ℤ"
  and "m zmod 𝟯 = 𝟬"
  shows "(m\<rs>𝟭) zmod 𝟯 = 𝟮"
proof -
  have "(m\<rs>𝟭) zmod 𝟯 = ((m zmod 𝟯)\<rs>𝟭) zmod 𝟯"
    using zmod_add_left[OF assms(1) int_neg_type[OF int_zero_one_are_int(2)] 
        int_two_three_are_int(2)] by auto
  also have "… = (𝟬\<rs>𝟭)zmod 𝟯" using assms(2) by auto
  also have "… = 𝟮"
  proof -
    have T: "𝟬∈ℤ" "𝟭∈ℤ" "𝟮∈ℤ" "𝟯∈ℤ"
      using int_zero_one_are_int int_two_three_are_int by auto
    have zr1: "𝟬\<rs>𝟭 ∈ ℤ"
      by (rule Int_ZF_1_1_L5(2)[OF T(1) T(2)])
    have S1: "(𝟬\<rs>𝟭)\<ra>𝟭 = 𝟬"
      using T Int_ZF_1_T2 group0.inv_cancel_two by simp
    have S2: "((𝟬\<rs>𝟭)\<ra>𝟭)\<ra>𝟮 = 𝟮"
      using S1 T(3) Int_ZF_1_1_L4(2) by simp
    have S3: "((𝟬\<rs>𝟭)\<ra>𝟭)\<ra>𝟮 = (𝟬\<rs>𝟭)\<ra>𝟯"
      using Int_ZF_1_1_L7(1)[OF zr1 T(2) T(3)] Int_ZF_1_1_L5(4)[OF T(2) T(3)] by simp
    have eq: "(𝟬\<rs>𝟭)\<ra>𝟯 = 𝟮" using S2 S3 by simp
    have conv: "((𝟬\<rs>𝟭)\<ra>𝟯) zmod 𝟯 = (𝟬\<rs>𝟭) zmod 𝟯"
      using Int_ZF_1_L2(1)[OF zr1 T(4)] zmod_zadd_self2 by simp
    have lt23: "𝟮 $< 𝟯"
    proof -
      have "𝟮 \<lsq> 𝟯" using T(3) Int_ZF_2_L12B by simp
      moreover have "𝟮 ≠ 𝟯" using Int_ZF_1_L14(1)[OF T(3)] by auto
      ultimately show ?thesis using Int_ZF_2_L9 by auto
    qed
    have le02: "𝟬 \<lsq> 𝟮"
    proof -
      have "𝟬 \<lsq> 𝟭" using T Int_ZF_2_L12B[of 𝟬] Int_ZF_1_1_L4 by simp
      moreover have "𝟭 \<lsq> 𝟮" using Int_ZF_2_L16B by simp
      ultimately show ?thesis using Int_order_transitive by auto
    qed
    have h1: "#0 $≤ 𝟮" using le02 Int_ZF_2_L1A Int_ZF_1_L8 by auto
    have two_mod: "𝟮 zmod 𝟯 = 𝟮"
      using zmod_pos_pos_trivial[OF h1 lt23] T(3) intify_ident by auto
    from eq conv two_mod show ?thesis by auto
  qed
  finally show ?thesis by auto
qed

text‹If $m \bmod 3 = 1$ then $(m-1) \bmod 3 = 0$.›

corollary (in int0) zmod_1_minus_one:
  assumes "m ∈ ℤ"
  and "m zmod 𝟯 = 𝟭"
  shows "(m\<rs>𝟭) zmod 𝟯 = 𝟬"
proof -
  have "(m\<rs>𝟭) zmod 𝟯 = ((m zmod 𝟯)\<rs>𝟭) zmod 𝟯"
    using zmod_add_left[OF assms(1) int_neg_type[OF int_zero_one_are_int(2)] 
        int_two_three_are_int(2)] by auto
  also have "… = (𝟭\<rs>𝟭)zmod 𝟯" using assms(2) by auto
  also have "… = 𝟬 zmod 𝟯" using Int_ZF_1_L10[OF int_zero_one_are_int(2) int_zero_one_are_int(2)]
    zadd_zminus_inverse[of 𝟭] Int_ZF_1_L8(1) by auto
  finally have "(m\<rs>𝟭) zmod 𝟯 = 𝟬 zmod 𝟯" by auto
  then show ?thesis using Int_ZF_1_L8(1) zmod_zero by auto
qed

text‹If $m \bmod 3 = 2$ then $(m-1) \bmod 3 = 1$.›

corollary (in int0) zmod_2_minus_one:
  assumes "m ∈ ℤ"
  and "m zmod 𝟯 = 𝟮"
  shows "(m\<rs>𝟭) zmod 𝟯 = 𝟭"
proof -
  have "(m\<rs>𝟭) zmod 𝟯 = ((m zmod 𝟯)\<rs>𝟭) zmod 𝟯"
    using zmod_add_left[OF assms(1) int_neg_type[OF int_zero_one_are_int(2)] 
        int_two_three_are_int(2)] by auto
  also have "… = (𝟮\<rs>𝟭)zmod 𝟯" using assms(2) by auto
  also have "… = 𝟭 zmod 𝟯"
    using int_zero_one_are_int int_two_three_are_int Int_ZF_1_T2 group0.inv_cancel_two by simp
  also have "… = 𝟭"
  proof -
    have T: "𝟬∈ℤ" "𝟭∈ℤ" "𝟮∈ℤ" "𝟯∈ℤ"
      using int_zero_one_are_int int_two_three_are_int by auto
    have h1: "#0 $≤ 𝟭"
    proof -
      have "𝟬 \<lsq> 𝟭" using T Int_ZF_2_L12B[of 𝟬] Int_ZF_1_1_L4 by simp
      then show ?thesis using Int_ZF_2_L1A Int_ZF_1_L8 by auto
    qed
    have h2: "𝟭 $< 𝟯"
    proof -
      have le12: "𝟭 \<lsq> 𝟮" using Int_ZF_2_L16B by simp
      moreover have "𝟭 ≠ 𝟮" using T(2) Int_ZF_1_L14(1)[of 𝟭] by auto
      ultimately have lt12: "𝟭 $< 𝟮" using Int_ZF_2_L9 by auto
      have "𝟮 \<lsq> 𝟯" using T(3) Int_ZF_2_L12B by simp
      then have le23: "𝟮 $≤ 𝟯" using Int_ZF_2_L1A by auto
      show ?thesis using lt12 le23 zless_zle_trans by auto
    qed
    show ?thesis using zmod_pos_pos_trivial[OF h1 h2] T(2) intify_ident by auto
  qed
  finally show ?thesis by auto
qed

text‹If $m \bmod 3 = 2$ then $(m+1) \bmod 3 = 0$.›

corollary (in int0) zmod_2_plus_one:
  assumes "m ∈ ℤ"
  and "m zmod 𝟯 = 𝟮"
  shows "(m \<ra> 𝟭) zmod 𝟯 = 𝟬"
proof -
  have "(m \<ra> 𝟭) zmod 𝟯 = ((m zmod 𝟯) \<ra> 𝟭) zmod 𝟯"
    using zmod_add_left[OF assms(1) Int_ZF_1_L8A(2) int_two_three_are_int(2)] by auto
  also have "… = (𝟮 \<ra> 𝟭) zmod 𝟯" using assms(2) by auto
  also have "… = 𝟯 zmod 𝟯" by simp
  also have "… = 𝟬"
  proof -
    have "𝟯 zmod 𝟯 = $#0" using zmod_self by simp
    then show ?thesis using Int_ZF_1_L8(1) by auto
  qed
  finally show ?thesis by auto
qed

text‹Adding three and dividing by three increments the quotient by one.›

corollary (in int0) zdiv_add_three:
  assumes "m ∈ ℤ"
  shows "(m \<ra> 𝟯) zdiv 𝟯 = m zdiv 𝟯 \<ra> 𝟭"
proof -
  have T3: "𝟯 ∈ ℤ" using int_two_three_are_int by auto
  have three_ne: "𝟯 ≠ $#0"
    using three_zmod_prec(2) Int_ZF_1_L8(1) by auto
  have add_eq: "m \<ra> 𝟯 = m $+ 𝟯"
    using Int_ZF_1_L2(1)[OF assms T3] by simp
  have div_eq: "(m $+ 𝟯) zdiv 𝟯 = m zdiv 𝟯 $+ $#1"
    using zdiv_zadd_self2 three_ne intify_ident T3 by simp
  have one_eq: "m zdiv 𝟯 \<ra> 𝟭 = m zdiv 𝟯 $+ $#1"
    using Int_ZF_1_L2(1)[OF _ Int_ZF_1_L8A(2)] zdiv_type
      Int_ZF_1_L8(2) by simp
  from add_eq div_eq one_eq show ?thesis by simp
qed

text‹Dividing $m+2$ by three and subtracting one gives the same result as
  dividing $m-1$ by three.›

corollary (in int0) zdiv_minus_one:
  assumes "m∈ℤ"
  shows "((m\<ra>𝟮) zdiv 𝟯) \<rs>𝟭 = (m\<rs>𝟭) zdiv 𝟯"
proof -
  have T1: "m\<rs>𝟭 ∈ ℤ" using assms int_zero_one_are_int Int_ZF_1_1_L5 by auto
  have arith: "m\<rs>𝟭\<ra>𝟯 = m\<ra>𝟮"
  proof -
    have h1: "m\<rs>𝟭\<ra>𝟭 = m"
      using assms int_zero_one_are_int
        group0.inv_cancel_two(1)[OF Int_ZF_1_T2(3)] by simp
    have h2: "m\<rs>𝟭\<ra>𝟭\<ra>𝟮 = m\<rs>𝟭\<ra>𝟯"
      using Int_ZF_1_2_L11(1)[OF T1] by simp
    show ?thesis using h1 h2[symmetric] by simp
  qed
  have main: "(m\<ra>𝟮) zdiv 𝟯 = (m\<rs>𝟭) zdiv 𝟯 \<ra> 𝟭"
    using arith zdiv_add_three[OF T1] by simp
  show ?thesis
    using main zdiv_type int_zero_one_are_int
      group0.inv_cancel_two(2)[OF Int_ZF_1_T2(3)] by simp
qed
