locale commutative_ring = ring0 +
  assumes commutative:"M{is commutative on}R" 

lemma (in commutative_ring) mult_by_elem:
  assumes "x‚ààR"
  shows "{x‚ãÖy. y‚ààR}‚óÉR"
proof(rule ideal_intro)
  have "x‚ãÖùü≠‚àà{x‚ãÖy. y‚ààR}" using Ring_ZF_1_L2(2) by auto
  then show "{x‚ãÖy. y‚ààR}‚â†0" by auto
  {
    fix t assume t:"t‚àà{x ‚ãÖ y . y ‚àà R}"
    then obtain y where y:"y‚ààR" "t=x‚ãÖy" by auto
    then have "t‚ààR" using Ring_ZF_1_L4(3) assms by auto
  }
  then show "{x ‚ãÖ y . y ‚àà R} ‚äÜ R" by auto
  {
    fix t h assume ty:"t‚àà{x ‚ãÖ y . y ‚àà R}" "h‚ààR"
    from ty(1) obtain y where y:"y:R" "t=x‚ãÖy" by auto
    from y(2) have "t‚ãÖh=(x‚ãÖy)‚ãÖh" by auto
    then have "t‚ãÖh= x‚ãÖ(y‚ãÖh)" using Ring_ZF_1_L11(2)
      assms y(1) ty(2) by auto
    moreover have "y‚ãÖh‚ààR" using y(1) ty(2) using Ring_ZF_1_L4(3) by auto
    ultimately have E1:"t‚ãÖh‚àà{x ‚ãÖ y . y ‚àà R}" by auto
  }
  then show "‚àÄt‚àà{x ‚ãÖ y . y ‚àà R}. ‚àÄh‚ààR. t ‚ãÖ h ‚àà {x ‚ãÖ y . y ‚àà R}" by auto
  {
    fix t h assume ty:"t‚àà{x ‚ãÖ y . y ‚àà R}" "h‚ààR"
    from ty(1) obtain y where y:"y:R" "t=x‚ãÖy" by auto
    from y(2) have "h‚ãÖt = h‚ãÖ(x‚ãÖy)" by auto
    then have "h‚ãÖt = h‚ãÖ(y‚ãÖx)" using commutative
      unfolding IsCommutative_def using assms y(1) by auto
    then have "h‚ãÖt = (h‚ãÖy)‚ãÖx" using Ring_ZF_1_L11(2)
      assms y(1) ty(2) by auto
    moreover have "h‚ãÖy‚ààR" using y(1) ty(2) using Ring_ZF_1_L4(3) by auto
    ultimately have "h‚ãÖt = x‚ãÖ(h‚ãÖy)" "h‚ãÖy‚ààR" using commutative
      unfolding IsCommutative_def using assms by auto
    then have "h‚ãÖt‚àà{x ‚ãÖ y . y ‚àà R}" by auto
  }
  then show " ‚àÄt‚àà{x ‚ãÖ y . y ‚àà R}. ‚àÄh‚ààR. h ‚ãÖ t ‚àà {x ‚ãÖ y . y ‚àà R}" by auto
  {
    fix t h assume th:"t:{x ‚ãÖ y . y ‚àà R}" "h:{x ‚ãÖ y . y ‚àà R}"
    then obtain yt yh where y:"t=x‚ãÖyt" "yt:R" "h=x‚ãÖyh" "yh:R" by auto
    then have "t\<ra>h = (x‚ãÖyt)\<ra>(x‚ãÖyh)" by auto
    then have "t\<ra>h = x‚ãÖ(yt\<ra>yh)" using ring_oper_distr(1) assms
      y(2,4) by auto
    moreover have "yt\<ra>yh :R" using y(2,4) Ring_ZF_1_L4(1) by auto
    ultimately have "t\<ra>h:{x ‚ãÖ y . y ‚àà R}" by auto
  }
  then show "‚àÄxa‚àà{x ‚ãÖ y . y ‚àà R}. ‚àÄy‚àà{x ‚ãÖ y . y ‚àà R}. xa \<ra> y ‚àà {x ‚ãÖ y . y ‚àà R}" by auto
qed

theorem (in commutative_ring) principal_ideal:
  assumes "x‚ààR"
  shows "‚ü®{x}‚ü©‚á©I = {x ‚ãÖ y . y ‚àà R}"
proof
  have "x‚ãÖùü≠‚àà{x‚ãÖy. y‚ààR}" using Ring_ZF_1_L2(2) by auto
  then have xx:"x‚àà{x‚ãÖy. y‚ààR}" using Ring_ZF_1_L3(5) assms by auto
  then show "‚ü®{x}‚ü©‚á©I ‚äÜ {x ‚ãÖ y . y ‚àà R}"
    using generated_ideal_small mult_by_elem assms by auto
  {
    fix J assume j:"{x} ‚äÜ J" "J‚óÉR" "J‚ààPow(R)"
    {
      fix t assume t:"t‚àà{x ‚ãÖ y . y ‚àà R}"
      then obtain y where y:"y:R" "t=x‚ãÖy" by auto
      with j have "t‚ààJ" using ideal_dest_mult(1) by auto
    }
    then have "{x ‚ãÖ y . y ‚àà R} ‚äÜ J" by auto
  } moreover
  have "{x ‚ãÖ y . y ‚àà R}‚àà{J‚ààPow(R). J ‚óÉR ‚àß {x}‚äÜJ}"
    using assms Ring_ZF_1_L4(3) xx mult_by_elem by auto
  then have "{J‚ààPow(R). J ‚óÉR ‚àß {x}‚äÜJ}‚â†0" by auto
  ultimately have "{x ‚ãÖ y . y ‚àà R} ‚äÜ ‚ãÇ{J‚ààPow(R). J ‚óÉR ‚àß {x}‚äÜJ}" by auto
  then show "{x ‚ãÖ y . y ‚àà R} ‚äÜ ‚ü®{x}‚ü©‚á©I"
    using generatedIdeal_def assms by auto
qed

text ‚ÄπCommutative prime rings are the same as 
commutative ring with no zero divisors.‚Ä∫

lemma (in commutative_ring) prime_ring_zero_divs_1:
  assumes "[R,A,M]{is a prime ring}"
  shows "HasNoZeroDivs(R,A,M)" unfolding HasNoZeroDivs_def
proof(safe)
  fix aa b assume as:"aa‚ààR" "b‚ààR" "M ` ‚ü®aa, b‚ü© =TheNeutralElement(R,A)" "b‚â†TheNeutralElement(R,A)"
  with assms have "(‚àÄz‚ààR. M ` ‚ü®M ` ‚ü®aa, z‚ü©, b‚ü© = ùü¨) ‚ü∂
                 (aa = ùü¨ ‚à® b = ùü¨)" unfolding primeRing_def[OF ringAssum] by auto
  moreover
  {
    fix z assume z:"z‚ààR"
    have "(aa‚ãÖz)‚ãÖb = (z‚ãÖaa)‚ãÖb" using as(1) z(1)
      commutative unfolding IsCommutative_def by auto
    then have "(aa‚ãÖz)‚ãÖb = z‚ãÖ(aa‚ãÖb)" using Ring_ZF_1_L11(2)
      as z by auto
    then have "(aa‚ãÖz)‚ãÖb = z‚ãÖùü¨" using as(3) by auto
    then have "(aa‚ãÖz)‚ãÖb = ùü¨" using Ring_ZF_1_L6(2) z by auto
  }
  ultimately have "aa = ùü¨ ‚à® b = ùü¨" by auto
  with as(4) show "aa=TheNeutralElement(R,A)" by auto
qed

lemma (in commutative_ring) prime_ring_zero_divs_2:
  assumes "HasNoZeroDivs(R,A,M)"
  shows "[R,A,M]{is a prime ring}" unfolding primeRing_def[OF ringAssum]
proof(safe)
  fix aa b assume as:"aa‚ààR" "b‚ààR" "‚àÄz‚ààR. M ` ‚ü®M ` ‚ü®aa, z‚ü©, b‚ü© = TheNeutralElement(R, A)" 
    "b‚â†TheNeutralElement(R,A)"
  with assms have "‚àÄc‚ààR. ‚àÄb‚ààR. M ` ‚ü®c, b‚ü© = ùü¨ ‚ü∂
                 c = ùü¨ ‚à® b = ùü¨" unfolding HasNoZeroDivs_def by auto
  with as(1,2) have "M ` ‚ü®aa, b‚ü© = ùü¨ ‚ü∂ aa = ùü¨ ‚à® b = ùü¨" by auto
  with as(4) have "M ` ‚ü®aa, b‚ü© = ùü¨ ‚ü∂ aa = ùü¨" by auto
  moreover
  from as(3) have "M`‚ü®M ` ‚ü®aa, ùü≠‚ü©, b‚ü© = ùü¨" using Ring_ZF_1_L2(2) by auto
  then have "aa‚ãÖb = ùü¨" using Ring_ZF_1_L3(5) as(1) by auto
  ultimately show "aa = TheNeutralElement(R, A)" by auto
qed

       ‚àÄy‚ààQuotientBy(I).

              ProjFun2(R, QuotientGroupRel(R, A, I), M) `

          x = TheNeutralElement(QuotientBy(I), QuotientGroupOp(R, A, I)) ‚à®
          y = TheNeutralElement(QuotientBy(I), QuotientGroupOp(R, A, I))) ‚ü∂
[QuotientBy
      (I),QuotientGroupOp
           (R, A, I),ProjFun2(R, QuotientGroupRel(R, A, I), M)]{is a prime ring}"

  have "‚àÄx‚ààQuotientBy(I).
       ‚àÄy‚ààQuotientBy(I).

              ProjFun2(R, QuotientGroupRel(R, A, I), M) `

              TheNeutralElement(QuotientBy(I), QuotientGroupOp(R, A, I))) ‚ü∂
          x = TheNeutralElement(QuotientBy(I), QuotientGroupOp(R, A, I)) ‚à®
          y = TheNeutralElement(QuotientBy(I), QuotientGroupOp(R, A, I))"
  proof(safe)
    fix x y assume as:"x‚ààQuotientBy(I)" "y‚ààQuotientBy(I)" "‚àÄz‚ààQuotientBy(I).
              ProjFun2(R, QuotientGroupRel(R, A, I), M) `
              ‚ü®ProjFun2(R, QuotientGroupRel(R, A, I), M) ` ‚ü®x, z‚ü©, y‚ü© =
              TheNeutralElement(QuotientBy(I), QuotientGroupOp(R, A, I))"

        fix yy assume yy:"yy‚ààR" "QuotientGroupRel(R, A, I)``{yy} = y"

          then have "QuotientGroupRel(R, A, I)``{xx‚ãÖzz‚ãÖyy} = 
            ProjFun2(R, QuotientGroupRel(R, A, I), M)`‚ü®ProjFun2(R, QuotientGroupRel(R, A, I), M)`‚ü®QuotientGroupRel(R, A, I)``{xx},QuotientGroupRel(R, A, I)``{zz}‚ü©,QuotientGroupRel(R, A, I)``{yy}‚ü©"

          with xx(2) yy(2) have "QuotientGroupRel(R, A, I)``{xx‚ãÖzz‚ãÖyy} =

          xx(1) yy(1) by auto

          then have False using add_group.Group_ZF_2_4_L5E[OF ideal_normal_add_subgroup[OF ideal], of yy "QuotientGroupRel(R,A,I)"] yy as(4)

      }
      with quotientE[of y R "QuotientGroupRel(R,A,I)" "xx‚ààI"] have "xx‚ààI" using as(2) unfolding QuotientBy_def[OF ideal] by auto
      then have "QuotientGroupRel(R, A, I)``{xx} = TheNeutralElement(QuotientBy(I), QuotientGroupOp(R, A, I))"
        using add_group.Group_ZF_2_4_L5E[OF ideal_normal_add_subgroup[OF ideal], of xx "QuotientGroupRel(R,A,I)"] 
        unfolding QuotientBy_def[OF ideal] using xx(1) by auto
      with xx(2) have "x = TheNeutralElement(QuotientBy(I), QuotientGroupOp(R, A, I))" by auto
    }
    with quotientE[of x R "QuotientGroupRel(R,A,I)" "x = TheNeutralElement(QuotientBy(I), QuotientGroupOp(R, A, I))"] 
    show "x = TheNeutralElement(QuotientBy(I), QuotientGroupOp(R, A, I))" using as(1) unfolding QuotientBy_def[OF ideal] by auto
  qed
  ultimately show ?thesis by auto
qed
