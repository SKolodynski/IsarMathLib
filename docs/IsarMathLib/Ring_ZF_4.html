section ‹Rings - Commutative Rings›

theory Ring_ZF_4 imports Ring_ZF_2 CommutativeSemigroup_ZF

begin

text‹This theory develops the theory of commutative rings. We define the
  ‹commutative_ring› locale, characterize principal ideals, and study the
  relationship between prime rings, prime ideals, and the absence of zero divisors.›

subsection‹Commutative rings›

text‹A commutative ring is a ring in which multiplication satisfies the
  commutativity law. We establish that principal ideals take a simple form
  in this setting.›

text‹A commutative ring is defined as a ‹ring0› in which multiplication
  is commutative.›

locale commutative_ring = ring0 +
  assumes commutative:"M{is commutative on}R"

text‹For any element $x$ of a commutative ring, the set $\{x \cdot y \mid y \in R\}$
  is an ideal of the ring.›

lemma (in commutative_ring) mult_by_elem:
  assumes "x∈R"
  shows "{x⋅y. y∈R}◃R"
proof-
  have "x⋅𝟭∈{x⋅y. y∈R}" using Ring_ZF_1_L2(2) by auto
  then have I:"{x⋅y. y∈R}≠0" by auto
  {
    fix t assume t:"t∈{x ⋅ y . y ∈ R}"
    then obtain y where y:"y∈R" "t=x⋅y" by auto
    then have "t∈R" using Ring_ZF_1_L4(3) assms by auto
  }
  then have II:"{x ⋅ y . y ∈ R} ⊆ R" by auto
  {
    fix t h assume ty:"t∈{x ⋅ y . y ∈ R}" "h∈R"
    from ty(1) obtain y where y:"y:R" "t=x⋅y" by auto
    from y(2) have "t⋅h=(x⋅y)⋅h" by auto
    then have "t⋅h= x⋅(y⋅h)" using Ring_ZF_1_L11(2)
      assms y(1) ty(2) by auto
    moreover have "y⋅h∈R" using y(1) ty(2) using Ring_ZF_1_L4(3) by auto
    ultimately have E1:"t⋅h∈{x ⋅ y . y ∈ R}" by auto
  }
  then have III: "∀t∈{x ⋅ y . y ∈ R}. ∀h∈R. t ⋅ h ∈ {x ⋅ y . y ∈ R}" by auto
  {
    fix t h assume ty:"t∈{x ⋅ y . y ∈ R}" "h∈R"
    from ty(1) obtain y where y:"y:R" "t=x⋅y" by auto
    from y(2) have "h⋅t = h⋅(x⋅y)" by auto
    then have "h⋅t = h⋅(y⋅x)" using commutative assms y(1)
      unfolding IsCommutative_def by auto
    then have "h⋅t = (h⋅y)⋅x" using Ring_ZF_1_L11(2)
      assms y(1) ty(2) by auto
    moreover have "h⋅y∈R" using y(1) ty(2) using Ring_ZF_1_L4(3) by auto
    ultimately have "h⋅t = x⋅(h⋅y)" "h⋅y∈R" using commutative assms
      unfolding IsCommutative_def by auto
    then have "h⋅t∈{x ⋅ y . y ∈ R}" by auto
  }
  then have IV: " ∀t∈{x ⋅ y . y ∈ R}. ∀h∈R. h ⋅ t ∈ {x ⋅ y . y ∈ R}" by auto
  {
    fix t h assume th:"t:{x ⋅ y . y ∈ R}" "h:{x ⋅ y . y ∈ R}"
    then obtain yt yh where y:"t=x⋅yt" "yt:R" "h=x⋅yh" "yh:R" by auto
    then have "t\<ra>h = (x⋅yt)\<ra>(x⋅yh)" by auto
    then have "t\<ra>h = x⋅(yt\<ra>yh)" using ring_oper_distr(1) assms
      y(2,4) by auto
    moreover have "yt\<ra>yh :R" using y(2,4) Ring_ZF_1_L4(1) by auto
    ultimately have "t\<ra>h:{x ⋅ y . y ∈ R}" by auto
  }
  then have V: "∀xa∈{x ⋅ y . y ∈ R}. ∀y∈{x ⋅ y . y ∈ R}. xa \<ra> y ∈ {x ⋅ y . y ∈ R}" by auto
  from I II III IV V show ?thesis using ideal_intro by auto
qed

text‹In a commutative ring the principal ideal generated by an element $x$
  equals $\{x \cdot y \mid y \in R\}$.›

theorem (in commutative_ring) principal_ideal:
  assumes "x∈R"
  shows "⟨{x}⟩⇩I = {x ⋅ y . y ∈ R}"
proof
  have "x⋅𝟭∈{x⋅y. y∈R}" using Ring_ZF_1_L2(2) by auto
  then have xx:"x∈{x⋅y. y∈R}" using Ring_ZF_1_L3(5) assms by auto
  then show "⟨{x}⟩⇩I ⊆ {x ⋅ y . y ∈ R}"
    using generated_ideal_small mult_by_elem assms by auto
  {
    fix J assume j:"{x} ⊆ J" "J◃R" "J∈Pow(R)"
    {
      fix t assume t:"t∈{x ⋅ y . y ∈ R}"
      then obtain y where y:"y:R" "t=x⋅y" by auto
      with j have "t∈J" using ideal_dest_mult(1) by auto
    }
    then have "{x ⋅ y . y ∈ R} ⊆ J" by auto
  } moreover
  have "{x ⋅ y . y ∈ R}∈{J∈Pow(R). J ◃R ∧ {x}⊆J}"
    using assms Ring_ZF_1_L4(3) xx mult_by_elem by auto
  then have "{J∈Pow(R). J ◃R ∧ {x}⊆J}≠0" by auto
  ultimately have "{x ⋅ y . y ∈ R} ⊆ ⋂{J∈Pow(R). J ◃R ∧ {x}⊆J}" by auto
  then show "{x ⋅ y . y ∈ R} ⊆ ⟨{x}⟩⇩I"
    using generatedIdeal_def assms by auto
qed

subsection‹Prime ideals›

text‹We characterize prime rings in terms of zero divisors and show that
  the quotient of a ring by a prime ideal is a prime ring, or equivalently,
  has no zero divisors in the commutative case.›

text‹Commutative prime rings are the same as
commutative rings with no zero divisors.›

lemma (in commutative_ring) prime_ring_zero_divs_1:
  assumes "[R,A,M]{is a prime ring}"
  shows "HasNoZeroDivs(R,A,M)"
proof -
  {
    fix aa b assume as:"aa∈R" "b∈R" "M ` ⟨aa, b⟩ =TheNeutralElement(R,A)" "b≠TheNeutralElement(R,A)"
    with assms have "(∀z∈R. M ` ⟨M ` ⟨aa, z⟩, b⟩ = 𝟬) ⟶
                   (aa = 𝟬 ∨ b = 𝟬)" using primeRing_def ringAssum by auto
    moreover
    {
      fix z assume z:"z∈R"
      have "(aa⋅z)⋅b = (z⋅aa)⋅b" using as(1) z(1)
        commutative unfolding IsCommutative_def by auto
      then have "(aa⋅z)⋅b = z⋅(aa⋅b)" using Ring_ZF_1_L11(2)
        as z by auto
      then have "(aa⋅z)⋅b = z⋅𝟬" using as(3) by auto
      then have "(aa⋅z)⋅b = 𝟬" using Ring_ZF_1_L6(2) z by auto
    }
    ultimately have "aa = 𝟬 ∨ b = 𝟬" by auto
    with as(4) have "aa=TheNeutralElement(R,A)" by auto
  }
  then show ?thesis unfolding HasNoZeroDivs_def by auto
qed

text‹A commutative ring with no zero divisors is a prime ring.›

lemma (in commutative_ring) prime_ring_zero_divs_2:
  assumes "HasNoZeroDivs(R,A,M)"
  shows "[R,A,M]{is a prime ring}"
proof -
  {
    fix aa b assume as:"aa∈R" "b∈R" "∀z∈R. M ` ⟨M ` ⟨aa, z⟩, b⟩ = TheNeutralElement(R, A)"
      "b≠TheNeutralElement(R,A)"
    with assms have "∀c∈R. ∀b∈R. M ` ⟨c, b⟩ = 𝟬 ⟶
                   c = 𝟬 ∨ b = 𝟬" unfolding HasNoZeroDivs_def by auto
    with as(1,2) have "M ` ⟨aa, b⟩ = 𝟬 ⟶ aa = 𝟬 ∨ b = 𝟬" by auto
    with as(4) have "M ` ⟨aa, b⟩ = 𝟬 ⟶ aa = 𝟬" by auto
    moreover
    from as(3) have "M`⟨M ` ⟨aa, 𝟭⟩, b⟩ = 𝟬" using Ring_ZF_1_L2(2) by auto
    then have "aa⋅b = 𝟬" using Ring_ZF_1_L3(5) as(1) by auto
    ultimately have "aa = TheNeutralElement(R, A)" by auto
  }
  then show ?thesis using primeRing_def ringAssum by auto
qed

text‹The quotient ring of a ring by a prime ideal is a prime ring.›

theorem (in ring0) prime_ideal_no_zero_divs:
  assumes "I◃⇩pR"
  shows "[QuotientBy(I),QuotientGroupOp(R, A, I),ProjFun2(R, QuotientGroupRel(R, A, I), M)]{is a prime ring}"
proof-
  have ideal:"I◃R" using assms unfolding primeIdeal_def by auto
  from primeRing_def quotientBy_is_ring assms have "
    (∀x∈QuotientBy(I).
       ∀y∈QuotientBy(I).
          (∀z∈QuotientBy(I).
              ProjFun2(R, QuotientGroupRel(R, A, I), M) `
              ⟨ProjFun2(R, QuotientGroupRel(R, A, I), M) ` ⟨x, z⟩, y⟩ =
              TheNeutralElement(QuotientBy(I), QuotientGroupOp(R, A, I))) ⟶
          x = TheNeutralElement(QuotientBy(I), QuotientGroupOp(R, A, I)) ∨
          y = TheNeutralElement(QuotientBy(I), QuotientGroupOp(R, A, I))) ⟶
[QuotientBy
      (I),QuotientGroupOp
           (R, A, I),ProjFun2(R, QuotientGroupRel(R, A, I), M)]{is a prime ring}" 
    unfolding primeIdeal_def by auto moreover
  have "∀x∈QuotientBy(I).
       ∀y∈QuotientBy(I).
          (∀z∈QuotientBy(I).
              ProjFun2(R, QuotientGroupRel(R, A, I), M) `
              ⟨ProjFun2(R, QuotientGroupRel(R, A, I), M) ` ⟨x, z⟩, y⟩ =
              TheNeutralElement(QuotientBy(I), QuotientGroupOp(R, A, I))) ⟶
          x = TheNeutralElement(QuotientBy(I), QuotientGroupOp(R, A, I)) ∨
          y = TheNeutralElement(QuotientBy(I), QuotientGroupOp(R, A, I))"
  proof -
    {
    fix x y assume as:"x∈QuotientBy(I)" "y∈QuotientBy(I)" "∀z∈QuotientBy(I).
              ProjFun2(R, QuotientGroupRel(R, A, I), M) `
              ⟨ProjFun2(R, QuotientGroupRel(R, A, I), M) ` ⟨x, z⟩, y⟩ =
              TheNeutralElement(QuotientBy(I), QuotientGroupOp(R, A, I))"
      "y ≠ TheNeutralElement(QuotientBy(I), QuotientGroupOp(R, A, I))"
    {
      fix xx assume xx:"xx∈R" "QuotientGroupRel(R, A, I)``{xx} = x"
      {
        fix yy assume yy:"yy∈R" "QuotientGroupRel(R, A, I)``{yy} = y"
        {
          fix zz assume zz:"zz∈R"
          from ideal have equiv:"equiv(R, QuotientGroupRel(R, A, I))" and
            cong2:"Congruent2(QuotientGroupRel(R, A, I), M)" using ideal_equiv_rel quotientBy_mul_monoid(1) by auto
          have "⋀x y r f. equiv(R,r) ⟹ Congruent2(r,f) ⟹ x∈R ⟹ y∈R ⟹ ProjFun2(R, r, f) `
              ⟨r `` {x}, r `` {y}⟩ =
                r ``{f ` ⟨x, y⟩}" using EquivClass_1_L10 by auto
          then have "⋀x y. equiv(R,  QuotientGroupRel(R, A, I)) ⟹ Congruent2( QuotientGroupRel(R, A, I),M) ⟹ x∈R ⟹ y∈R
            ⟹ ProjFun2(R, QuotientGroupRel(R, A, I), M) `
              ⟨QuotientGroupRel(R, A, I) `` {x},  QuotientGroupRel(R, A, I) `` {y}⟩ =
                QuotientGroupRel(R, A, I) ``{M ` ⟨x, y⟩}" by auto
          with equiv cong2
          have eq:"⋀x y. x∈R ⟹ y∈R ⟹ ProjFun2(R, QuotientGroupRel(R, A, I), M) `
              ⟨QuotientGroupRel(R, A, I) `` {x},  QuotientGroupRel(R, A, I) `` {y}⟩ =
                QuotientGroupRel(R, A, I) ``{M ` ⟨x, y⟩}" by auto
          have "QuotientGroupRel(R, A, I)``{xx⋅zz⋅yy} =
            ProjFun2(R, QuotientGroupRel(R, A, I), M)`⟨QuotientGroupRel(R, A, I)``{xx⋅zz},QuotientGroupRel(R, A, I)``{yy}⟩"
            using eq xx(1) yy(1) zz Ring_ZF_1_L4(3) by auto
          then have "QuotientGroupRel(R, A, I)``{xx⋅zz⋅yy} = 
            ProjFun2(R, QuotientGroupRel(R, A, I), M)`⟨ProjFun2(R, QuotientGroupRel(R, A, I), M)`⟨QuotientGroupRel(R, A, I)``{xx},QuotientGroupRel(R, A, I)``{zz}⟩,QuotientGroupRel(R, A, I)``{yy}⟩"
            using eq xx(1) zz by auto
          with xx(2) yy(2) have "QuotientGroupRel(R, A, I)``{xx⋅zz⋅yy} = 
            ProjFun2(R, QuotientGroupRel(R, A, I), M)`⟨ProjFun2(R, QuotientGroupRel(R, A, I), M)`⟨x,QuotientGroupRel(R, A, I)``{zz}⟩,y⟩"
            by auto
          moreover have "QuotientGroupRel(R, A, I)``{zz}∈QuotientBy(I)" using QuotientBy_def ideal zz
            unfolding quotient_def by auto
          moreover note as(3) ultimately
          have "QuotientGroupRel(R, A, I)``{xx⋅zz⋅yy} = TheNeutralElement(QuotientBy(I), QuotientGroupOp(R, A, I))"
            by auto
          then have "xx⋅zz⋅yy∈I" using add_group.Group_ZF_2_4_L5E ideal_normal_add_subgroup
            xx(1) yy(1) zz Ring_ZF_1_L4(3) QuotientBy_def ideal by auto
        }
        then have "∀zz∈R. xx⋅zz⋅yy∈I" by auto
        then have D:"xx:I∨yy:I" using assms equivalent_prime_ideal
          xx(1) yy(1) by auto
        {
          assume "yy∈I"
          then have False using add_group.Group_ZF_2_4_L5E ideal_normal_add_subgroup yy as(4)
            QuotientBy_def ideal by auto
        }
        with D have "xx∈I" by auto
      } moreover
      have "⋀P. y ∈ R // QuotientGroupRel(R, A, I) ⟹
        (⋀x. y = QuotientGroupRel(R, A, I) `` {x} ⟹ x ∈ R ⟹ P) ⟹ P" by (rule quotientE)
      then have " y ∈ R // QuotientGroupRel(R, A, I) ⟹
        (⋀x. y = QuotientGroupRel(R, A, I) `` {x} ⟹ x ∈ R ⟹ xx∈I) ⟹ xx∈I" by (rule quotientE)
      ultimately have "xx∈I" using as(2) QuotientBy_def ideal by auto
      then have "QuotientGroupRel(R, A, I)``{xx} = TheNeutralElement(QuotientBy(I), QuotientGroupOp(R, A, I))"
        using add_group.Group_ZF_2_4_L5E ideal_normal_add_subgroup
        QuotientBy_def ideal xx(1) by auto
      with xx(2) have "x = TheNeutralElement(QuotientBy(I), QuotientGroupOp(R, A, I))" by auto
    } moreover
    have "⋀P. x ∈ R // QuotientGroupRel(R, A, I) ⟹
        (⋀t. x = QuotientGroupRel(R, A, I) `` {t} ⟹ t ∈ R ⟹ P) ⟹ P" by (rule quotientE)
    then have "x ∈ R // QuotientGroupRel(R, A, I) ⟹
        (⋀t. x = QuotientGroupRel(R, A, I) `` {t} ⟹ t ∈ R ⟹ x = TheNeutralElement(QuotientBy(I), QuotientGroupOp(R, A, I)))
         ⟹ x = TheNeutralElement(QuotientBy(I), QuotientGroupOp(R, A, I))" by (rule quotientE)
    ultimately
    have "x = TheNeutralElement(QuotientBy(I), QuotientGroupOp(R, A, I))" using as(1) QuotientBy_def ideal by auto
    }
    then show ?thesis by auto
  qed
  ultimately show ?thesis by auto
qed

text‹In a commutative ring, the quotient by a prime ideal has no zero divisors.
  This is a specialization of ‹prime_ideal_no_zero_divs› using commutativity
  to convert the prime ring condition directly into the no-zero-divisors property.›

theorem (in commutative_ring) comm_prime_ideal_no_zero_divs:
  assumes "I◃⇩pR"
  shows "HasNoZeroDivs(QuotientBy(I), QuotientGroupOp(R,A,I), ProjFun2(R, QuotientGroupRel(R,A,I), M))"
proof -
  have ideal:"I◃R" using assms unfolding primeIdeal_def by auto
  {
    fix xx yy assume as:
      "xx∈QuotientBy(I)" "yy∈QuotientBy(I)"
      "ProjFun2(R, QuotientGroupRel(R,A,I), M)`⟨xx,yy⟩ =
         TheNeutralElement(QuotientBy(I), QuotientGroupOp(R,A,I))"
      "yy ≠ TheNeutralElement(QuotientBy(I), QuotientGroupOp(R,A,I))"
    {
      fix xx0 assume xx0:"xx0∈R" "QuotientGroupRel(R,A,I)``{xx0} = xx"
      {
        fix yy0 assume yy0:"yy0∈R" "QuotientGroupRel(R,A,I)``{yy0} = yy"
        have "ProjFun2(R, QuotientGroupRel(R,A,I), M)`⟨xx,yy⟩ =
            QuotientGroupRel(R,A,I)``{xx0⋅yy0}"
          using EquivClass_1_L10 ideal_equiv_rel quotientBy_mul_monoid(1) ideal
            xx0 yy0 by auto
        with as(3) have prod_in_I:"xx0⋅yy0∈I"
          using add_group.Group_ZF_2_4_L5E ideal_normal_add_subgroup
            xx0(1) yy0(1) Ring_ZF_1_L4(3) QuotientBy_def ideal by auto
        {
          fix zz assume zz:"zz∈R"
          have "xx0⋅zz⋅yy0 = (xx0⋅zz)⋅yy0" by auto
          moreover have "(xx0⋅zz)⋅yy0 = yy0⋅(xx0⋅zz)"
            using commutative xx0(1) yy0(1) zz Ring_ZF_1_L4(3)
            unfolding IsCommutative_def by auto
          moreover have "yy0⋅(xx0⋅zz) = (yy0⋅xx0)⋅zz"
            using Ring_ZF_1_L11(2) yy0(1) xx0(1) zz by auto
          moreover have "(yy0⋅xx0)⋅zz = (xx0⋅yy0)⋅zz"
            using commutative xx0(1) yy0(1) unfolding IsCommutative_def by auto
          ultimately have "xx0⋅zz⋅yy0 = (xx0⋅yy0)⋅zz" by auto
          then have "xx0⋅zz⋅yy0∈I"
            using ideal_dest_mult(1) ideal prod_in_I zz by auto
        }
        then have "∀zz∈R. xx0⋅zz⋅yy0∈I" by auto
        then have D:"xx0∈I ∨ yy0∈I"
          using assms equivalent_prime_ideal xx0(1) yy0(1) by auto
        {
          assume "yy0∈I"
          then have False
            using add_group.Group_ZF_2_4_L5E ideal_normal_add_subgroup yy0 as(4)
              QuotientBy_def ideal by auto
        }
        with D have "xx0∈I" by auto
      } moreover
      have "⋀P. yy ∈ R // QuotientGroupRel(R, A, I) ⟹
          (⋀t. yy = QuotientGroupRel(R, A, I) `` {t} ⟹ t ∈ R ⟹ P) ⟹ P" by (rule quotientE)
      then have "yy ∈ R // QuotientGroupRel(R, A, I) ⟹
          (⋀t. yy = QuotientGroupRel(R, A, I) `` {t} ⟹ t ∈ R ⟹ xx0∈I)
           ⟹ xx0∈I" by (rule quotientE)
      ultimately have "xx0∈I" using as(2) QuotientBy_def ideal by auto
      then have "QuotientGroupRel(R,A,I)``{xx0} =
          TheNeutralElement(QuotientBy(I), QuotientGroupOp(R,A,I))"
        using add_group.Group_ZF_2_4_L5E ideal_normal_add_subgroup
        QuotientBy_def ideal xx0(1) by auto
      with xx0(2) have "xx = TheNeutralElement(QuotientBy(I), QuotientGroupOp(R,A,I))" by auto
    } moreover
    have "⋀P. xx ∈ R // QuotientGroupRel(R, A, I) ⟹
        (⋀t. xx = QuotientGroupRel(R, A, I) `` {t} ⟹ t ∈ R ⟹ P) ⟹ P" by (rule quotientE)
    then have "xx ∈ R // QuotientGroupRel(R, A, I) ⟹
        (⋀t. xx = QuotientGroupRel(R, A, I) `` {t} ⟹ t ∈ R ⟹ xx = TheNeutralElement(QuotientBy(I), QuotientGroupOp(R,A,I)))
         ⟹ xx = TheNeutralElement(QuotientBy(I), QuotientGroupOp(R,A,I))" by (rule quotientE)
    ultimately
    have "xx = TheNeutralElement(QuotientBy(I), QuotientGroupOp(R,A,I))"
      using as(1) QuotientBy_def ideal by auto
  }
  then show ?thesis unfolding HasNoZeroDivs_def by auto
qed
