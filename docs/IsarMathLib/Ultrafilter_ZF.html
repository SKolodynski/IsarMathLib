text‚ÄπThis theory deals with ultrafilters; which is a maximal filter.‚Ä∫

definition IsUltrafilter ("_{is an ultrafilter on}_" 80)
  where "ùîâ{is an ultrafilter on}X ‚â° (ùîâ{is a filter on}X) ‚àß (‚àÄùîä. (ùîä{is a filter on}X) ‚ü∂ (ùîâ ‚äÜ ùîä ‚ü∂ ùîä = ùîâ))"

text‚ÄπEvery set or its opposite is in an ultrafilter‚Ä∫

lemma set_ultrafilter:
  assumes "A‚ààPow(X)" "ùîâ{is an ultrafilter on}X"
  shows "A‚ààùîâ ‚à® (X-A)‚ààùîâ"
proof-
  from assms(2) have filter:"ùîâ{is a filter on}X" unfolding IsUltrafilter_def by auto
  {
    assume a0:"A‚â†0"
    {
      assume a:"A‚àâùîâ" "X-A‚àâùîâ"
      then obtain ùîä where g:"ùîä{is a filter on}X" "A ‚àà ùîä" "ùîâ ‚äÜ ùîä"
        using extend_filter[OF assms(1) _ a0, of ùîâ] filter by auto
      with assms(2) have "A‚ààùîâ" unfolding IsUltrafilter_def by auto
      with a have False by auto
    }
    then have "A‚ààùîâ ‚à® (X-A)‚ààùîâ" by auto
  } moreover
  {
    assume "A=0"
    then have "X-A=X" by auto
    then have "X-A‚ààùîâ" using filter unfolding IsFilter_def by auto
    then have "A‚ààùîâ ‚à® (X-A)‚ààùîâ" by auto
  } ultimately
  show "A‚ààùîâ ‚à® (X-A)‚ààùîâ" by blast
qed

text‚ÄπIt contains only one of them‚Ä∫

lemma only_set_or_opposite:
  assumes "A‚ààPow(X)" "ùîâ{is a filter on}X" "A‚ààùîâ"
  shows "(X-A)‚àâùîâ"
proof
  assume "X-A‚ààùîâ"
  with assms have "A‚à©(X-A)‚ààùîâ" using is_filter_def_split(4) by auto
  moreover have "A‚à©(X-A) = 0" by auto
  ultimately have "0‚ààùîâ" by auto
  then show False using assms(2) is_filter_def_split(1) by auto
qed

text‚ÄπIf a filter contains a set or its opposite, it is in an ultrafilter‚Ä∫

lemma set_ultrafilter_equiv:
  assumes "ùîâ{is a filter on}X" "‚àÄA‚ààPow(X). A‚ààùîâ ‚à® (X-A)‚ààùîâ"
  shows "ùîâ{is an ultrafilter on}X" unfolding IsUltrafilter_def
proof(safe)
  from assms(1) show "ùîâ{is a filter on}X".
  {
    fix ùîä assume g:"ùîä{is a filter on}X" "ùîâ ‚äÜ ùîä"
    {
      assume "ùîä‚â†ùîâ"
      with g(2) obtain h where h:"h:ùîä" "h‚àâùîâ" by auto
      from h(1) g(1) have "h‚ààPow(X)" "h‚â†0" unfolding IsFilter_def by auto
      with assms(2) h(2) have "X-h‚ààùîâ" by auto
      with g(2) have "X-h:ùîä" by auto
      with h(1) g(1) have "(X-h)‚à©h‚ààùîä" unfolding IsFilter_def by auto moreover
      have "(X-h)‚à©h=0" by auto ultimately
      have "0‚ààùîä" by auto
      with g(1) have False unfolding IsFilter_def by auto
    }
    then show "ùîä=ùîâ" by auto
  }
qed

text‚ÄπFilters of containing a point, are ultrafilters‚Ä∫

lemma include_point_ultrafilter:
  assumes "x‚ààX"
  shows "{A‚ààPow(X). x‚ààA}{is an ultrafilter on}X"
proof(rule set_ultrafilter_equiv)
  show "{A ‚àà Pow(X) . x ‚àà A} {is a filter on} X" unfolding IsFilter_def
    apply safe using assms apply simp by auto
  {
    fix A assume A:"A‚ààPow(X)"
    have "x:A ‚à® x‚àâA" by auto
    with assms have "x‚ààA ‚à® x‚ààX-A" by auto
    with A(1) have "A‚àà{A ‚àà Pow(X) . x ‚àà A} ‚à® X-A: {A ‚àà Pow(X) . x ‚àà A} " by auto
  }
  then show "‚àÄA‚ààPow(X). A ‚àà {A ‚àà Pow(X) . x ‚àà A} ‚à® X - A ‚àà {A ‚àà Pow(X) . x ‚àà A}" by auto
qed

text‚ÄπAn ultrafilter that has no singleton sets, does not have finite sets‚Ä∫

lemma ultrafilter_finite_set:
  assumes "ùîâ{is an ultrafilter on}X" "‚àÄx‚ààX. {x}‚àâùîâ" "A‚ààFinPow(X)"
  shows "A‚àâùîâ"
proof(rule FinPow_induct[of "Œªq. q‚àâùîâ"])
  from assms(1) show "0‚àâùîâ" unfolding IsUltrafilter_def IsFilter_def by auto
  show "A‚ààFinPow(X)" using assms(3).
  {
    fix B assume b:"B‚ààFinPow(X)" "B‚àâùîâ"
    {
      fix s assume s:"s‚ààX"
      {
        assume cont:"B‚à™{s}‚ààùîâ"
        {
          assume "B=0"
          then have "B‚à™{s} = {s}" by auto
          with s assms(2) cont have False by auto
        } moreover
        {
          assume "B‚â†0"
          with b have "X-B ‚ààùîâ" using assms(1) set_ultrafilter[of B X ùîâ]
            unfolding FinPow_def by auto moreover
          from assms(1,2) s have "X-{s}‚ààùîâ" using set_ultrafilter[of "{s}" X ùîâ]
            by auto
          ultimately have "(X-B)‚à©(X-{s}) ‚ààùîâ" using assms(1)
            unfolding IsUltrafilter_def IsFilter_def by auto moreover
          {
            fix q assume "q‚àà(X-B)‚à©(X-{s})"
            then have "q:X" "q‚àâB" "q‚àâ{s}" by auto
            then have "q‚ààX-(B‚à™{s})" by auto
          }
          then have "(X-B)‚à©(X-{s}) ‚äÜ X-(B‚à™{s})" by auto moreover
          have "X-(B‚à™{s})‚ààPow(X)" by auto
          ultimately have "X-(B‚à™{s}) ‚ààùîâ" using assms(1) unfolding IsFilter_def IsUltrafilter_def
            by blast
          with cont have "(X-(B‚à™{s}))‚à©(B‚à™{s}):  ùîâ" using assms(1)
            unfolding IsFilter_def IsUltrafilter_def by auto moreover
          {
            fix q assume "q‚àà(X-(B‚à™{s}))‚à©(B‚à™{s})"
            then have False by auto
          }
          then have "(X-(B‚à™{s}))‚à©(B‚à™{s}) = 0" by auto
          ultimately have "0:  ùîâ" by auto
          with assms(1) have False unfolding IsUltrafilter_def IsFilter_def by auto
        }
        ultimately have False by auto
      }
      then have "B‚à™{s}‚àâùîâ" by auto
    }
    then have "‚àÄq‚ààX. B ‚à™ {q} ‚àâ ùîâ" by auto
  }
  then show "‚àÄA‚ààFinPow(X). A ‚àâ ùîâ ‚ü∂ (‚àÄq‚ààX. A ‚à™ {q} ‚àâ ùîâ)" by auto
qed

text‚ÄπThe cofinite filter plays an important role in ultrafilters.‚Ä∫

corollary ultrafilter_contains_cofinite:
  assumes "ùîâ{is an ultrafilter on}X" "‚àÄx‚ààX. {x}‚àâùîâ"
  shows "(CoFinite(X))-{0} ‚äÜ ùîâ"
proof
  fix S assume "S‚àà(CoFinite(X))-{0}"
  then have s:"S ‚àà Pow(X)" "X - S ‚â∫ nat" "S‚â†0" unfolding CoCardinal_def Cofinite_def by auto
  then have "Finite(X-S)" using lesspoll_nat_is_Finite by auto
  then have "X-S‚ààFinPow(X)" unfolding FinPow_def by auto
  with ultrafilter_finite_set assms have "X-S‚àâùîâ" by auto
  with assms(1) show "S‚ààùîâ" using set_ultrafilter s(1,3) by auto
qed

text‚ÄπAn ultrafilter is free or principal.‚Ä∫

corollary ultrafilter_principal_or_free:
  assumes "ùîâ{is an ultrafilter on}X"
  shows "‚ãÇùîâ = 0 ‚à® (‚àÉx‚ààX. ‚ãÇùîâ = {x})"
proof(safe)
  assume "¬¨(‚àÉx‚ààX. ‚ãÇùîâ = {x})"
  then have r:"‚àÄx‚ààX. ‚ãÇùîâ‚â†{x}" by auto
  {
    fix x assume x:"x:X" "{x}:ùîâ"
    from this(2) assms have "‚àÄC‚ààPow(X). {x} ‚äÜ C ‚ü∂ C ‚àà ùîâ" 
      unfolding IsFilter_def IsUltrafilter_def by auto
    then have s:"{C‚ààPow(X). x‚ààC} ‚äÜ ùîâ" by auto moreover
    {
      fix q assume q:"q‚ààùîâ" "x‚àâq"
      from q(2) s x(1) have "X-q:ùîâ" by blast
      with q(1) have "(X-q)‚à©q‚ààùîâ" using assms unfolding IsUltrafilter_def IsFilter_def by auto
      moreover have "(X-q)‚à©q = 0" by auto
      ultimately have "0:ùîâ" by auto
      with assms have False unfolding IsUltrafilter_def IsFilter_def by auto
    }
    then have "ùîâ‚äÜ{C‚ààPow(X). x‚ààC}" using assms unfolding IsUltrafilter_def IsFilter_def by auto
    with s have "ùîâ={C‚ààPow(X). x‚ààC}" by auto
    then have "‚ãÇùîâ={x}" using x(2) by auto
    with r x(1) have False by auto
  }
  then have "‚àÄx‚ààX. {x}‚àâùîâ" by auto
  then have sub:"(Cofinite(X))-{0} ‚äÜ ùîâ" using ultrafilter_contains_cofinite assms by auto moreover
  from assms have a:"X‚â†0" unfolding IsUltrafilter_def IsFilter_def by auto
  have b:"X ‚àà(Cofinite(X))" using carr_open[of "Cofinite(X)"] 
      union_cocardinal[of nat X] CoCar_is_topology[OF InfCard_nat] unfolding Cofinite_def by auto
  from a b have "X‚àà(Cofinite(X))-{0}" by auto
  with sub have A:"‚ãÇùîâ ‚äÜ ‚ãÇ((Cofinite(X))-{0})" by auto
  {
    fix q assume q:"q‚àà‚ãÇ((Cofinite(X))-{0})"
    from a b have B:"‚ãÇ((Cofinite(X))-{0}) ‚äÜ X" by auto
    with q have "q‚ààX" by auto
    then have "X-(X-{q}) = {q}" by auto
    then have "X-(X-{q})‚âà1" using singleton_eqpoll_1 by auto
    then have "X-(X-{q}) ‚â∫ nat" using eq_lesspoll_trans n_lesspoll_nat[OF nat_1I] by auto
    then have cof:"X-{q}‚àà(Cofinite(X))" unfolding Cofinite_def CoCardinal_def by auto
    {
      assume "X-{q} = 0"
      then have "X={q}" using `q:X` by auto
      then have "‚ãÇùîâ ‚äÜ {q}"  using assms unfolding IsUltrafilter_def IsFilter_def by auto
      with `q‚ààX` r have "‚ãÇùîâ = 0" by blast
    } moreover
    {
      assume "X-{q} ‚â† 0"
      with cof have "X-{q}‚àà(Cofinite(X)) -{0}" by auto
      then have "‚ãÇ((Cofinite(X))-{0}) ‚äÜ X-{q}" by auto
      with q have False by auto
      then have "‚ãÇùîâ = 0" by auto
    } ultimately
    have "‚ãÇùîâ = 0" by auto
  }
  then have "‚ãÇùîâ=0 ‚à® ‚ãÇ((Cofinite(X))-{0}) = 0" by blast
  with A show "‚ãÇùîâ=0" by auto
qed
