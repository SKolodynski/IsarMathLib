text‹The natural quotient map $q: G \to G/H$ (for a normal subgroup $H$) is a group
  homomorphism. Recall that $\lambda x\in X. p(x)$ is an alternative notation for a
  function defined as a set of pairs, see lemma ‹lambda_fun_alt› in ‹func1.thy›.›

lemma (in group0) quotient_map:
  assumes "IsAnormalSubgroup(G,P,H)"
  defines "r ≡ QuotientGroupRel(G,P,H)" and "q ≡ λx∈G. QuotientGroupRel(G,P,H)``{x}"
  shows "Homomor(q,G,P,G//r,QuotientGroupOp(G,P,H))"
  using groupAssum assms group_op_closed lam_funtype lamE EquivClass_1_L10
    Group_ZF_2_4_L3 Group_ZF_2_4_L5A Group_ZF_2_4_T1
  unfolding IsAnormalSubgroup_def QuotientGroupOp_def Homomor_def IsMorphism_def
  by simp

subsection‹The group homomorphism locale›

text‹The next locale packages two groups together with a group homomorphism between them.
  We fix an origin group $(G, P)$ and a target group $(H, F)$ with homomorphism
  $f: G \to H$, and introduce subscripted notation for both groups.›

locale group_homo =
  fixes G P H F f

  assumes origin: "IsAgroup(G, P)"
    and target: "IsAgroup(H, F)"
    and hom: "Homomor(f, G, P, H, F)"

  fixes neut ("𝟭⇩G")
  defines neut_def [simp]: "𝟭⇩G ≡ TheNeutralElement(G, P)"

  fixes groper (infixl "⋅⇩G" 70)
  defines groper_def [simp]: "x ⋅⇩G y ≡ P`⟨x, y⟩"

  fixes inv ("_¯⇩G" [90] 91)
  defines inv_def [simp]: "x¯⇩G ≡ GroupInv(G, P)`x"

  fixes listprod ("∏⇩G _" 70)
  defines listprod_def [simp]: "∏⇩G s ≡ Fold(P, 𝟭⇩G, s)"

  fixes pow
  defines pow_def [simp]: "pow(n, x) ≡ ∏⇩G {⟨k, x⟩. k ∈ n}"

  fixes neutH ("𝟭⇩H")
  defines neutH_def [simp]: "𝟭⇩H ≡ TheNeutralElement(H, F)"

  fixes groperH (infixl "⋅⇩H" 70)
  defines groperH_def [simp]: "x ⋅⇩H y ≡ F`⟨x, y⟩"

  fixes invH ("_¯⇩H" [90] 91)
  defines invH_def [simp]: "x¯⇩H ≡ GroupInv(H, F)`x"

  fixes listprodH ("∏⇩H _" 70)
  defines listprodH_def [simp]: "∏⇩H s ≡ Fold(F, 𝟭⇩H, s)"

  fixes powH
  defines powH_def [simp]: "powH(n, x) ≡ ∏⇩H {⟨k, x⟩. k ∈ n}"

text‹‹ker› denotes the kernel of $f$, i.e. the preimage of the neutral element
  of the target group.›

abbreviation (in group_homo) kernel ("ker" 90) where
  "ker ≡ f-``{𝟭⇩H}"

text‹The theorems proven in the ‹group0› context are valid in the ‹group_homo›
  context when applied to the origin group $G$.›

sublocale group_homo < org: group0 G P neut groper inv listprod pow
  using origin unfolding group0_def groper_def inv_def neut_def by auto

text‹The theorems proven in the ‹group0› context are valid in the ‹group_homo›
  context when applied to the target group $H$.›

sublocale group_homo < tgt: group0 H F neutH groperH invH listprodH powH
  using target unfolding group0_def groperH_def invH_def neutH_def by auto

text‹$f$ is a function from $G$ to $H$.›

lemma (in group_homo) f_is_fun: shows "f: G → H"
  using hom unfolding Homomor_def by simp

text‹$f$ is a group homomorphism: $f(x \cdot_G y) = f(x) \cdot_H f(y)$.›

lemma (in group_homo) f_hom:
  assumes "x ∈ G" "y ∈ G"
  shows "f`(x ⋅⇩G y) = f`x ⋅⇩H f`y"
  using hom assms homomor_eq by simp

text‹$f$ maps the neutral element of $G$ to the neutral element of $H$.›

lemma (in group_homo) f_neutral: shows "f`(𝟭⇩G) = 𝟭⇩H"
proof -
  from org.group0_2_L2 have eG: "𝟭⇩G ∈ G" and neut: "𝟭⇩G ⋅⇩G 𝟭⇩G = 𝟭⇩G" by simp_all
  from eG have fval: "f`(𝟭⇩G) ∈ H" using f_is_fun apply_type by simp
  have eq: "f`(𝟭⇩G) = f`(𝟭⇩G) ⋅⇩H f`(𝟭⇩G)"
    using f_hom[OF eG eG] neut by auto 
  then have "((f`(𝟭⇩G))¯⇩H)⋅⇩H(f`(𝟭⇩G)) = ((f`(𝟭⇩G))¯⇩H)⋅⇩H(f`(𝟭⇩G) ⋅⇩H f`(𝟭⇩G))" by auto
  with fval show ?thesis
    using tgt.group0_2_L6 tgt.group_oper_assoc tgt.inverse_in_group
      tgt.group0_2_L2 by auto
qed

text‹$f$ commutes with the group inverse: $f(x^{-1}) = f(x)^{-1}$.›

lemma (in group_homo) f_inv:
  assumes "x ∈ G"
  shows "f`(x¯⇩G) = (f`x)¯⇩H"
proof -
  from assms have invxG: "x¯⇩G ∈ G" using org.inverse_in_group by simp
  from assms have fxH: "f`x ∈ H" using f_is_fun apply_type by simp
  from invxG have finvH: "f`(x¯⇩G) ∈ H" using f_is_fun apply_type by simp
  have "f`x ⋅⇩H f`(x¯⇩G) = f`(x ⋅⇩G x¯⇩G)"
    using assms invxG f_hom by simp
  also from assms have "… = f`(𝟭⇩G)"
    using org.group0_2_L6 by simp
  also have "… = 𝟭⇩H" using f_neutral by simp
  finally have "f`x ⋅⇩H f`(x¯⇩G) = 𝟭⇩H" .
  with fxH finvH show ?thesis using tgt.group0_2_L9 by simp
qed

text‹The preimage of a subgroup of $H$ is a subgroup of $G$.›

lemma (in group_homo) preimage_is_subgroup:
  assumes sub: "IsAsubgroup(K, F)"
  shows "IsAsubgroup(f-``(K), P)"
proof -
  from sub have KH: "K ⊆ H" using tgt.group0_3_L2 by simp
  from sub have neutH_K: "𝟭⇩H ∈ K" using tgt.group0_3_L5 by simp
  from org.group0_2_L2 f_neutral neutH_K have ne: "f-``(K) ≠ 0"
    using f_is_fun func1_1_L15 by auto
  have subG: "f-``(K) ⊆ G" using f_is_fun func1_1_L3 by simp
  have inv_cl: "∀x ∈ f-``(K). x¯⇩G ∈ f-``(K)"
  proof
    fix x assume "x ∈ f-``(K)"
    with f_is_fun have xG: "x ∈ G" and fxK: "f`x ∈ K" using func1_1_L15 by auto
    from xG have "f`(x¯⇩G) = (f`x)¯⇩H" using f_inv by simp
    also from sub fxK have "(f`x)¯⇩H ∈ K" using tgt.group0_3_T3A by simp
    finally have "f`(x¯⇩G) ∈ K" .
    with xG show "x¯⇩G ∈ f-``(K)"
      using f_is_fun org.inverse_in_group func1_1_L15 by simp
  qed
  have op_cl: "f-``(K) {is closed under} P"
    unfolding IsOpClosed_def
  proof (intro ballI)
    fix x y assume "x ∈ f-``(K)" "y ∈ f-``(K)"
    with f_is_fun have xG: "x ∈ G" "f`x ∈ K" and yG: "y ∈ G" "f`y ∈ K"
      using func1_1_L15 by auto
    from xG yG have "f`(x ⋅⇩G y) = f`x ⋅⇩H f`y" using f_hom by simp
    also from sub xG(2) yG(2) have "f`x ⋅⇩H f`y ∈ K"
      using tgt.group0_3_L6 by simp
    finally have "f`(x ⋅⇩G y) ∈ K" .
    with xG(1) yG(1) show "P`⟨x,y⟩ ∈ f-``(K)"
      using f_is_fun org.group_op_closed func1_1_L15 by simp
  qed
  from subG ne op_cl inv_cl show ?thesis using org.group0_3_T3 by simp
qed

text‹The image of a subgroup of $G$ is a subgroup of $H$.›

lemma (in group_homo) image_is_subgroup:
  assumes sub: "IsAsubgroup(K, P)"
  shows "IsAsubgroup(f``K, F)"
proof -
  from sub have KG: "K ⊆ G" using org.group0_3_L2 by simp
  have subH: "f``K ⊆ H" using f_is_fun KG func_imagedef apply_type by auto
  from sub KG have neutK: "𝟭⇩G ∈ K" using org.group0_3_L5 by simp
  from neutK KG have "f`(𝟭⇩G) ∈ f``K"
    using f_is_fun func_imagedef by auto
  hence ne: "f``K ≠ 0" by blast
  have inv_cl: "∀x ∈ f``K. x¯⇩H ∈ f``K"
  proof
    fix x assume "x ∈ f``K"
    with f_is_fun KG obtain k where kK: "k ∈ K" and xfk: "x = f`k"
      using func_imagedef by auto
    from kK KG have kG: "k ∈ G" by auto
    from sub kK have "k¯⇩G ∈ K" using org.group0_3_T3A by simp
    with KG have "f`(k¯⇩G) ∈ f``K" using f_is_fun func_imagedef by auto
    with kG xfk show "x¯⇩H ∈ f``K" using f_inv by simp
  qed
  have op_cl: "f``K {is closed under} F"
    unfolding IsOpClosed_def
  proof (intro ballI)
    fix x y assume "x ∈ f``K" "y ∈ f``K"
    with f_is_fun KG obtain kx ky where
      kx: "kx ∈ K" "x = f`kx" and ky: "ky ∈ K" "y = f`ky"
      using func_imagedef by auto
    from sub kx(1) ky(1) have "kx ⋅⇩G ky ∈ K" using org.group0_3_L6 by simp
    with KG have "f`(kx ⋅⇩G ky) ∈ f``K" using f_is_fun func_imagedef by auto
    with kx ky KG show "F`⟨x,y⟩ ∈ f``K" using f_hom[of kx ky] by force
  qed
  from subH ne op_cl inv_cl show ?thesis using tgt.group0_3_T3 by simp
qed

text‹The image of $G$ under $f$ is a subgroup of $H$.›

lemma (in group_homo) image_is_group:
  shows "IsAsubgroup(f``G, F)"
proof -
  from origin have "restrict(P, G × G) = P"
    using org.group_oper_fun restrict_domain unfolding group0_def by blast
  with origin have "IsAsubgroup(G, P)"
    unfolding IsAsubgroup_def IsAgroup_def by auto
  thus ?thesis using image_is_subgroup by simp
qed

text‹Membership in the kernel: $x \in G$ belongs to $\mathrm{ker}(f)$ iff
  $f(x) = \mathbf{1}_H$.›

lemma (in group_homo) kernel_mem:
  assumes "x ∈ G"
  shows "x ∈ ker ⟷ f`x = 𝟭⇩H"
  using f_is_fun assms func1_1_L15 by auto

text‹The kernel is a subset of $G$.›

lemma (in group_homo) kernel_in_G: shows "ker ⊆ G"
  using f_is_fun func1_1_L3 by simp

text‹The neutral element of $G$ belongs to the kernel.›

lemma (in group_homo) neutral_in_kernel: shows "𝟭⇩G ∈ ker"
  using f_is_fun org.group0_2_L2 f_neutral kernel_mem by simp

text‹The preimage of a normal subgroup of $H$ is a normal subgroup of $G$.›

theorem (in group_homo) preimage_normal_subgroup:
  assumes "IsAnormalSubgroup(H, F, K)"
  shows "IsAnormalSubgroup(G, P, f-``(K))"
proof -
  from assms have sub: "IsAsubgroup(K, F)"
    unfolding IsAnormalSubgroup_def by simp
  from sub have prsub: "IsAsubgroup(f-``(K), P)"
    using preimage_is_subgroup by simp
  from sub have KH: "K ⊆ H" using tgt.group0_3_L2 by simp
  have "∀g∈G. {P`⟨g, P`⟨h, GroupInv(G, P)`g⟩⟩. h ∈ f-``(K)} ⊆ f-``(K)"
  proof
    fix g assume gG: "g ∈ G"
    { fix h assume "h ∈ {P`⟨g, P`⟨h, GroupInv(G, P)`g⟩⟩. h ∈ f-``(K)}"
      then obtain k where
        k: "h = P`⟨g, P`⟨k, GroupInv(G, P)`g⟩⟩" "k ∈ f-``(K)"
        by auto
      from f_is_fun k(2) have kG: "k ∈ G" using func1_1_L3 by blast
      from f_is_fun k(2) have fkK: "f`k ∈ K" using func1_1_L15 by simp
      from fkK KH have fkH: "f`k ∈ H" by auto
      from gG have fgH: "f`g ∈ H" using f_is_fun apply_type by simp
      from gG have invgG: "GroupInv(G, P)`g ∈ G"
        using org.inverse_in_group by simp
      from kG invgG have pkig: "P`⟨k, GroupInv(G, P)`g⟩ ∈ G"
        using org.group_op_closed by simp
      from gG pkig have hG: "h ∈ G"
        using org.group_op_closed k(1) by simp
      from gG kG invgG pkig have step1:
        "f`(P`⟨g, P`⟨k, GroupInv(G, P)`g⟩⟩) =
         F`⟨f`g, F`⟨f`k, GroupInv(H, F)`(f`g)⟩⟩"
        using f_hom f_inv by simp
      from assms fkK fgH have "F`⟨F`⟨f`g, f`k⟩, GroupInv(H, F)`(f`g)⟩ ∈ K"
        unfolding IsAnormalSubgroup_def by auto
      moreover from fgH fkH have
        "F`⟨f`g, F`⟨f`k, GroupInv(H, F)`(f`g)⟩⟩ =
         F`⟨F`⟨f`g, f`k⟩, GroupInv(H, F)`(f`g)⟩"
        using tgt.inverse_in_group tgt.group_oper_assoc by simp
      ultimately have "f`h ∈ K" using step1 k(1) by simp
      with f_is_fun hG have "h ∈ f-``(K)" using func1_1_L15 by simp
    }
    thus "{P`⟨g, P`⟨h, GroupInv(G, P)`g⟩⟩. h ∈ f-``(K)} ⊆ f-``(K)"
      by blast
  qed
  with origin prsub show ?thesis using org.cont_conj_is_normal by simp
qed

text‹The kernel is a normal subgroup of $G$.›

theorem (in group_homo) kernel_normal_subgroup:
  shows "IsAnormalSubgroup(G, P, ker)"
proof -
  from target have "𝟭⇩H ∈ H" using tgt.group0_2_L2 by simp
  hence "IsAnormalSubgroup(H, F, {𝟭⇩H})"
    using tgt.trivial_normal_subgroup by simp
  thus ?thesis using preimage_normal_subgroup by simp
qed

text‹In particular, the kernel is a subgroup of $G$.›

corollary (in group_homo) kernel_is_subgroup:
  shows "IsAsubgroup(ker, P)"
  using kernel_normal_subgroup unfolding IsAnormalSubgroup_def by simp

text‹The inverse of a bijective homomorphism is a homomorphism.›

theorem (in group_homo) bijective_hom_inverse:
  assumes bij: "f ∈ bij(G, H)"
  shows "Homomor(converse(f), H, F, G, P)"
proof -
  { fix h⇩1 h⇩2 assume "h⇩1 ∈ H" "h⇩2 ∈ H"
    with bij obtain g⇩1 g⇩2 where
      g1: "g⇩1 ∈ G" "f`g⇩1 = h⇩1" and g2: "g⇩2 ∈ G" "f`g⇩2 = h⇩2"
      unfolding bij_def surj_def by blast
    from g1(1) g2(1) have
      "converse(f)`(f`(g⇩1 ⋅⇩G g⇩2)) = converse(f)`(f`g⇩1 ⋅⇩H f`g⇩2)"
      using f_hom by simp
    with bij g1 g2 have
      "P`⟨converse(f)`h⇩1, converse(f)`h⇩2⟩ = converse(f)`(F`⟨h⇩1,h⇩2⟩)"
      using left_inverse org.group_op_closed unfolding bij_def by auto
  }
  with bij show ?thesis using bij_converse_bij bij_is_fun
    unfolding Homomor_def IsMorphism_def by simp
qed

text‹The first isomorphism theorem: $f$ induces a bijective homomorphism
  $G/\mathrm{ker}(f) \cong f(G)$.›

theorem (in group_homo) first_isomorphism_theorem:
  defines "r ≡ QuotientGroupRel(G, P, ker)"
      and "𝒫 ≡ QuotientGroupOp(G, P, ker)"
  shows "∃𝔣. Homomor(𝔣, G//r, 𝒫, f``G, restrict(F, f``G × f``G)) ∧
              𝔣 ∈ bij(G//r, f``G)"
proof -
  let ?𝔣 = "{⟨r``{g}, f`g⟩. g ∈ G}"
  from org.Group_ZF_2_4_L3 kernel_normal_subgroup have equiv: "equiv(G, r)"
    unfolding r_def IsAnormalSubgroup_def by simp
  from r_def f_is_fun have "?𝔣 ∈ Pow((G//r) × f``G)"
    unfolding quotient_def using func_imagedef by auto
  moreover have "(G//r) ⊆ domain(?𝔣)" unfolding domain_def quotient_def by auto
  moreover
  { fix x y t assume A: "⟨x,y⟩ ∈ ?𝔣" "⟨x,t⟩ ∈ ?𝔣"
    then obtain g⇩y g⇩r where
      "⟨x,y⟩ = ⟨r``{g⇩y},f`g⇩y⟩" "⟨x,t⟩ = ⟨r``{g⇩r},f`g⇩r⟩" "g⇩r ∈ G" "g⇩y ∈ G" by auto
    hence B: "r``{g⇩y} = r``{g⇩r}" "y = f`g⇩y" "t = f`g⇩r" by auto
    from f_is_fun ‹g⇩y ∈ G› ‹g⇩r ∈ G› B(2,3) have yH: "y ∈ H" and tH: "t ∈ H"
      using apply_type by simp_all
    with equiv ‹g⇩r ∈ G› B(1) have "⟨g⇩y,g⇩r⟩ ∈ r" using same_image_equiv by simp
    with r_def f_is_fun have
      "f`(P`⟨g⇩y, GroupInv(G,P)`g⇩r⟩) = TheNeutralElement(H,F)"
      unfolding QuotientGroupRel_def using func1_1_L15 by simp
    with f_hom f_inv[of "g⇩r"]  ‹g⇩y ∈ G› ‹g⇩r ∈ G›
    have "F`⟨f`(g⇩y),GroupInv(H,F)`(f`(g⇩r))⟩ = TheNeutralElement(H,F)"
      using org.inverse_in_group
      by auto
    with B(2,3) yH tH have "y = t"
      using tgt.group0_2_L11A[of y t] by auto
  } hence "∀x y. ⟨x,y⟩ ∈ ?𝔣 ⟶ (∀z. ⟨x,z⟩ ∈ ?𝔣 ⟶ y = z)" by auto
  ultimately have ff_fun: "?𝔣: G//r → f``G" unfolding Pi_def function_def by auto
  { fix a⇩1 a⇩2 assume AS: "a⇩1 ∈ G//r" "a⇩2 ∈ G//r"
    then obtain g⇩1 g⇩2 where g1: "g⇩1 ∈ G" and g2: "g⇩2 ∈ G"
        and a: "a⇩1 = r``{g⇩1}" "a⇩2 = r``{g⇩2}"
      unfolding quotient_def by auto
    with origin equiv have "⟨𝒫`⟨a⇩1,a⇩2⟩, f`(P`⟨g⇩1,g⇩2⟩)⟩ ∈ ?𝔣"
      using Group_ZF_2_4_L5A kernel_normal_subgroup
        org.Group_ZF_2_2_L2 org.group_op_closed
      unfolding 𝒫_def QuotientGroupOp_def r_def by auto
    with ff_fun have eq: "?𝔣`(𝒫`⟨a⇩1,a⇩2⟩) = f`(P`⟨g⇩1,g⇩2⟩)" using apply_equality by simp
    from g1 g2 a have p1: "⟨a⇩1,f`g⇩1⟩ ∈ ?𝔣" and p2: "⟨a⇩2,f`g⇩2⟩ ∈ ?𝔣" by auto
    with ff_fun have "?𝔣`(a⇩1) = f`(g⇩1)" "?𝔣`(a⇩2) = f`(g⇩2)"
      using apply_equality by auto
    with g1 g2 eq have "F`⟨?𝔣`a⇩1, ?𝔣`a⇩2⟩ = ?𝔣`(𝒫`⟨a⇩1,a⇩2⟩)"
      using f_hom unfolding Homomor_def IsMorphism_def by simp
    moreover from AS ff_fun have "?𝔣`a⇩1 ∈ f``G" "?𝔣`a⇩2 ∈ f``G" using apply_type by auto
    ultimately have "restrict(F,f``G×f``G)`⟨?𝔣`a⇩1,?𝔣`a⇩2⟩ = ?𝔣`(𝒫`⟨a⇩1,a⇩2⟩)" by simp
  } hence rr: "∀a⇩1∈G//r. ∀a⇩2∈G//r.

text‹A group homomorphism is injective if and only if its kernel is trivial.›

theorem (in group_homo) kernel_trivial_iff_injective:
  shows "f ∈ inj(G, H) ⟷ ker = {𝟭⇩G}"
proof
  assume inj: "f ∈ inj(G, H)"
  show "ker = {𝟭⇩G}"
  proof
    show "ker ⊆ {𝟭⇩G}"
    proof
      fix x assume xker: "x ∈ ker"
      with f_is_fun have xG: "x ∈ G" using kernel_in_G by auto
      from xker xG have fx: "f`x = 𝟭⇩H" using kernel_mem by simp
      from xG org.group0_2_L2 f_neutral fx inj
      show "x ∈ {𝟭⇩G}" unfolding inj_def by auto
    qed
    show "{𝟭⇩G} ⊆ ker" using neutral_in_kernel by simp
  qed
next
  assume triv: "ker = {𝟭⇩G}"
  { fix x y assume xG: "x ∈ G" and yG: "y ∈ G" and eq: "f`x = f`y"
    from yG have invyG: "y¯⇩G ∈ G" using org.inverse_in_group by simp
    from xG invyG have prodG: "x ⋅⇩G y¯⇩G ∈ G" using org.group_op_closed by simp
    from xG invyG have step1: "f`(x ⋅⇩G y¯⇩G) = f`x ⋅⇩H f`(y¯⇩G)"
      using f_hom by simp
    from yG have step2: "f`(y¯⇩G) = (f`y)¯⇩H" using f_inv by simp
    from f_is_fun yG have fyH: "f`y ∈ H" using apply_type by simp
    from fyH have step3: "f`y ⋅⇩H (f`y)¯⇩H = 𝟭⇩H"
      using tgt.group0_2_L6 by simp
    from step1 step2 step3 eq have feq: "f`(x ⋅⇩G y¯⇩G) = 𝟭⇩H" by simp
    from prodG feq have "x ⋅⇩G y¯⇩G ∈ ker" using kernel_mem by simp
    with triv have "x ⋅⇩G y¯⇩G = 𝟭⇩G" by simp
    with xG yG have "x = y" using org.group0_2_L11A by simp
  }
  with f_is_fun show "f ∈ inj(G, H)" unfolding inj_def by auto
qed

subsection‹Exact sequences at a term›

text‹A pair of composable group homomorphisms $A \xrightarrow{f} B \xrightarrow{g} C$
  is \emph{exact at $B$} when the image of $f$ equals the kernel of $g$.›

definition IsExactAt :: "[i, i, i, i, i] ⇒ o" where
  "IsExactAt(f, A, g, C, Q) ≡ f``A = g-``{TheNeutralElement(C, Q)}"

text‹A locale for two composable group homomorphisms $f\colon A \to B$ and
  $g\colon B \to C$. Lemmas are proved using the two ‹group_homo› instances
  directly rather than via sublocale inheritance.›

locale two_group_homo =
  fixes A P B F C Q f g
  assumes hf: "group_homo(A, P, B, F, f)"
    and hg: "group_homo(B, F,C, Q, g)"

abbreviation (in two_group_homo) ExactAtB where
  "ExactAtB ≡ IsExactAt(f, A, g, C, Q)"


sublocale two_group_homo < second_homo: group_homo B F C Q g
  "TheNeutralElement(B,F)" "λx y. F`⟨x,y⟩"
  "λx. GroupInv(B,F)`x" "λs.  Fold(F, TheNeutralElement(B, F), s)"
  "λn x.  Fold(F, TheNeutralElement(B, F), {⟨k, x⟩ . k ∈ n})"
  "TheNeutralElement(C,Q)" "λx y. Q`⟨x,y⟩"
  "λx. GroupInv(C,Q)`x" "λs.  Fold(Q, TheNeutralElement(C, Q), s)"
  "λn x.  Fold(Q, TheNeutralElement(C, Q), {⟨k, x⟩ . k ∈ n})"
  apply auto using hg by auto

sublocale two_group_homo < first_homo: group_homo A P B F f
  "TheNeutralElement(A,P)" "λx y. P`⟨x,y⟩"
  "λx. GroupInv(A,P)`x" "λs.  Fold(P, TheNeutralElement(A, P), s)"
  "λn x.  Fold(P, TheNeutralElement(A, P), {⟨k, x⟩ . k ∈ n})"
  "TheNeutralElement(B,F)" "λx y. F`⟨x,y⟩"
  "λx. GroupInv(B,F)`x" "λs.  Fold(F, TheNeutralElement(B, F), s)"
  "λn x.  Fold(F, TheNeutralElement(B, F), {⟨k, x⟩ . k ∈ n})"
  apply (simp only: hf) apply simp apply simp apply simp apply simp apply simp
  by assumption


text‹Exactness at $B$ means the image of $f$ equals the kernel of $g$.›

lemma (in two_group_homo) ExactAtB_unfold:
  shows "ExactAtB ⟷ f``A = g-``{TheNeutralElement(C, Q)}"
  unfolding IsExactAt_def by (simp only:)

text‹If the sequence is exact at $B$, the composition $g \circ f$ maps every element
  of $A$ to the neutral element of $C$.›

lemma (in two_group_homo) exact_comp_trivial:
  assumes "ExactAtB" and "x ∈ A"
  shows "g`(f`x) = TheNeutralElement(C, Q)"
proof -
  from assms(2) have "f`x ∈ f``A"
    using first_homo.f_is_fun func_imagedef by auto
  with assms(1) have ker: "f`x ∈ g-``{TheNeutralElement(C, Q)}"
    unfolding IsExactAt_def by (simp only:)
  from assms(2) have "f`x ∈ B" using first_homo.f_is_fun apply_type by simp
  with ker show ?thesis using group_homo.kernel_mem[OF hg] by (simp only:)
qed

text‹A sufficient condition for exactness: if $g \circ f$ is trivial and the
  kernel of $g$ is contained in the image of $f$, then the sequence is exact at $B$.›

lemma (in two_group_homo) exact_at_B_intro:
  assumes "∀x ∈ A. g`(f`x) = TheNeutralElement(C, Q)"
    and "g-``{TheNeutralElement(C, Q)} ⊆ f``A"
  shows "ExactAtB"
  unfolding IsExactAt_def
proof
  show "f``A ⊆ g-``{TheNeutralElement(C, Q)}"
  proof
    fix y assume "y ∈ f``A"
    with first_homo.f_is_fun obtain x where xA: "x ∈ A" and yf: "y = f`x"
      using func_imagedef by auto
    from xA first_homo.f_is_fun yf have yB: "y ∈ B" using apply_type by simp
    from assms(1) xA yf have "g`y = TheNeutralElement(C, Q)" by (simp only:)
    with yB show "y ∈ g-``{TheNeutralElement(C, Q)}"
      using group_homo.kernel_mem[OF hg] by (simp only:)
  qed
  from assms(2) show "g-``{TheNeutralElement(C, Q)} ⊆ f``A" .
qed

subsection‹The trivial group and zero maps›

text‹The \emph{trivial group} has carrier $1 = \{0\}$ and operation
  $\mathbf{0} \colon 1\times 1 \to 1$ defined by $\mathbf{0}(0,0)=0$.›

definition TrivOp :: i where
  "TrivOp ≡ ConstantFunction(1×1, 0)"

text‹$\mathbf{0}$ is a function $1\times 1 \to 1$.›

lemma TrivOp_type: "TrivOp : 1×1 → 1"
  unfolding TrivOp_def using func1_3_L1 by simp

text‹The value of $\mathbf{0}$ at any pair from $1\times 1$ is $0$.›

lemma TrivOp_val:
  assumes "x ∈ 1" "y ∈ 1"
  shows "TrivOp`⟨x,y⟩ = 0"
proof -
  from assms have "⟨x,y⟩ ∈ 1×1" by auto
  then show ?thesis unfolding TrivOp_def using func1_3_L2 by simp
qed

text‹The neutral element of $(1, \mathbf{0})$ is $0$.›

lemma TrivOp_ne: "TheNeutralElement(1, TrivOp) = 0"
  unfolding TheNeutralElement_def
proof (rule the_equality)
  show "0 ∈ 1 ∧ (∀g∈1. TrivOp`⟨0,g⟩ = g ∧ TrivOp`⟨g,0⟩ = g)"
  proof (intro conjI ballI conjI)
    show "0 ∈ 1" by simp
    fix g assume g1: "g ∈ 1"
    then have g0: "g = 0" by auto
    then show "TrivOp`⟨0,g⟩ = g" using TrivOp_val g1 by auto
    from g0 show "TrivOp`⟨g,0⟩ = g" using TrivOp_val g1 by auto
  qed
next
  fix e assume "e ∈ 1 ∧ (∀g∈1. TrivOp`⟨e,g⟩ = g ∧ TrivOp`⟨g,e⟩ = g)"
  then show "e = 0" by auto
qed

text‹$(1, \mathbf{0})$ is a group.›

lemma trivial_group: "IsAgroup(1, TrivOp)"
proof -
  have assoc: "TrivOp {is associative on} 1"
    unfolding IsAssociative_def
  proof (intro conjI)
    show "TrivOp : 1×1 → 1" using TrivOp_type .
    show "∀x∈1. ∀y∈1. ∀z∈1.
        TrivOp`⟨TrivOp`⟨x,y⟩,z⟩ = TrivOp`⟨x,TrivOp`⟨y,z⟩⟩"
    proof (intro ballI)
      fix x y z assume H: "x ∈ 1" "y ∈ 1" "z ∈ 1"
      have m1: "TrivOp`⟨x,y⟩ ∈ 1" using TrivOp_val H by auto
      have m2: "TrivOp`⟨y,z⟩ ∈ 1" using TrivOp_val H by auto
      have "TrivOp`⟨TrivOp`⟨x,y⟩,z⟩ = 0"
        using TrivOp_val m1 H(3) by simp
      moreover have "TrivOp`⟨x,TrivOp`⟨y,z⟩⟩ = 0"
        using TrivOp_val H(1) m2 by simp
      ultimately show "TrivOp`⟨TrivOp`⟨x,y⟩,z⟩ = TrivOp`⟨x,TrivOp`⟨y,z⟩⟩" by simp
    qed
  qed
  have neutral: "∃e∈1. ∀g∈1. TrivOp`⟨e,g⟩ = g ∧ TrivOp`⟨g,e⟩ = g"
  proof (rule bexI[of _ 0])
    show "∀g∈1. TrivOp`⟨0,g⟩ = g ∧ TrivOp`⟨g,0⟩ = g"
    proof (intro ballI conjI)
      fix g assume g1: "g ∈ 1"
      then have g0: "g = 0" by auto
      then show "TrivOp`⟨0,g⟩ = g" using TrivOp_val g1 by auto
      from g0 show "TrivOp`⟨g,0⟩ = g" using TrivOp_val g1 by auto
    qed
    show "0 ∈ 1" by simp
  qed
  have monoid: "IsAmonoid(1, TrivOp)"
    unfolding IsAmonoid_def using assoc neutral by blast
  have inv: "∀g∈1. ∃b∈1. TrivOp`⟨g,b⟩ = TheNeutralElement(1,TrivOp)"
  proof (intro ballI)
    fix g assume g1: "g ∈ 1"
    show "∃b∈1. TrivOp`⟨g,b⟩ = TheNeutralElement(1,TrivOp)"
      using TrivOp_val TrivOp_ne g1 by auto
  qed
  from monoid inv show "IsAgroup(1,TrivOp)" unfolding IsAgroup_def by blast
qed

text‹$(1, \mathbf{0})$ is abelian.›

lemma trivial_group_abelian: "TrivOp {is commutative on} 1"
  unfolding IsCommutative_def using TrivOp_val by auto

text‹The \emph{zero map} from $(G_1,P_1)$ to $(G_2,P_2)$ sends every element
  to the neutral element of $G_2$.›

definition ZeroMap :: "[i, i, i] ⇒ i" where
  "ZeroMap(G⇩1, G⇩2, P⇩2) ≡ ConstantFunction(G⇩1, TheNeutralElement(G⇩2, P⇩2))"

text‹The zero map is a group homomorphism.›

lemma ZeroMap_hom:
  assumes grp1: "IsAgroup(G⇩1, P⇩1)" and grp2: "IsAgroup(G⇩2, P⇩2)"
  shows "Homomor(ZeroMap(G⇩1, G⇩2, P⇩2), G⇩1, P⇩1, G⇩2, P⇩2)"
proof -
  let ?e⇩1 = "TheNeutralElement(G⇩1, P⇩1)"
  let ?e⇩2 = "TheNeutralElement(G⇩2, P⇩2)"
  interpret g1: group0 G⇩1 P⇩1 ?e⇩1 "λx y. P⇩1`⟨x,y⟩" "λx. GroupInv(G⇩1,P⇩1)`x"
    "λs. Fold(P⇩1,?e⇩1,s)" "λn x. Fold(P⇩1,?e⇩1,{⟨k,x⟩. k∈n})"
    using grp1 unfolding group0_def by auto
  interpret g2: group0 G⇩2 P⇩2 ?e⇩2 "λx y. P⇩2`⟨x,y⟩" "λx. GroupInv(G⇩2,P⇩2)`x"
    "λs. Fold(P⇩2,?e⇩2,s)" "λn x. Fold(P⇩2,?e⇩2,{⟨k,x⟩. k∈n})"
    using grp2 unfolding group0_def by auto
  have eG2: "?e⇩2 ∈ G⇩2" using g2.group0_2_L2_1 by assumption
  have ee2: "P⇩2`⟨?e⇩2,?e⇩2⟩ = ?e⇩2"
    using g2.group0_2_L2_2[OF g2.group0_2_L2_1] by assumption
  have fun_t: "ZeroMap(G⇩1, G⇩2, P⇩2) : G⇩1 → G⇩2"
    unfolding ZeroMap_def using func1_3_L1 eG2 by blast
  have morph: "IsMorphism(G⇩1, P⇩1, P⇩2, ZeroMap(G⇩1, G⇩2, P⇩2))"
    unfolding IsMorphism_def ZeroMap_def
  proof (intro ballI)
    fix x y assume xy: "x ∈ G⇩1" "y ∈ G⇩1"
    from xy have pxy: "P⇩1`⟨x,y⟩ ∈ G⇩1" using g1.group_op_closed by blast
    have "ConstantFunction(G⇩1,?e⇩2)`(P⇩1`⟨x,y⟩) = ?e⇩2"
      using func1_3_L2 pxy by blast
    also have "… = P⇩2`⟨?e⇩2,?e⇩2⟩" using ee2 by (simp only:sym)
    also have "… = P⇩2`⟨ConstantFunction(G⇩1,?e⇩2)`x,
                         ConstantFunction(G⇩1,?e⇩2)`y⟩"
      using func1_3_L2 xy by (simp only:)
    finally show "ConstantFunction(G⇩1,?e⇩2)`(P⇩1`⟨x,y⟩) =
      P⇩2`⟨ConstantFunction(G⇩1,?e⇩2)`x,ConstantFunction(G⇩1,?e⇩2)`y⟩" .
  qed
  from fun_t morph show ?thesis unfolding Homomor_def by simp
qed

text‹The image of the zero map over its domain is the singleton of the neutral element.›

lemma ZeroMap_image:

lemma ZeroMap_preimage_full:

    unfolding Homomor_def by simp
  have val: "∀x ∈ G⇩1. ZeroMap(G⇩1, G⇩2, P⇩2)`x = ?e⇩2"
    unfolding ZeroMap_def using func1_3_L2 by simp
  with fun_t show ?thesis using func1_1_L15 by auto
qed

subsection‹Short exact sequences›

text‹A \emph{short exact sequence} $0 \to A \xrightarrow{f} B \xrightarrow{g} C \to 0$
  is expressed by exactness at all three terms: the sequence is exact at $A$
  (i.e.\ $f$ is injective), at $B$ (i.e.\ $\mathrm{im}(f) = \ker(g)$), and at $C$
  (i.e.\ $g$ is surjective), each encoded via zero maps to or from the trivial group.›

definition IsAshortExactSequence :: "[i, i, i, i, i, i, i, i] ⇒ o" where
  "IsAshortExactSequence(A, P, B, F, C, Q, f, g) ≡
    group_homo(A, P, B, F, f) ∧
    group_homo(B, F, C, Q, g) ∧
    IsExactAt(ZeroMap(1, A, P), 1, f, B, F) ∧
    IsExactAt(f, A, g, C, Q) ∧
    IsExactAt(g, B, ZeroMap(C, 1, TrivOp), 1, TrivOp)"

text‹The ‹short_exact_sequence› locale packages a short exact sequence,
  expressed by exactness at all three terms.›

locale short_exact_sequence = two_group_homo +
  assumes ses_exact_at_A: "IsExactAt(ZeroMap(1, A, P), 1, f, B, F)"
    and ses_exact: "ExactAtB"
    and ses_exact_at_C: "IsExactAt(g, B, ZeroMap(C, 1, TrivOp), 1, TrivOp)"

text‹The image of $f$ equals the kernel of $g$.›

lemma (in short_exact_sequence) ses_im_eq_ker:
  shows "f``A = g-``{TheNeutralElement(C, Q)}"
  using ses_exact unfolding IsExactAt_def by (simp only:)

text‹$f$ is injective: exactness at $A$ forces the kernel of $f$ to be trivial.›

lemma (in short_exact_sequence) ses_inj:
  shows "f ∈ inj(A, B)"
proof -
  from ses_exact_at_A have "ZeroMap(1, A, P)``1 = f-``{TheNeutralElement(B, F)}"
    unfolding IsExactAt_def by (simp only:)
  moreover have "ZeroMap(1, A, P)``1 = {TheNeutralElement(A, P)}"
    using ZeroMap_image[OF trivial_group first_homo.origin] by (simp only:)
  ultimately have "f-``{TheNeutralElement(B, F)} = {TheNeutralElement(A, P)}" by (simp only:)
  with first_homo.kernel_trivial_iff_injective show ?thesis by (simp only:)
qed

text‹$g$ is surjective: exactness at $C$ forces $g$ to hit all of $C$.›

lemma (in short_exact_sequence) ses_surj:
  shows "g``B = C"
proof -
  from ses_exact_at_C have "g``B = ZeroMap(C, 1, TrivOp)-``{TheNeutralElement(1, TrivOp)}"
    unfolding IsExactAt_def by simp
  also have "… = C"
    using ZeroMap_preimage_full[OF second_homo.target trivial_group] by simp
  finally show ?thesis .
qed

text‹The kernel of $f$ is trivial.›

lemma (in short_exact_sequence) ses_ker_trivial:
  shows "f-``{TheNeutralElement(B, F)} = {TheNeutralElement(A, P)}"
  using ses_inj first_homo.kernel_trivial_iff_injective by (simp only:)

text‹$g$ maps each element of $B$ to an element of $C$.›

lemma (in short_exact_sequence) ses_g_type:
  assumes "b ∈ B"
  shows "g`b ∈ C"
  using assms second_homo.f_is_fun apply_type by simp

text‹An element of $B$ in the kernel of $g$ is in the image of $f$.›

lemma (in short_exact_sequence) ses_ker_in_im:
  assumes "b ∈ B" and "g`b = TheNeutralElement(C, Q)"
  shows "b ∈ f``A"
  using assms second_homo.kernel_mem ses_im_eq_ker by (simp only:)

text‹An element in the image of $f$ maps to the neutral element under $g$.›

lemma (in short_exact_sequence) ses_im_to_ker:
  assumes "b ∈ f``A"
  shows "g`b = TheNeutralElement(C, Q)"
proof -
  from assms ses_im_eq_ker have "b ∈ g-``{TheNeutralElement(C, Q)}" by (simp only:)
  with assms first_homo.f_is_fun have "b ∈ B"
    using func1_1_L6(2) by blast
  with ‹b ∈ g-``{TheNeutralElement(C, Q)}›
  show ?thesis using second_homo.kernel_mem by (simp only:)
qed

text‹First isomorphism theorem for a short exact sequence: the quotient $B/f(A)$
  is isomorphic to $C$ via the map induced by $g$.›

theorem (in short_exact_sequence) ses_first_iso:
  defines "r ≡ QuotientGroupRel(B, F, f``A)"
      and "𝒫 ≡ QuotientGroupOp(B, F, f``A)"
  shows "∃𝔣. Homomor(𝔣, B//r, 𝒫, C, Q) ∧ 𝔣 ∈ bij(B//r, C)"
proof -
  have ker_eq: "second_homo.kernel = f``A"
    using ses_im_eq_ker by (simp only:sym)
  have sur: "g``B = C" using ses_surj .
  have restQ: "restrict(Q, C × C) = Q"
    using second_homo.tgt.group_oper_fun restrict_domain by simp
  from second_homo.first_isomorphism_theorem obtain 𝔣 where
    HOM: "Homomor(𝔣, B // QuotientGroupRel(B, F, second_homo.kernel),
                  QuotientGroupOp(B, F, second_homo.kernel),
                  g``B, restrict(Q, g``B × g``B))"
    and BIJ: "𝔣 ∈ bij(B // QuotientGroupRel(B, F, second_homo.kernel), g``B)"
    by blast
  from HOM ker_eq sur restQ have "Homomor(𝔣, B//r, 𝒫, C, Q)"
    unfolding r_def 𝒫_def by force
  moreover from BIJ ker_eq sur have "𝔣 ∈ bij(B//r, C)"
    unfolding r_def by force
  ultimately show ?thesis by blast
qed

subsection‹The cokernel›

text‹A \emph{cokernel} of $f \colon A \to B$ is a group homomorphism
  $q \colon B \to C$ satisfying (i) $q \circ f = 0$, and (ii) the universal
  property: every $h \colon B \to D$ with $h \circ f = 0$ factors uniquely
  through $q$.›

definition IsAcokernel :: "[i, i, i, i, i, i, i] ⇒ o" where
  "IsAcokernel(f, A, B, F, q, C, Q) ≡
    group_homo(B, F, C, Q, q) ∧
    (∀x ∈ A. q`(f`x) = TheNeutralElement(C, Q)) ∧
    (∀D R h. IsAgroup(D, R) ∧ Homomor(h, B, F, D, R) ∧
             (∀x ∈ A. h`(f`x) = TheNeutralElement(D, R)) ⟶
      (∃!hh. Homomor(hh, C, Q, D, R) ∧ (∀b ∈ B. hh`(q`b) = h`b)))"

text‹If the target group $H$ is abelian then every subgroup is normal, so
  in particular $f(G)$ is a normal subgroup of $H$.›

lemma (in group_homo) image_normal_if_abelian:
  assumes "F {is commutative on} H"
  shows "IsAnormalSubgroup(H, F, f``G)"
  using target assms image_is_group Group_ZF_2_4_L6(1) by simp

text‹The natural quotient map $q \colon H \to H/f(G)$ is a cokernel of $f$
  whenever $H$ is abelian.  The factored map for the universal property is
  $\bar h = \{\langle r``\{b\}, h`b\rangle \mid b \in H\}$, which is exactly the
  construction used in the first isomorphism theorem.›

theorem (in group_homo) quotient_is_cokernel:
  assumes abel: "F {is commutative on} H"
  defines "r ≡ QuotientGroupRel(H, F, f``G)"
      and "q ≡ λh∈H. QuotientGroupRel(H, F, f``G)``{h}"
      and "Q ≡ QuotientGroupOp(H, F, f``G)"
  shows "IsAcokernel(f, G, H, F, q, H//r, Q)"
  unfolding IsAcokernel_def
proof (intro conjI)
  from image_normal_if_abelian[OF abel]
  have norm: "IsAnormalSubgroup(H, F, f``G)" .
  from norm have equiv_r: "equiv(H, r)"
    unfolding r_def IsAnormalSubgroup_def using tgt.Group_ZF_2_4_L3 by simp
  from norm target have Qgrp: "IsAgroup(H//r, Q)"
    unfolding Q_def r_def using Group_ZF_2_4_T1 by simp
  let ?e⇩Q = "TheNeutralElement(H//r, Q)"
  from norm target have neutQ: "r``{𝟭⇩H} = ?e⇩Q"
    using Group_ZF_2_4_L5B unfolding r_def Q_def by simp
  show "group_homo(H, F, H//r, Q, q)"
  proof -
    have "Homomor(q, H, F, H//r, Q)"
      using tgt.quotient_map[OF norm] unfolding r_def q_def Q_def by simp
    with target Qgrp show ?thesis unfolding group_homo_def by simp
  qed
  show "∀x ∈ G. q`(f`x) = ?e⇩Q"
  proof
    fix x assume aG: "x ∈ G"
    from aG have faH: "f`x ∈ H" using f_is_fun apply_type by simp
    from aG have faim: "f`x ∈ f``G" using f_is_fun func_imagedef by auto
    from faH faim have in_r: "⟨f`x, 𝟭⇩H⟩ ∈ r"
      unfolding r_def using tgt.Group_ZF_2_4_L5C by simp
    from in_r tgt.group0_2_L2 have "r``{f`x} = r``{𝟭⇩H}"
      using equiv_r equiv_class_eq by simp
    also have "… = ?e⇩Q" using neutQ by simp
    finally have cls: "r``{f`x} = ?e⇩Q" .
    with faH show "q`(f`x) = ?e⇩Q" unfolding q_def r_def by simp
  qed
  show "∀D R h. IsAgroup(D, R) ∧ Homomor(h, H, F, D, R) ∧
               (∀x ∈ G. h`(f`x) = TheNeutralElement(D, R)) ⟶
    (∃!hh. Homomor(hh, H//r, Q, D, R) ∧ (∀b ∈ H. hh`(q`b) = h`b))"
  proof (intro allI)
    fix D R h
    show "IsAgroup(D, R) ∧ Homomor(h, H, F, D, R) ∧
               (∀x ∈ G. h`(f`x) = TheNeutralElement(D, R)) ⟶
      (∃!hh. Homomor(hh, H//r, Q, D, R) ∧ (∀b ∈ H. hh`(q`b) = h`b))"
    proof (intro impI)
    assume conj: "IsAgroup(D, R) ∧ Homomor(h, H, F, D, R) ∧
                  (∀x ∈ G. h`(f`x) = TheNeutralElement(D, R))"
    from conj have Dgrp: "IsAgroup(D, R)" and hHom: "Homomor(h, H, F, D, R)"
      and hf0: "∀x ∈ G. h`(f`x) = TheNeutralElement(D, R)" by auto
    let ?e⇩D = "TheNeutralElement(D, R)"
    let ?h = "{⟨r``{b}, h`b⟩. b ∈ H}"
    interpret h_hom: group_homo H F D R h 
      "TheNeutralElement(H,F)" "λx y. F`⟨x,y⟩"
      "λx. GroupInv(H,F)`x" "λs. Fold(F, TheNeutralElement(H, F),s)"
      "λn x.  Fold(F, TheNeutralElement(H, F), {⟨k, x⟩ . k ∈ n})"
      "TheNeutralElement(D,R)" "λx y. R`⟨x,y⟩"
      "λx. GroupInv(D,R)`x" "λs. Fold(R, TheNeutralElement(D, R),s)"
      "λn x.  Fold(R, TheNeutralElement(D, R), {⟨k, x⟩ . k ∈ n})"
      using target Dgrp hHom unfolding group_homo_def by simp
    have hfun: "h: H → D" using h_hom.f_is_fun .
    have im_in_ker: "∀x ∈ f``G. h`x = ?e⇩D"
    proof
      fix x assume "x ∈ f``G"
      with f_is_fun obtain ga where "ga ∈ G" "x = f`ga"
        using func_imagedef by auto
      with hf0 show "h`x = ?e⇩D" by blast
    qed
    have bhfun: "?h : H//r → D"
    proof -
      have sub: "?h ∈ Pow((H//r) × D)"
        unfolding quotient_def using hfun apply_type by auto
      have dom: "H//r ⊆ domain(?h)"
        unfolding domain_def quotient_def by force
      have sv: "∀c y. ⟨c,y⟩ ∈ ?h ⟶ (∀z. ⟨c,z⟩ ∈ ?h ⟶ y = z)"
      proof (intro allI impI)
        fix c y z
        assume H1: "⟨c,y⟩ ∈ ?h" and H2: "⟨c,z⟩ ∈ ?h"
        from H1 obtain b⇩1 where b1: "b⇩1 ∈ H" "c = r``{b⇩1}" "y = h`b⇩1" by auto
        from H2 obtain b⇩2 where b2: "b⇩2 ∈ H" "c = r``{b⇩2}" "z = h`b⇩2" by auto
        from b1(2) b2(2) have eq: "r``{b⇩1} = r``{b⇩2}" by simp
        from equiv_r b2(1) eq have rel: "⟨b⇩1, b⇩2⟩ ∈ r"
          using same_image_equiv by simp
        have R:"⋀t q X P. ⟨t,q⟩∈{⟨x,y⟩∈X. P(x,y)} ⟹ P(t,q)"
          by auto
        from rel have "⟨b⇩1,b⇩2⟩ ∈{⟨x,y⟩∈H×H. F`⟨x, GroupInv(H,F)`y⟩ ∈ f``G }"
          unfolding r_def QuotientGroupRel_def by assumption
        then have "F`⟨b⇩1, GroupInv(H,F)`b⇩2⟩ ∈ f``G"
          using R[of "b⇩1" "b⇩2" "H×H" "λx y. F`⟨x, GroupInv(H,F)`y⟩ ∈ f``G"]
          by (simp only:)
        with im_in_ker have step: "h`(F`⟨b⇩1, GroupInv(H,F)`b⇩2⟩) = ?e⇩D" by (simp only:)
        from b2(1) have inv2H: "GroupInv(H,F)`b⇩2 ∈ H"
          using tgt.inverse_in_group tgt.inv_def by (simp only:)
        from b1(1) inv2H have
          "h`(F`⟨b⇩1, GroupInv(H,F)`b⇩2⟩) = R`⟨h`b⇩1, h`(GroupInv(H,F)`b⇩2)⟩"
          using h_hom.f_hom by (simp only:)
        also from b2(1) have "… = R`⟨h`b⇩1, GroupInv(D,R)`(h`b⇩2)⟩"
          using h_hom.f_inv by (simp only:)
        finally have "R`⟨h`b⇩1, GroupInv(D,R)`(h`b⇩2)⟩ = ?e⇩D"
          using step by (simp only:)
        with b1(1) b2(1) have "h`b⇩1 = h`b⇩2"
          using h_hom.tgt.group0_2_L11A[OF apply_type[OF hfun] apply_type[OF hfun]] by blast
        with b1(3) b2(3) show "y = z" by simp
      qed
      from sub dom sv show ?thesis unfolding Pi_def function_def by auto
    qed
    have bhHom: "Homomor(?h, H//r, Q, D, R)"
      unfolding Homomor_def IsMorphism_def
    proof (intro conjI bhfun ballI)
      fix c⇩1 c⇩2 assume c1: "c⇩1 ∈ H//r" and c2: "c⇩2 ∈ H//r"
      show "?h`(Q`⟨c⇩1,c⇩2⟩) = R`⟨?h`c⇩1, ?h`c⇩2⟩"
      proof -
        from c1 obtain b⇩1 where b1: "b⇩1 ∈ H" "c⇩1 = r``{b⇩1}"
          unfolding quotient_def by auto
        from c2 obtain b⇩2 where b2: "b⇩2 ∈ H" "c⇩2 = r``{b⇩2}"
          unfolding quotient_def by auto
        from b1(1) b2(1) have b12H: "F`⟨b⇩1,b⇩2⟩ ∈ H"
          using tgt.group_op_closed tgt.groper_def by (simp only:)
        have "Q`⟨c⇩1,c⇩2⟩ = ProjFun2(H, QuotientGroupRel(H, F, f `` G), F) `⟨c⇩1, c⇩2⟩"
          unfolding Q_def QuotientGroupOp_def by auto
        with b1(2) b2(2) have opQ:
          "Q`⟨c⇩1,c⇩2⟩ = r``{F`⟨b⇩1,b⇩2⟩}"
          using Group_ZF_2_4_L5A[OF target norm] tgt.Group_ZF_2_2_L2[OF equiv_r _ _ b1(1) b2(1), of "ProjFun2(H,r,F)"]
          unfolding Q_def r_def tgt.groper_def by (simp only:)
        from b12H have Fh:"⟨r ``{F ` ⟨b⇩1, b⇩2⟩},h`(F ` ⟨b⇩1, b⇩2⟩)⟩:?h" by blast
        have lhs: "?h`(Q`⟨c⇩1,c⇩2⟩) = h`(F`⟨b⇩1,b⇩2⟩)"
          apply (rule apply_equality[OF _ bhfun, of "Q ` ⟨c⇩1, c⇩2⟩"
              "h`(F`⟨b⇩1,b⇩2⟩)"]) using opQ Fh by (simp only:sym) 
        from b1(1) b2(1) have hprod: "h`(F`⟨b⇩1,b⇩2⟩) = R`⟨h`b⇩1, h`b⇩2⟩"
          using h_hom.f_hom by (simp only:)
        have r:"⋀p. p∈H ⟹ ?h`(r``{p}) = h`p" using apply_equality[OF _ bhfun] by auto
        from b1(2) b1(1) r have rhs1: "?h`c⇩1 = h`b⇩1" by (simp only:)
        from b2(2) b2(1) r have rhs2: "?h`c⇩2 = h`b⇩2" by (simp only:)
        from lhs hprod rhs1 rhs2 show
          "?h`(Q`⟨c⇩1,c⇩2⟩) = R`⟨?h`c⇩1, ?h`c⇩2⟩" by (simp only:)
      qed
    qed
    have factor: "∀b ∈ H. ?h`(q`b) = h`b"
    proof
      fix b assume bH: "b ∈ H"
      from bH have "q`b = r``{b}" unfolding q_def r_def by force
      with bH bhfun show "?h`(q`b) = h`b" using apply_equality[OF _ bhfun, of "r``{b}" "h`b"] by force
    qed
    have uniq: "⋀k. Homomor(k, H//r, Q, D, R) ∧ (∀b ∈ H. k`(q`b) = h`b) ⟹ k = ?h"
    proof -
      fix k assume kH: "Homomor(k, H//r, Q, D, R) ∧ (∀b ∈ H. k`(q`b) = h`b)"
      from kH have kHom: "Homomor(k, H//r, Q, D, R)" and kfactor: "∀b ∈ H. k`(q`b) = h`b"
        by auto
      from kHom have kfun: "k: H//r → D" unfolding Homomor_def by simp
      show "k = ?h"
      proof (rule fun_extension[OF kfun bhfun])
        fix c assume cC: "c ∈ H//r"
        from cC obtain b where bH: "b ∈ H" and cb: "c = r``{b}" unfolding quotient_def by auto
        from bH have "k`c = k`(q`b)" unfolding q_def cb r_def by simp
        also from bH kfactor have "… = h`b" by simp
        also from bH bhfun cb have "… = ?h`c" using apply_equality[OF _ bhfun, of c "h`b"] by force
        finally show "k`c = ?h`c" .
      qed
    qed
    from bhHom factor uniq show
      "∃!hh. Homomor(hh, H//r, Q, D, R) ∧ (∀b ∈ H. hh`(q`b) = h`b)"
      by (intro ex1I[of _ "?h"]) auto
    qed
  qed
qed
